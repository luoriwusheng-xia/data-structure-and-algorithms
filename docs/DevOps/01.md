# 前端部署：SPA & SSR

前端部署模式主要分为 **SPA（单页应用）** 和 **SSR（服务端渲染）** 两大类，每种模式在云平台的配置策略不同。

## 一、SPA 部署

### 1.1 典型项目结构

```
my-spa/
├── dist/                 # 构建产物
│   ├── index.html
│   ├── assets/
│   └── ...
├── public/
├── src/
├── vite.config.ts
└── package.json
```

### 1.2 Vite 构建配置

```typescript
// vite.config.ts
import { defineConfig } from 'vite'
import react from '@vitejs/plugin-react'

export default defineConfig({
  plugins: [react()],
  base: '/', // 部署到子目录时改为 '/subpath/'
  build: {
    outDir: 'dist',
    assetsDir: 'assets',
    // 代码分割策略
    rollupOptions: {
      output: {
        manualChunks: {
          vendor: ['react', 'react-dom'],
          ui: ['antd'], // 或 '@mui/material'
        },
      },
    },
  },
})
```

### 1.3 路由模式处理

SPA 使用 History 路由时，需要服务器将所有请求指向 `index.html`：

```javascript
// vercel.json (Vercel 部署)
{
  "routes": [
    { "handle": "filesystem" },
    { "src": "/(.*)", "dest": "/index.html" }
  ]
}
```

```nginx
# Nginx 配置
server {
    listen 80;
    server_name app.example.com;
    root /var/www/app;
    index index.html;

    location / {
        try_files $uri $uri/ /index.html;
    }

    # 静态资源缓存
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
    }
}
```

### 1.4 GitHub Actions 自动构建

```yaml
# .github/workflows/deploy-spa.yml
name: Deploy SPA

on:
  push:
    branches: [main]

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'

      - name: Install dependencies
        run: npm ci

      - name: Build
        run: npm run build
        env:
          VITE_API_BASE_URL: ${{ secrets.API_BASE_URL }}

      - name: Deploy to Server
        uses: appleboy/scp-action@v0.1.7
        with:
          host: ${{ secrets.SSH_HOST }}
          username: ${{ secrets.SSH_USER }}
          key: ${{ secrets.SSH_KEY }}
          source: 'dist/'
          target: '/var/www/app'
          strip_components: 1
```

## 二、SSR 部署

### 2.1 Next.js SSR 部署

```javascript
// next.config.js
/** @type {import('next').NextConfig} */
const nextConfig = {
  output: 'standalone', // 生成独立部署包，体积更小
  images: {
    unoptimized: true, // 如果使用自托管而非 Vercel
  },
}

module.exports = nextConfig
```

```dockerfile
# Dockerfile (Next.js SSR)
FROM node:20-alpine AS base

# 依赖安装
FROM base AS deps
RUN apk add --no-cache libc6-compat
WORKDIR /app
COPY package.json package-lock.json ./
RUN npm ci

# 构建
FROM base AS builder
WORKDIR /app
COPY --from=deps /app/node_modules ./node_modules
COPY . .
RUN npm run build

# 生产运行
FROM base AS runner
WORKDIR /app
ENV NODE_ENV=production

COPY --from=builder /app/public ./public
COPY --from=builder /app/.next/standalone ./
COPY --from=builder /app/.next/static ./.next/static

EXPOSE 3000
ENV PORT=3000
ENV HOSTNAME="0.0.0.0"

CMD ["node", "server.js"]
```

### 2.2 Nuxt 3 SSR 部署

```typescript
// nuxt.config.ts
export default defineNuxtConfig({
  nitro: {
    preset: 'node-server', // 或 'vercel-edge' / 'cloudflare-pages'
  },
  runtimeConfig: {
    // 服务端私有配置
    apiSecret: process.env.API_SECRET,
    // 客户端公开配置
    public: {
      apiBase: process.env.NUXT_PUBLIC_API_BASE,
    },
  },
})
```

```dockerfile
# Dockerfile (Nuxt 3)
FROM node:20-alpine

WORKDIR /app
COPY package.json ./
RUN npm install

COPY . .
RUN npm run build

EXPOSE 3000
CMD ["node", ".output/server/index.mjs"]
```

### 2.3 SSR 部署到自有服务器

```yaml
# docker-compose.ssr.yml
version: '3.8'

services:
  next-app:
    build:
      context: .
      dockerfile: Dockerfile
    container_name: next-ssr
    restart: unless-stopped
    ports:
      - "3000:3000"
    environment:
      - NODE_ENV=production
      - DATABASE_URL=${DATABASE_URL}
    networks:
      - app-network

  nginx:
    image: nginx:alpine
    container_name: next-nginx
    restart: unless-stopped
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx.conf:/etc/nginx/nginx.conf:ro
      - ./ssl:/etc/nginx/ssl:ro
    depends_on:
      - next-app
    networks:
      - app-network

networks:
  app-network:
    driver: bridge
```

```nginx
# nginx.conf (SSR 反向代理)
upstream nextjs {
    server next-app:3000;
}

server {
    listen 80;
    server_name ssr.example.com;

    location / {
        proxy_pass http://nextjs;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_cache_bypass $http_upgrade;
    }

    # 静态资源直接走 Nginx 缓存
    location /_next/static {
        proxy_pass http://nextjs;
        proxy_cache_valid 200 365d;
        add_header Cache-Control "public, immutable";
    }
}
```

## 三、环境变量管理

```typescript
// 前端：Vite 环境变量 (.env 文件)
// .env.development
VITE_API_BASE_URL=http://localhost:3001
VITE_APP_NAME=MyApp

// .env.production
VITE_API_BASE_URL=https://api.example.com
VITE_APP_NAME=MyApp

// 代码中使用
const apiBase = import.meta.env.VITE_API_BASE_URL
```

```javascript
// 后端/SSR：Next.js runtimeConfig
// 服务端获取
import { getConfig } from 'next/config'
const { serverRuntimeConfig } = getConfig()
const secret = serverRuntimeConfig.apiSecret

// 客户端获取
const { publicRuntimeConfig } = getConfig()
const apiBase = publicRuntimeConfig.apiBase
```

## 四、CDN 与缓存策略

```yaml
# .github/workflows/cdn-cache-purge.yml
name: Purge CDN Cache

on:
  push:
    branches: [main]

jobs:
  purge:
    runs-on: ubuntu-latest
    steps:
      - name: Purge Cloudflare Cache
        run: |
          curl -X POST "https://api.cloudflare.com/client/v4/zones/${{ secrets.CF_ZONE_ID }}/purge_cache" \
            -H "Authorization: Bearer ${{ secrets.CF_API_TOKEN }}" \
            -H "Content-Type: application/json" \
            --data '{"purge_everything":true}'
```

```javascript
// Service Worker 缓存策略 (Vite PWA)
// vite.config.ts
import { VitePWA } from 'vite-plugin-pwa'

export default defineConfig({
  plugins: [
    VitePWA({
      registerType: 'autoUpdate',
      workbox: {
        runtimeCaching: [
          {
            urlPattern: /^https:\/\/api\.example\.com\/.*/,
            handler: 'NetworkFirst',
            options: {
              cacheName: 'api-cache',
              expiration: { maxEntries: 100, maxAgeSeconds: 86400 },
            },
          },
        ],
      },
    }),
  ],
})
```
