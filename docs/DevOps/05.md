## 二、备份策略

### 2.1 自动备份脚本

```bash
#!/bin/bash
# backup.sh

DB_NAME="app_prod"
DB_USER="postgres"
BACKUP_DIR="/var/backups/postgres"
RETENTION_DAYS=30
DATE=$(date +%Y%m%d_%H%M%S)

# 创建备份目录
mkdir -p $BACKUP_DIR

# 全量备份
echo "[$(date)] 开始全量备份..."
pg_dump -h localhost -U $DB_USER -d $DB_NAME \
  --format=custom \
  --file=$BACKUP_DIR/full_$DATE.dump

# 压缩
gzip $BACKUP_DIR/full_$DATE.dump

# 清理旧备份
echo "[$(date)] 清理 $RETENTION_DAYS 天前的备份..."
find $BACKUP_DIR -name "full_*.dump.gz" -mtime +$RETENTION_DAYS -delete

# 上传到对象存储（可选）
# aws s3 cp $BACKUP_DIR/full_$DATE.dump.gz s3://my-backup-bucket/postgres/

echo "[$(date)] 备份完成: full_$DATE.dump.gz"
```

### 2.3 恢复流程

```bash
#!/bin/bash
# restore.sh

BACKUP_FILE=$1
DB_NAME="app_prod"
DB_USER="postgres"

echo "⚠️  即将恢复数据库，当前数据将被覆盖！"
read -p "确认继续? (yes/no): " confirm

if [ "$confirm" != "yes" ]; then
  echo "已取消"
  exit 1
fi

# 终止现有连接
echo "终止现有连接..."
psql -U $DB_USER -c "SELECT pg_terminate_backend(pid) FROM pg_stat_activity WHERE datname = '$DB_NAME' AND pid <> pg_backend_pid();"

# 删除并重建数据库
echo "重建数据库..."
dropdb -U $DB_USER $DB_NAME
createdb -U $DB_USER $DB_NAME

# 恢复
echo "恢复数据..."
if [[ $BACKUP_FILE == *.gz ]]; then
  gunzip -c $BACKUP_FILE | pg_restore -U $DB_USER -d $DB_NAME --verbose
else
  pg_restore -U $DB_USER -d $DB_NAME --verbose $BACKUP_FILE
fi

echo "✅ 恢复完成"
```

## 三、性能监控

### 3.1 慢查询日志

```sql
-- 启用慢查询日志
ALTER SYSTEM SET log_min_duration_statement = '1000';  -- 1秒
ALTER SYSTEM SET log_line_prefix = '%t [%p]: [%l-1] user=%u,db=%d,app=%a,client=%h ';
SELECT pg_reload_conf();

-- 查看慢查询
SELECT
  query,
  calls,
  total_exec_time,
  mean_exec_time,
  max_exec_time,
  rows
FROM pg_stat_statements
ORDER BY mean_exec_time DESC
LIMIT 10;
```

### 3.2 常用监控查询

```sql
-- 当前连接数
SELECT count(*) FROM pg_stat_activity;

-- 长时间运行的查询
SELECT
  pid,
  now() - query_start AS duration,
  state,
  query
FROM pg_stat_activity
WHERE state != 'idle'
  AND now() - query_start > interval '5 minutes'
ORDER BY duration DESC;

-- 表大小
SELECT
  schemaname,
  relname,
  pg_size_pretty(pg_total_relation_size(relid)) AS total_size,
  pg_size_pretty(pg_relation_size(relid)) AS data_size
FROM pg_stat_user_tables
ORDER BY pg_total_relation_size(relid) DESC
LIMIT 20;

-- 索引使用情况
SELECT
  schemaname,
  relname,
  indexrelname,
  idx_scan,
  idx_tup_read,
  idx_tup_fetch
FROM pg_stat_user_indexes
ORDER BY idx_scan DESC
LIMIT 20;

-- 缓存命中率（应 > 99%）
SELECT
  sum(heap_blks_hit) / (sum(heap_blks_hit) + sum(heap_blks_read)) AS cache_hit_ratio
FROM pg_statio_user_tables;
```

### 3.3 Node.js 监控集成

```typescript
// src/utils/db-monitor.ts
import { Pool } from 'pg'

export function setupDbMonitoring(pool: Pool) {
  // 定期收集指标
  setInterval(async () => {
    try {
      const client = await pool.connect()

      // 连接数
      const connResult = await client.query(
        'SELECT count(*) as count FROM pg_stat_activity'
      )

      // 慢查询数
      const slowResult = await client.query(`
        SELECT count(*) as count
        FROM pg_stat_activity
        WHERE state != 'idle'
          AND now() - query_start > interval '1 second'
      `)

      // 缓存命中率
      const cacheResult = await client.query(`
        SELECT
          sum(heap_blks_hit) / NULLIF(sum(heap_blks_hit) + sum(heap_blks_read), 0)
          AS hit_ratio
        FROM pg_statio_user_tables
      `)

      console.log('DB Metrics:', {
        timestamp: new Date().toISOString(),
        connections: parseInt(connResult.rows[0].count),
        slowQueries: parseInt(slowResult.rows[0].count),
        cacheHitRatio: parseFloat(cacheResult.rows[0]?.hit_ratio || 0),
        poolTotal: pool.totalCount,
        poolIdle: pool.idleCount,
        poolWaiting: pool.waitingCount,
      })

      client.release()
    } catch (err) {
      console.error('DB monitoring error:', err)
    }
  }, 30000) // 每 30 秒
}
```

```typescript
// 应用层加密示例
import { createCipheriv, createDecipheriv, randomBytes, scryptSync } from 'crypto'

const ALGORITHM = 'aes-256-gcm'

function getKey(password: string): Buffer {
  return scryptSync(password, 'salt', 32)
}

export function encrypt(text: string, password: string): string {
  const key = getKey(password)
  const iv = randomBytes(16)
  const cipher = createCipheriv(ALGORITHM, key, iv)

  let encrypted = cipher.update(text, 'utf8', 'hex')
  encrypted += cipher.final('hex')

  const authTag = cipher.getAuthTag()

  return `${iv.toString('hex')}:${authTag.toString('hex')}:${encrypted}`
}

export function decrypt(encryptedData: string, password: string): string {
  const key = getKey(password)
  const [ivHex, authTagHex, encrypted] = encryptedData.split(':')

  const iv = Buffer.from(ivHex, 'hex')
  const authTag = Buffer.from(authTagHex, 'hex')

  const decipher = createDecipheriv(ALGORITHM, key, iv)
  decipher.setAuthTag(authTag)

  let decrypted = decipher.update(encrypted, 'hex', 'utf8')
  decrypted += decipher.final('utf8')

  return decrypted
}
```
