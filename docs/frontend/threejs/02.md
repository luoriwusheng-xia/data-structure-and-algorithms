```typescript
// stores/twinStore.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

interface Equipment {
  id: string
  name: string
  type: 'machine' | 'conveyor' | 'sensor' | 'agv'
  status: 'running' | 'idle' | 'warning' | 'error' | 'offline'
  position: { x: number; y: number; z: number }
  metrics: {
    temperature: number
    vibration: number
    oee: number // 设备综合效率
  }
  runningTime: number // 累计运行时间（秒）
}

export const useTwinStore = defineStore('twin', () => {
  const equipments = ref<Equipment[]>([])
  const selectedEquipment = ref<Equipment | null>(null)
  const wsConnected = ref(false)

  // 实时统计
  const stats = computed(() => ({
    total: equipments.value.length,
    running: equipments.value.filter(e => e.status === 'running').length,
    warning: equipments.value.filter(e => e.status === 'warning').length,
    error: equipments.value.filter(e => e.status === 'error').length,
    avgOEE: equipments.value.reduce((s, e) => s + e.metrics.oee, 0) / equipments.value.length
  }))

  // WebSocket 连接
  let ws: WebSocket

  function connect(url: string) {
    ws = new WebSocket(url)

    ws.onopen = () => {
      wsConnected.value = true
      // 请求初始数据
      ws.send(JSON.stringify({ type: 'init', factoryId: 'factory_01' }))
    }

    ws.onmessage = (event) => {
      const payload = JSON.parse(event.data)

      switch (payload.type) {
        case 'snapshot':
          // 初始全量数据
          equipments.value = payload.data
          break
        case 'delta':
          // 增量更新——关键优化
          payload.changes.forEach((change: any) => {
            const equip = equipments.value.find(e => e.id === change.id)
            if (equip) {
              Object.assign(equip, change.updates)
            }
          })
          break
        case 'alert':
          // 告警推送
          handleAlert(payload.alert)
          break
      }
    }

    ws.onclose = () => {
      wsConnected.value = false
      setTimeout(() => connect(url), 5000) // 自动重连
    }
  }

  function handleAlert(alert: any) {
    // 触发浏览器通知
    if (Notification.permission === 'granted') {
      new Notification('设备告警', {
        body: `${alert.equipmentName}: ${alert.message}`,
        icon: '/alert-icon.png'
      })
    }
  }

  return { equipments, selectedEquipment, stats, wsConnected, connect }
})
```

```typescript
// components/FactoryScene.vue
<script setup lang="ts">
import { ref, onMounted, watch } from 'vue'
import * as THREE from 'three'
import { OrbitControls } from 'three/addons/controls/OrbitControls.js'
import { GLTFLoader } from 'three/addons/loaders/GLTFLoader.js'
import { useTwinStore } from '../stores/twinStore'

const container = ref<HTMLElement>()
const store = useTwinStore()

let scene: THREE.Scene
let camera: THREE.PerspectiveCamera
let renderer: THREE.WebGLRenderer
let controls: OrbitControls
let equipmentMeshes = new Map<string, THREE.Mesh>()
let mixer: THREE.AnimationMixer

// 状态颜色映射
const STATUS_COLORS = {
  running: 0x00c853,
  idle: 0xffd600,
  warning: 0xff9100,
  error: 0xff1744,
  offline: 0x757575
}

onMounted(() => {
  initScene()
  loadFactoryModel()
  store.connect('wss://api.factory.com/ws/twin')
})

function initScene() {
  const { clientWidth, clientHeight } = container.value!

  scene = new THREE.Scene()
  scene.background = new THREE.Color(0x0a0a1a)

  // 相机
  camera = new THREE.PerspectiveCamera(45, clientWidth / clientHeight, 0.1, 500)
  camera.position.set(30, 25, 30)

  // 渲染器
  renderer = new THREE.WebGLRenderer({ antialias: true })
  renderer.setSize(clientWidth, clientHeight)
  renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2))
  renderer.shadowMap.enabled = true
  renderer.shadowMap.type = THREE.PCFSoftShadowMap
  container.value!.appendChild(renderer.domElement)

  // 控制器
  controls = new OrbitControls(camera, renderer.domElement)
  controls.enableDamping = true
  controls.maxPolarAngle = Math.PI / 2.1 // 限制俯视角度
  controls.minDistance = 10
  controls.maxDistance = 100

  // 光照
  const ambientLight = new THREE.AmbientLight(0xffffff, 0.4)
  scene.add(ambientLight)

  const dirLight = new THREE.DirectionalLight(0xffffff, 1.5)
  dirLight.position.set(20, 40, 20)
  dirLight.castShadow = true
  dirLight.shadow.mapSize.set(2048, 2048)
  dirLight.shadow.camera.near = 0.1
  dirLight.shadow.camera.far = 100
  dirLight.shadow.camera.left = -50
  dirLight.shadow.camera.right = 50
  dirLight.shadow.camera.top = 50
  dirLight.shadow.camera.bottom = -50
  scene.add(dirLight)

  // 动画循环
  const clock = new THREE.Clock()
  function animate() {
    requestAnimationFrame(animate)
    const delta = clock.getDelta()
    mixer?.update(delta)
    controls.update()
    renderer.render(scene, camera)
  }
  animate()
}

function loadFactoryModel() {
  const loader = new GLTFLoader()
  loader.load('/models/factory.glb', (gltf) => {
    const factory = gltf.scene

    // 遍历模型，为设备位置创建标记
    factory.traverse((child) => {
      if (child.isMesh) {
        child.castShadow = true
        child.receiveShadow = true

        // 如果 mesh 名称匹配设备ID，建立映射
        if (child.name.startsWith('EQ_')) {
          const equipmentId = child.name.replace('EQ_', '')
          equipmentMeshes.set(equipmentId, child)
          child.userData.isEquipment = true
          child.userData.equipmentId = equipmentId
        }
      }
    })

    scene.add(factory)
    mixer = new THREE.AnimationMixer(factory)
    gltf.animations.forEach(clip => mixer.clipAction(clip).play())
  })
}

// 监听设备数据变化，更新 3D 场景
watch(() => store.equipments, (equipments) => {
  equipments.forEach(equip => {
    const mesh = equipmentMeshes.get(equip.id)
    if (!mesh) return

    const color = STATUS_COLORS[equip.status]

    // 更新材质颜色
    if (mesh.material instanceof THREE.MeshStandardMaterial) {
      mesh.material.color.setHex(color)
      mesh.material.emissive.setHex(color)
      mesh.material.emissiveIntensity = equip.status === 'error' ? 0.8 : 0.2
    }

    // 更新用户数据
    mesh.userData.metrics = equip.metrics
  })
}, { deep: true })

// Raycaster 点击选中设备
const raycaster = new THREE.Raycaster()
const mouse = new THREE.Vector2()

function onMouseClick(event: MouseEvent) {
  const rect = renderer.domElement.getBoundingClientRect()
  mouse.x = ((event.clientX - rect.left) / rect.width) * 2 - 1
  mouse.y = -((event.clientY - rect.top) / rect.height) * 2 + 1

  raycaster.setFromCamera(mouse, camera)
  const intersects = raycaster.intersectObjects(scene.children, true)

  for (const intersect of intersects) {
    const obj = intersect.object
    if (obj.userData.isEquipment) {
      store.selectedEquipment = store.equipments.find(
        e => e.id === obj.userData.equipmentId
      ) || null

      // 相机聚焦动画
      focusOnObject(obj)
      break
    }
  }
}

function focusOnObject(object: THREE.Object3D) {
  const targetPosition = new THREE.Vector3()
  object.getWorldPosition(targetPosition)

  // 使用 GSAP 或手动插值实现平滑过渡
  const startPos = camera.position.clone()
  const endPos = targetPosition.clone().add(new THREE.Vector3(5, 5, 5))
  let progress = 0

  function tween() {
    progress += 0.02
    if (progress >= 1) return

    camera.position.lerpVectors(startPos, endPos, progress)
    controls.target.lerp(targetPosition, progress)
    requestAnimationFrame(tween)
  }
  tween()
}
```

## 案例二：电商产品 3D 配置器

### 业务背景

家具、汽车、电子产品等需要让用户在线定制外观。3D 配置器可以：
- 降低退货率（用户看到实物效果）
- 提升客单价（配件连带销售）
- 缩短决策时间

### 核心实现

```typescript
// composables/useProductConfigurator.ts
import { ref, computed } from 'vue'
import * as THREE from 'three'
import { GLTFLoader } from 'three/addons/loaders/GLTFLoader.js'

interface PartConfig {
  name: string          // 如 "seat", "backrest", "legs"
  currentOption: string // 如 "leather_red"
  availableOptions: {
    id: string
    name: string
    color?: string
    texture?: string
    material?: string
    price: number
  }[]
}

export function useProductConfigurator(containerRef: Ref<HTMLElement>) {
  const scene = new THREE.Scene()
  const parts = ref<PartConfig[]>([])
  const totalPrice = computed(() => {
    return parts.value.reduce((sum, part) => {
      const option = part.availableOptions.find(o => o.id === part.currentOption)
      return sum + (option?.price || 0)
    }, 1000) // 基础价格
  })

  let renderer: THREE.WebGLRenderer
  let camera: THREE.PerspectiveCamera
  let model: THREE.Group
  const partMeshes = new Map<string, THREE.Mesh>()
  const textureLoader = new THREE.TextureLoader()

  // 材质缓存
  const materialCache = new Map<string, THREE.Material>()

  async function init() {
    const { clientWidth, clientHeight } = containerRef.value

    camera = new THREE.PerspectiveCamera(45, clientWidth / clientHeight, 0.1, 100)
    camera.position.set(3, 2, 4)

    renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true })
    renderer.setSize(clientWidth, clientHeight)
    renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2))
    renderer.toneMapping = THREE.ACESFilmicToneMapping
    renderer.toneMappingExposure = 1.2
    renderer.shadowMap.enabled = true
    containerRef.value.appendChild(renderer.domElement)

    // 影棚灯光
    setupStudioLighting()

    // 加载模型
    await loadModel('/models/chair.glb')

    // 自动旋转展示
    autoRotate()
  }

  function setupStudioLighting() {
    // 主光
    const keyLight = new THREE.DirectionalLight(0xffffff, 2)
    keyLight.position.set(5, 10, 5)
    keyLight.castShadow = true
    scene.add(keyLight)

    // 补光
    const fillLight = new THREE.DirectionalLight(0xe8f4ff, 0.8)
    fillLight.position.set(-5, 5, 5)
    scene.add(fillLight)

    // 轮廓光
    const rimLight = new THREE.DirectionalLight(0xffffff, 1)
    rimLight.position.set(0, 5, -5)
    scene.add(rimLight)

    // 环境光
    scene.add(new THREE.AmbientLight(0xffffff, 0.3))

    // HDR 环境贴图（反射用）
    new THREE.TextureLoader().load('/env/studio.jpg', (texture) => {
      texture.mapping = THREE.EquirectangularReflectionMapping
      scene.environment = texture
    })
  }

  async function loadModel(url: string) {
    const loader = new GLTFLoader()
    const gltf = await loader.loadAsync(url)
    model = gltf.scene

    model.traverse((child) => {
      if (child.isMesh) {
        child.castShadow = true
        child.receiveShadow = true

        // 按命名约定识别部件
        const partName = child.name.split('_')[0]
        if (!partMeshes.has(partName)) {
          partMeshes.set(partName, child)
        }
      }
    })

    // 居中模型
    const box = new THREE.Box3().setFromObject(model)
    const center = box.getCenter(new THREE.Vector3())
    model.position.sub(center)
    model.position.y += box.getSize(new THREE.Vector3()).y / 2

    scene.add(model)
  }

  function updatePartMaterial(partName: string, optionId: string) {
    const mesh = partMeshes.get(partName)
    if (!mesh) return

    const part = parts.value.find(p => p.name === partName)
    const option = part?.availableOptions.find(o => o.id === optionId)
    if (!option) return

    const cacheKey = `${optionId}_${option.texture || option.color}`

    if (!materialCache.has(cacheKey)) {
      const material = new THREE.MeshStandardMaterial({
        roughness: option.material === 'leather' ? 0.6 : 0.3,
        metalness: option.material === 'metal' ? 0.8 : 0.1
      })

      if (option.color) {
        material.color.set(option.color)
      }

      if (option.texture) {
        const texture = textureLoader.load(option.texture)
        texture.colorSpace = THREE.SRGBColorSpace
        material.map = texture
      }

      materialCache.set(cacheKey, material)
    }

    mesh.material = materialCache.get(cacheKey)!

    // 更新配置
    const partIdx = parts.value.findIndex(p => p.name === partName)
    if (partIdx >= 0) {
      parts.value[partIdx].currentOption = optionId
    }

    // URL 同步
    syncUrl()
  }

  function syncUrl() {
    const params = new URLSearchParams()
    parts.value.forEach(part => {
      params.set(part.name, part.currentOption)
    })
    window.history.replaceState({}, '', `?${params.toString()}`)
  }

  function loadFromUrl() {
    const params = new URLSearchParams(window.location.search)
    params.forEach((value, key) => {
      updatePartMaterial(key, value)
    })
  }

  function generateSnapshot(): string {
    renderer.render(scene, camera)
    return renderer.domElement.toDataURL('image/png', 0.9)
  }

  function autoRotate() {
    function animate() {
      requestAnimationFrame(animate)
      if (model) {
        model.rotation.y += 0.002
      }
      renderer.render(scene, camera)
    }
    animate()
  }

  // AR 预览
  async function enterAR() {
    if (!navigator.xr) {
      alert('您的设备不支持 AR')
      return
    }

    const session = await navigator.xr.requestSession('immersive-ar', {
      requiredFeatures: ['hit-test']
    })

    renderer.xr.enabled = true
    await renderer.xr.setSession(session)

    // 放置模型
    const controller = renderer.xr.getController(0)
    controller.addEventListener('select', () => {
      // 用户点击放置位置
    })
  }

  return {
    parts,
    totalPrice,
    init,
    updatePartMaterial,
    loadFromUrl,
    generateSnapshot,
    enterAR
  }
}
```

## 案例三：3D 数据可视化大屏

### 业务背景

智慧城市、交通监控、网络拓扑等场景，数据具有空间属性，2D 图表无法表达三维关系。

### 核心实现

```typescript
// 3D 城市建筑热力图
import * as THREE from 'three'
import { OrbitControls } from 'three/addons/controls/OrbitControls.js'

interface BuildingData {
  id: string
  lat: number
  lng: number
  height: number
  floors: number
  energy: number      // 能耗
  occupancy: number   // 入住率
  alertLevel: 0 | 1 | 2 | 3 // 0=正常 1=注意 2=警告 3=紧急
}

class CityHeatmap3D {
  scene: THREE.Scene
  renderer: THREE.WebGLRenderer
  camera: THREE.PerspectiveCamera
  buildings: THREE.InstancedMesh
  data: BuildingData[]

  constructor(container: HTMLElement, data: BuildingData[]) {
    this.data = data

    // 场景
    this.scene = new THREE.Scene()
    this.scene.background = new THREE.Color(0x050510)
    this.scene.fog = new THREE.FogExp2(0x050510, 0.002)

    // 相机
    this.camera = new THREE.PerspectiveCamera(
      60,
      container.clientWidth / container.clientHeight,
      0.1,
      2000
    )
    this.camera.position.set(0, 200, 300)

    // 渲染器
    this.renderer = new THREE.WebGLRenderer({ antialias: true })
    this.renderer.setSize(container.clientWidth, container.clientHeight)
    this.renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2))
    container.appendChild(this.renderer.domElement)

    // 控制器
    const controls = new OrbitControls(this.camera, this.renderer.domElement)
    controls.enableDamping = true
    controls.maxPolarAngle = Math.PI / 2.2
    controls.minDistance = 50
    controls.maxDistance = 800

    this.createBuildings()
    this.createGround()
    this.createRoads()
    this.animate()
  }

  createBuildings() {
    const geometry = new THREE.BoxGeometry(1, 1, 1)
    // 将几何体原点移到底部
    geometry.translate(0, 0.5, 0)

    const material = new THREE.MeshStandardMaterial({
      roughness: 0.4,
      metalness: 0.6
    })

    this.buildings = new THREE.InstancedMesh(geometry, material, this.data.length)
    this.buildings.castShadow = true
    this.buildings.receiveShadow = true

    const dummy = new THREE.Object3D()
    const color = new THREE.Color()

    this.data.forEach((building, i) => {
      // 经纬度转平面坐标（简化墨卡托投影）
      const x = (building.lng - 116.4) * 10000
      const z = -(building.lat - 39.9) * 10000

      dummy.position.set(x, 0, z)
      dummy.scale.set(8, building.height, 8)
      dummy.updateMatrix()
      this.buildings.setMatrixAt(i, dummy.matrix)

      // 颜色：根据 alertLevel + energy 综合计算
      const hue = building.alertLevel === 0
        ? 0.3 + (1 - building.energy / 100) * 0.2  // 绿色系
        : building.alertLevel === 1
          ? 0.15 // 黄色
          : building.alertLevel === 2
            ? 0.08 // 橙色
            : 0.0  // 红色

      const lightness = 0.3 + (building.occupancy / 100) * 0.4
      color.setHSL(hue, 0.8, lightness)
      this.buildings.setColorAt(i, color)
    })

    this.buildings.instanceMatrix.needsUpdate = true
    this.buildings.instanceColor!.needsUpdate = true
    this.scene.add(this.buildings)
  }

  createGround() {
    const geometry = new THREE.PlaneGeometry(2000, 2000)
    const material = new THREE.MeshStandardMaterial({
      color: 0x0a0a1a,
      roughness: 0.8,
      metalness: 0.2
    })
    const ground = new THREE.Mesh(geometry, material)
    ground.rotation.x = -Math.PI / 2
    this.scene.add(ground)

    // 网格辅助
    const grid = new THREE.GridHelper(2000, 100, 0x1a1a3e, 0x0f0f28)
    this.scene.add(grid)
  }

  createRoads() {
    // 根据道路数据绘制发光道路线
    const roadMaterial = new THREE.LineBasicMaterial({
      color: 0x4fc3f7,
      transparent: true,
      opacity: 0.6
    })

    // 示例道路
    const roadPoints = [
      new THREE.Vector3(-500, 0.5, 0),
      new THREE.Vector3(500, 0.5, 0)
    ]
    const roadGeometry = new THREE.BufferGeometry().setFromPoints(roadPoints)
    const road = new THREE.Line(roadGeometry, roadMaterial)
    this.scene.add(road)
  }

  // 动态更新数据
  updateData(newData: BuildingData[]) {
    const color = new THREE.Color()

    newData.forEach((building, i) => {
      const hue = building.alertLevel === 0
        ? 0.3 + (1 - building.energy / 100) * 0.2
        : building.alertLevel === 1 ? 0.15
          : building.alertLevel === 2 ? 0.08 : 0.0

      const lightness = 0.3 + (building.occupancy / 100) * 0.4
      color.setHSL(hue, 0.8, lightness)
      this.buildings.setColorAt(i, color)
    })

    this.buildings.instanceColor!.needsUpdate = true
  }

  animate() {
    requestAnimationFrame(() => this.animate())
    this.renderer.render(this.scene, this.camera)
```

## 案例四：实时监控系统

### 核心架构

```typescript
// 流式数据驱动的 3D 粒子系统
// 场景：监控全球网络流量，用粒子流表示数据包传输

import * as THREE from 'three'
import { pass, texture, uniform, vec2, float, sin, cos, time } from 'three/tsl'

interface FlowData {
  source: { lat: number; lng: number }
  target: { lat: number; lng: number }
  volume: number      // 流量大小
  latency: number     // 延迟（ms）
  packets: number     // 包数量
}

class NetworkFlowVisualization {
  scene: THREE.Scene
  renderer: THREE.WebGLRenderer
  flowLines: THREE.LineSegments
  flowParticles: THREE.Points
  data: FlowData[]

  constructor(container: HTMLElement) {
    this.scene = new THREE.Scene()

    this.renderer = new THREE.WebGLRenderer({ antialias: true })
    this.renderer.setSize(container.clientWidth, container.clientHeight)
    container.appendChild(this.renderer.domElement)

    this.createGlobe()
  }

  createGlobe() {
    // 地球
    const globeGeometry = new THREE.SphereGeometry(100, 64, 64)
    const globeMaterial = new THREE.MeshStandardMaterial({
      color: 0x1a1a3e,
      roughness: 0.8,
      metalness: 0.3,
      wireframe: true
    })
    const globe = new THREE.Mesh(globeGeometry, globeMaterial)
    this.scene.add(globe)

    // 地球表面（点阵）
    const dotCount = 5000
    const dotGeometry = new THREE.BufferGeometry()
    const positions = []

    for (let i = 0; i < dotCount; i++) {
      const phi = Math.acos(-1 + (2 * i) / dotCount)
      const theta = Math.sqrt(dotCount * Math.PI) * phi

      const x = 100 * Math.cos(theta) * Math.sin(phi)
      const y = 100 * Math.sin(theta) * Math.sin(phi)
      const z = 100 * Math.cos(phi)

      positions.push(x, y, z)
    }

    dotGeometry.setAttribute('position', new THREE.Float32BufferAttribute(positions, 3))

    const dotMaterial = new THREE.PointsMaterial({
      color: 0x4fc3f7,
      size: 0.8,
      transparent: true,
      opacity: 0.6
    })

    this.scene.add(new THREE.Points(dotGeometry, dotMaterial))
  }

  // 创建流量弧线
  createFlowArcs(data: FlowData[]) {
    const linePositions: number[] = []
    const lineColors: number[] = []

    data.forEach(flow => {
      const start = this.latLngToVector3(flow.source.lat, flow.source.lng, 100)
      const end = this.latLngToVector3(flow.target.lat, flow.target.lng, 100)

      // 贝塞尔曲线中点（抬高形成弧线）
      const mid = new THREE.Vector3().addVectors(start, end).multiplyScalar(0.5)
      mid.normalize().multiplyScalar(150)

      // 生成弧线点
      const curve = new THREE.QuadraticBezierCurve3(start, mid, end)
      const points = curve.getPoints(50)

      // 颜色：根据延迟决定（绿色=低延迟 红色=高延迟）
      const t = Math.min(flow.latency / 300, 1)
      const color = new THREE.Color().setHSL(0.3 * (1 - t), 1, 0.5)

      points.forEach((point, i) => {
        if (i < points.length - 1) {
          linePositions.push(point.x, point.y, point.z)
          linePositions.push(points[i + 1].x, points[i + 1].y, points[i + 1].z)
          lineColors.push(color.r, color.g, color.b)
          lineColors.push(color.r, color.g, color.b)
        }
      })
    })

    const geometry = new THREE.BufferGeometry()
    geometry.setAttribute('position', new THREE.Float32BufferAttribute(linePositions, 3))
    geometry.setAttribute('color', new THREE.Float32BufferAttribute(lineColors, 3))

    const material = new THREE.LineBasicMaterial({
      vertexColors: true,
      transparent: true,
      opacity: 0.8,
      blending: THREE.AdditiveBlending
    })

    this.flowLines = new THREE.LineSegments(geometry, material)
    this.scene.add(this.flowLines)
  }

  latLngToVector3(lat: number, lng: number, radius: number): THREE.Vector3 {
    const phi = (90 - lat) * (Math.PI / 180)
    const theta = (lng + 180) * (Math.PI / 180)

    return new THREE.Vector3(
      -radius * Math.sin(phi) * Math.cos(theta),
      radius * Math.cos(phi),
      radius * Math.sin(phi) * Math.sin(theta)
    )
  }

  // 通过 WebSocket 接收实时数据
  connectStream(url: string) {
    const ws = new WebSocket(url)

    ws.onmessage = (event) => {
      const flows: FlowData[] = JSON.parse(event.data)

      // 清除旧线
      if (this.flowLines) {
        this.scene.remove(this.flowLines)
        this.flowLines.geometry.dispose()
        ;(this.flowLines.material as THREE.Material).dispose()
      }

      // 创建新线
      this.createFlowArcs(flows)
    }
  }
}
```

```typescript
// utils/performanceMonitor.ts
import Stats from 'three/addons/libs/stats.module.js'

interface PerformanceMetrics {
  fps: number
  drawCalls: number
  triangles: number
  geometries: number
  textures: number
  memory: {
    geometries: number
    textures: number
  }

class PerformanceMonitor {
  stats: Stats
  renderer: THREE.WebGLRenderer
  metrics: PerformanceMetrics
  onReport: ((metrics: PerformanceMetrics) => void) | null = null

  constructor(renderer: THREE.WebGLRenderer, showPanel = true) {
    this.renderer = renderer
    this.stats = new Stats()
    this.metrics = {
      fps: 0,
      drawCalls: 0,
      triangles: 0,
      geometries: 0,
      textures: 0,
      memory: { geometries: 0, textures: 0 }
    }

    if (showPanel) {
      this.stats.dom.style.cssText = 'position:absolute;top:0;left:0;'
      document.body.appendChild(this.stats.dom)
    }

    this.startMonitoring()
  }

  startMonitoring() {
    let frameCount = 0
    let lastTime = performance.now()

    const check = () => {
      this.stats.update()

      frameCount++
      const now = performance.now()

      if (now - lastTime >= 1000) {
        this.metrics.fps = Math.round((frameCount * 1000) / (now - lastTime))
        this.metrics.drawCalls = this.renderer.info.render.calls
        this.metrics.triangles = this.renderer.info.render.triangles
        this.metrics.geometries = this.renderer.info.memory.geometries
        this.metrics.textures = this.renderer.info.memory.textures

        // 上报
        this.onReport?.(this.metrics)

        // 自动降级策略
        if (this.metrics.fps < 30) {
          this.applyDegradation()
        }

        frameCount = 0
        lastTime = now
      }

      requestAnimationFrame(check)
    }
    check()
  }

  applyDegradation() {
    console.warn('FPS 过低，应用降级策略')

    // 1. 降低像素比
```

```typescript
// 3D 场景内的用户行为追踪
class SceneAnalytics {
  scene: THREE.Scene
  camera: THREE.Camera
  raycaster: THREE.Raycaster

  // 记录用户看了哪些物体、看了多久
  gazeMap = new Map<string, { enterTime: number; totalTime: number; count: number }>()
  currentGazedObject: string | null = null
  gazeStartTime: number = 0

  constructor(scene: THREE.Scene, camera: THREE.Camera) {
    this.scene = scene
    this.camera = camera
    this.raycaster = new THREE.Raycaster()

    // 每 100ms 检测视线方向
    setInterval(() => this.checkGaze(), 100)
  }

  checkGaze() {
    // 从相机中心发射射线
    this.raycaster.setFromCamera(new THREE.Vector2(0, 0), this.camera)
    const intersects = this.raycaster.intersectObjects(this.scene.children, true)

    const gazedObject = intersects.length > 0
      ? (intersects[0].object.name || intersects[0].object.uuid)
      : null

    if (gazedObject !== this.currentGazedObject) {
      // 离开上一个物体
      if (this.currentGazedObject) {
        const gaze = this.gazeMap.get(this.currentGazedObject)!
        gaze.totalTime += performance.now() - this.gazeStartTime
      }

      // 进入新物体
      if (gazedObject) {
        if (!this.gazeMap.has(gazedObject)) {
          this.gazeMap.set(gazedObject, { enterTime: performance.now(), totalTime: 0, count: 0 })
        }
        this.gazeMap.get(gazedObject)!.count++
        this.gazeStartTime = performance.now()
      }

      this.currentGazedObject = gazedObject
    }
  }

  // 上报分析数据
  getReport() {
    return Array.from(this.gazeMap.entries()).map(([id, data]) => ({
      objectId: id,
      viewCount: data.count,
      totalViewTime: data.totalTime,
      avgViewTime: data.totalTime / data.count
    }))
  }
}
```
