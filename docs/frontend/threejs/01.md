### 1.2 核心三要素：Scene + Camera + Renderer

Three.js 的一切渲染都围绕三个对象展开，理解它们的职责是后续所有内容的基础。

```javascript
import * as THREE from 'three'

// 1. Scene（场景）—— 3D 世界的容器
const scene = new THREE.Scene()
scene.background = new THREE.Color(0x1a1a2e)

// 2. Camera（相机）—— 观察世界的眼睛
const camera = new THREE.PerspectiveCamera(
  75,                               // FOV：视野角度
  window.innerWidth / window.innerHeight, // 宽高比
  0.1,                              // 近裁剪面
  1000                              // 远裁剪面
)
camera.position.z = 5

// 3. Renderer（渲染器）—— 把 3D 画到屏幕上
const renderer = new THREE.WebGLRenderer({ antialias: true })
renderer.setSize(window.innerWidth, window.innerHeight)
renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2)) // 视网膜屏适配
renderer.shadowMap.enabled = true // 开启阴影
document.body.appendChild(renderer.domElement)
```

**为什么设置 `pixelRatio` 上限为 2？**

高 DPR 屏幕（如 iPhone 的 3x）如果按实际 DPR 渲染，帧率会暴跌。限制在 2 是性能与清晰度的平衡点，这是企业级项目的标配做法。

### 1.3 第一个完整场景：带阴影的旋转立方体

```javascript
import * as THREE from 'three'
import { OrbitControls } from 'three/addons/controls/OrbitControls.js'

// === 场景初始化 ===
const scene = new THREE.Scene()
scene.background = new THREE.Color(0x1a1a2e)

const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 100)
camera.position.set(3, 3, 5)

const renderer = new THREE.WebGLRenderer({ antialias: true })
renderer.setSize(window.innerWidth, window.innerHeight)
renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2))
renderer.shadowMap.enabled = true
renderer.shadowMap.type = THREE.PCFSoftShadowMap // 软阴影
document.body.appendChild(renderer.domElement)

// === 控制器 ===
const controls = new OrbitControls(camera, renderer.domElement)
controls.enableDamping = true // 惯性阻尼，体验更丝滑

// === 几何体 ===
const geometry = new THREE.BoxGeometry(1, 1, 1)
const material = new THREE.MeshStandardMaterial({
  color: 0x4ecca3,
  roughness: 0.3,
  metalness: 0.1
})
const cube = new THREE.Mesh(geometry, material)
cube.castShadow = true
cube.receiveShadow = true
scene.add(cube)

// === 地面 ===
const planeGeometry = new THREE.PlaneGeometry(10, 10)
const planeMaterial = new THREE.MeshStandardMaterial({ color: 0x232946 })
const plane = new THREE.Mesh(planeGeometry, planeMaterial)
plane.rotation.x = -Math.PI / 2
plane.position.y = -0.5
plane.receiveShadow = true
scene.add(plane)

// === 光照系统 ===
const ambientLight = new THREE.AmbientLight(0xffffff, 0.4)
scene.add(ambientLight)

const dirLight = new THREE.DirectionalLight(0xffffff, 2)
dirLight.position.set(5, 10, 5)
dirLight.castShadow = true
dirLight.shadow.mapSize.width = 1024
dirLight.shadow.mapSize.height = 1024
scene.add(dirLight)

// === 动画循环 ===
function animate() {
  requestAnimationFrame(animate)

  cube.rotation.x += 0.01
  cube.rotation.y += 0.01

  controls.update()
  renderer.render(scene, camera)
}
animate()

// === 响应式 ===
window.addEventListener('resize', () => {
  camera.aspect = window.innerWidth / window.innerHeight
  camera.updateProjectionMatrix()
  renderer.setSize(window.innerWidth, window.innerHeight)
})
```

### 1.5 材质（Material）体系

| 材质 | 是否受光 | 适用场景 |
|------|---------|---------|
| `MeshBasicMaterial` | 否 | UI 元素、全息投影、不受光物体 |
| `MeshLambertMaterial` | 是（漫反射） | 性能敏感场景、远处物体 |
| `MeshPhongMaterial` | 是（高光） | 需要镜面反射的物体 |
| `MeshStandardMaterial` | 是（PBR） | **最常用**，金属度/粗糙度工作流 |
| `MeshPhysicalMaterial` | 是（高级PBR） | 玻璃、车漆、透光材质 |
| `MeshNormalMaterial` | - | 调试法线、特效 |
| `MeshDepthMaterial` | - | 深度图、后处理 |
| `MeshDistanceMaterial` | - | 点光源阴影 |
| `ShaderMaterial` | 自定义 | 自定义 Shader |
| `NodeMaterial` | 自定义 | **v0.184 推荐**，TSL 节点材质 |

```javascript
// PBR 材质示例（企业项目标配）
const pbrMaterial = new THREE.MeshStandardMaterial({
  color: 0x8B4513,
  roughness: 0.4,      // 粗糙度 0=镜面 1= matte
  metalness: 0.8,      // 金属度
  envMapIntensity: 1.0 // 环境贴图强度
})

// 带贴图的 PBR
const textureLoader = new THREE.TextureLoader()
const material = new THREE.MeshStandardMaterial({
  map: textureLoader.load('diffuse.jpg'),         // 漫反射贴图
  roughnessMap: textureLoader.load('roughness.jpg'), // 粗糙度贴图
  metalnessMap: textureLoader.load('metalness.jpg'), // 金属度贴图
  normalMap: textureLoader.load('normal.jpg'),     // 法线贴图
  aoMap: textureLoader.load('ao.jpg'),             // 环境光遮蔽
})
```

### 2.2 GLTF/GLB 模型加载（企业标准格式）

GLB 是 Three.js 项目的事实标准，它包含模型、材质、动画、骨骼等所有数据。

```javascript
import { GLTFLoader } from 'three/addons/loaders/GLTFLoader.js'
import { DRACOLoader } from 'three/addons/loaders/DRACOLoader.js'

const dracoLoader = new DRACOLoader()
dracoLoader.setDecoderPath('https://unpkg.com/three@0.184.0/examples/jsm/libs/draco/')

const gltfLoader = new GLTFLoader()
gltfLoader.setDRACOLoader(dracoLoader)

// 加载带压缩的模型（企业模型通常很大，Draco 压缩是必须的）
gltfLoader.load('model.glb', (gltf) => {
  const model = gltf.scene

  // 统一处理模型中的材质
  model.traverse((child) => {
    if (child.isMesh) {
      child.castShadow = true
      child.receiveShadow = true

      // 开启透明材质的双面渲染
      if (child.material.transparent) {
        child.material.side = THREE.DoubleSide
      }
    }
  })

  scene.add(model)

  // 播放动画
  const mixer = new THREE.AnimationMixer(model)
  gltf.animations.forEach((clip) => {
    mixer.clipAction(clip).play()
  })
})
```

> **企业建议**：所有生产环境模型都必须经过 Draco/KTX2 压缩。一个 50MB 的模型压缩后可能只有 2MB。

### 2.4 后期处理（Post-Processing）

v0.184 引入了新的后处理管道，基于 `RenderPipeline`：

```javascript
import { EffectComposer } from 'three/addons/postprocessing/EffectComposer.js'

import { UnrealBloomPass } from 'three/addons/postprocessing/UnrealBloomPass.js'

const composer = new EffectComposer(renderer)

const renderPass = new RenderPass(scene, camera)
composer.addPass(renderPass)

// 泛光效果——数字孪生/可视化大屏常用
const bloomPass = new UnrealBloomPass(
  new THREE.Vector2(window.innerWidth, window.innerHeight),
  0.5,  // strength
  0.4,  // radius
  0.85  // threshold
)
composer.addPass(bloomPass)

const outputPass = new OutputPass()
composer.addPass(outputPass)

// 动画循环改为 composer

  requestAnimationFrame(animate)
  controls.update()
  composer.render()
}
```

## 三、精通篇

### 3.1 性能优化：从 60fps 到 120fps

性能优化是企业 Three.js 项目的生死线。以下是经过生产验证的优化清单：

#### 3.1.1 InstancedMesh——批量渲染相同几何体

场景中有 1000 棵树？不要用 1000 个 Mesh，用 1 个 InstancedMesh。

```javascript
const geometry = new THREE.BoxGeometry(1, 1, 1)
const material = new THREE.MeshStandardMaterial({ color: 0x4ecca3 })

// 10000 个实例
const count = 10000
const instancedMesh = new THREE.InstancedMesh(geometry, material, count)

const dummy = new THREE.Object3D()
for (let i = 0; i < count; i++) {
  dummy.position.set(
    (Math.random() - 0.5) * 100,
    (Math.random() - 0.5) * 100,
    (Math.random() - 0.5) * 100
  )
  dummy.rotation.set(Math.random() * Math.PI, Math.random() * Math.PI, 0)
  dummy.updateMatrix()
  instancedMesh.setMatrixAt(i, dummy.matrix)
}

scene.add(instancedMesh)
```

> **效果**：10000 个独立 Mesh 可能只有 5fps，InstancedMesh 可以轻松跑到 120fps。

#### 3.1.4 内存管理

```javascript
// Three.js 对象需要手动释放，否则会导致 GPU 内存泄漏
function disposeObject(obj) {
  if (obj.geometry) obj.geometry.dispose()

  if (obj.material) {
    if (Array.isArray(obj.material)) {
      obj.material.forEach(m => disposeMaterial(m))
    } else {
      disposeMaterial(obj.material)
    }
  }

  if (obj.parent) obj.parent.remove(obj)
}

function disposeMaterial(material) {
  Object.keys(material).forEach(key => {
    const value = material[key]
    if (value && value.isTexture) {
      value.dispose()
    }
  })
  material.dispose()
}
```

### 3.2 TSL：Three.js Shading Language（v0.184 核心新特性）

TSL 是 Three.js v0.184 最重要的新特性。它用 JavaScript 语法编写 Shader，自动编译为 WebGL/WebGPU。

```javascript
import * as THREE from 'three/webgpu'
import { vec3, color, uniform, mix, uv, sin, time, positionLocal } from 'three/tsl'

// 创建一个 pulsating 材质
const material = new THREE.MeshStandardNodeMaterial()

// uniform 可以在运行时更新
const baseColor = uniform(color('#4ecca3'))
const pulseSpeed = uniform(2.0)

// 用 TSL 定义颜色节点
material.colorNode = mix(
  baseColor,
  color('#ff6b6b'),
  sin(positionLocal.y.mul(10).add(time.mul(pulseSpeed))).mul(0.5).add(0.5)
)

// 创建网格
const mesh = new THREE.Mesh(new THREE.SphereGeometry(1, 64, 64), material)
scene.add(mesh)
```

**TSL 解决了什么问题？**

| 问题 | 传统 ShaderMaterial | TSL NodeMaterial |
|------|-------------------|-----------------|
| 语法 | GLSL 字符串，无类型检查 | JavaScript，有类型提示 |
| 跨平台 | 只支持 WebGL | WebGL/WebGPU 自动适配 |
| 可维护性 | 拼接字符串，容易出错 | 组合节点，可复用 |
| 与材质集成 | 需要重写全部光照 | 只改需要改的部分 |

### 3.3 WebGPU 渲染器

WebGPU 是 WebGL 的继任者，提供更低开销的 GPU 计算和渲染。

```javascript
import * as THREE from 'three/webgpu'

// 创建 WebGPU 渲染器
const renderer = new THREE.WebGPURenderer({ antialias: true })
renderer.setSize(window.innerWidth, window.innerHeight)
renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2))
document.body.appendChild(renderer.domElement)

// WebGPU 需要异步初始化
await renderer.init()

// 动画循环
renderer.setAnimationLoop(() => {
  controls.update()
  renderer.render(scene, camera)
})
```

**WebGPU vs WebGL 选择指南：**

- **用 WebGL**：需要兼容 Safari < 17、老旧设备、稳定优先
- **用 WebGPU**：需要计算 Shader、大量实例化、追求极致性能、目标用户用 Chrome/Edge

> v0.184 中，同一个场景代码可以通过切换 `three` 和 `three/webgpu` 导入来切换渲染后端。

### 3.4 自定义 BufferGeometry

当内置几何体不够用时，直接操作顶点数据：

```javascript
// 创建一个动态波浪平面
const width = 100
const depth = 100
const geometry = new THREE.BufferGeometry()

const positions = []
const indices = []

for (let x = 0; x <= width; x++) {
  for (let z = 0; z <= depth; z++) {
    positions.push(x - width / 2, 0, z - depth / 2)
  }
}

for (let x = 0; x < width; x++) {
  for (let z = 0; z < depth; z++) {
    const a = x * (depth + 1) + z
    const b = a + depth + 1
    indices.push(a, b, a + 1)
    indices.push(b, b + 1, a + 1)
  }
}

geometry.setAttribute('position', new THREE.Float32BufferAttribute(positions, 3))
geometry.setIndex(indices)
geometry.computeVertexNormals()

const mesh = new THREE.Mesh(geometry, material)
scene.add(mesh)

// 动态更新顶点
const positionAttribute = geometry.getAttribute('position')
function animateWave() {
  const time = performance.now() * 0.001
  for (let i = 0; i < positionAttribute.count; i++) {
    const x = positionAttribute.getX(i)
    const z = positionAttribute.getZ(i)
    const y = Math.sin(x * 0.5 + time) * Math.cos(z * 0.3 + time) * 2
    positionAttribute.setY(i, y)
  }
  positionAttribute.needsUpdate = true
  geometry.computeVertexNormals()
}
```

### 3.5 物理引擎集成

```javascript
// 使用 Rapier（Rust 编写，性能最好）
import RAPIER from '@dimforge/rapier3d-compat'

await RAPIER.init()

function animate() {
  world.step()

  // 同步 Three.js 对象位置

  mesh.position.set(position.x, position.y, position.z)

  renderer.render(scene, camera)
}
```

## 四、企业落地篇

> **核心观点**：Three.js 在企业中的价值不是"画一个好看的 3D 场景"，而是**用 3D 交互解决业务问题**。

### 4.1 与前端框架集成（Vue 3 / React）

**Vue 3 组合式函数封装：**

```typescript
// composables/useThree.ts
import { ref, onMounted, onUnmounted, shallowRef } from 'vue'
import * as THREE from 'three'
import { OrbitControls } from 'three/addons/controls/OrbitControls.js'

export function useThree(containerRef: Ref<HTMLElement | null>) {
  const scene = shallowRef(new THREE.Scene())
  const camera = shallowRef<THREE.PerspectiveCamera>()
  const renderer = shallowRef<THREE.WebGLRenderer>()
  const controls = shallowRef<OrbitControls>()

  let animationId: number

  onMounted(() => {
    if (!containerRef.value) return

    const { clientWidth, clientHeight } = containerRef.value

    camera.value = new THREE.PerspectiveCamera(75, clientWidth / clientHeight, 0.1, 1000)
    renderer.value = new THREE.WebGLRenderer({ antialias: true, alpha: true })
    renderer.value.setSize(clientWidth, clientHeight)
    renderer.value.setPixelRatio(Math.min(window.devicePixelRatio, 2))
    containerRef.value.appendChild(renderer.value.domElement)

    controls.value = new OrbitControls(camera.value, renderer.value.domElement)
    controls.value.enableDamping = true

    const animate = () => {
      animationId = requestAnimationFrame(animate)
      controls.value?.update()
      renderer.value?.render(scene.value, camera.value!)
    }
    animate()
  })

  onUnmounted(() => {
    cancelAnimationFrame(animationId)
    renderer.value?.dispose()
  })

  return { scene, camera, renderer, controls }
}
```

```javascript
// 3D 热力图示例：展示数据中心机房温度分布
function createHeatmap(data) {
  const instancedMesh = new THREE.InstancedMesh(
    new THREE.BoxGeometry(1, 0.1, 1),
    new THREE.MeshBasicMaterial(),
    data.length
  )

  const dummy = new THREE.Object3D()
  const color = new THREE.Color()

  data.forEach((point, i) => {
    dummy.position.set(point.x, 0, point.y)
    dummy.updateMatrix()
    instancedMesh.setMatrixAt(i, dummy.matrix)

    // 温度映射到颜色（蓝->绿->黄->红）
    const t = (point.temp - 20) / 40 // 归一化
    color.setHSL(0.7 - t * 0.7, 1, 0.5)
    instancedMesh.setColorAt(i, color)
  })

  instancedMesh.instanceMatrix.needsUpdate = true
  instancedMesh.instanceColor.needsUpdate = true

  return instancedMesh
}
```

```javascript
// 数字孪生架构
class DigitalTwinSystem {
  constructor(scene, websocketUrl) {
    this.scene = scene
    this.ws = new WebSocket(websocketUrl)
    this.equipmentMap = new Map() // 设备ID -> Mesh

    this.ws.onmessage = (event) => {
      const data = JSON.parse(event.data)
      this.updateEquipment(data)
    }
  }

  updateEquipment(data) {
    data.forEach(({ id, status, temperature, vibration }) => {
      const mesh = this.equipmentMap.get(id)
      if (!mesh) return

      // 状态颜色
      const colors = {
        normal: 0x4ecca3,
        warning: 0xffaa00,
        error: 0xff4444,
        offline: 0x666666
      }

      mesh.material.color.setHex(colors[status] || colors.offline)

      // 振动效果（异常时抖动）
      if (status === 'error') {
        mesh.position.x += (Math.random() - 0.5) * vibration * 0.1
      }
    })
  }

  // 点击设备弹出详情面板
  onEquipmentClick(meshId, callback) {
    const mesh = this.equipmentMap.get(meshId)
    if (mesh) {
      callback({
        name: mesh.userData.name,
        model: mesh.userData.model,
        installDate: mesh.userData.installDate,
        // ... 从后端获取实时数据
      })
    }
  }
}
```

```javascript
        // 使用 Object3D.userData 存储业务数据
        mesh.userData = { ...mesh.userData, ...data.metadata }
      }
```
