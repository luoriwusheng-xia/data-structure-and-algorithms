### background.js —— 初始化与状态同步

```javascript
/**
 * Service Worker —— 扩展的后台事件中枢
 * 负责：初始化默认配置、响应来自 popup/content 的消息
 *
 * 注意：Service Worker 是事件驱动的，不会持久运行。
 * 所有状态必须使用 chrome.storage 持久化，不能存在内存中。
 */

// 扩展安装时触发（首次安装或更新时）
chrome.runtime.onInstalled.addListener((details) => {
  /**
   * details.reason 可能的值：
   * - 'install'    : 首次安装
   * - 'update'     : 扩展更新
   * - 'chrome_update': Chrome 浏览器更新
   * - 'shared_module_update': 共享模块更新
   */
  if (details.reason === 'install') {
    // 初始化默认配置到 chrome.storage.sync
    // sync 会自动同步到用户登录的所有 Chrome 设备
    chrome.storage.sync.set({
      darkModeEnabled: false,  // 深色模式默认关闭
      intensity: 0.85          // 反色强度 0-1
    });
  }
});

/**
 * 监听来自 popup 或 content script 的单次消息
 *
 * 消息格式约定：
 * {
 *   action: string,    // 操作类型
 *   payload?: any      // 可选数据负载
 * }
 *
 * 注意：如果需要异步响应（如调用 API、读写 storage），
 * 必须返回 true 来保持消息通道开启，否则 sendResponse 会被忽略。
 */
chrome.runtime.onMessage.addListener((request, sender, sendResponse) => {
  const { action, payload } = request;

  switch (action) {
    // 切换当前标签页的深色模式
    case 'TOGGLE_DARK_MODE': {
      /**
       * chrome.tabs.query 获取符合条件的标签页
       * active: true     —— 当前激活的标签页
       * currentWindow: true —— 当前窗口中的标签页
       */
      chrome.tabs.query({ active: true, currentWindow: true })
        .then(([tab]) => {
          if (!tab?.id) {
            sendResponse({ success: false, error: '无法获取当前标签页' });
            return;
          }

          /**
           * chrome.scripting.executeScript —— V3 动态注入脚本
           * target.tabId: 目标标签页
           * func: 要执行的函数（会被序列化后注入到页面执行）
           * args: 传递给 func 的参数（必须是可序列化的）
           *
           * ⚠️ func 中不能使用外部作用域变量！
           * 它会被 toString() 后发送到页面执行，闭包会丢失。
           */
          return chrome.scripting.executeScript({
            target: { tabId: tab.id },
            func: (enabled, intensity) => {
              /**
               * 这段代码运行在目标页面的「隔离世界」中
               * 可以读写 DOM，但无法访问页面自身的 JS 变量
               */

              const STYLE_ID = 'ext-dark-mode-style';

              if (enabled) {
                // 启用深色模式：注入 CSS filter
                let style = document.getElementById(STYLE_ID);
                if (!style) {
                  style = document.createElement('style');
                  style.id = STYLE_ID;
                  document.documentElement.appendChild(style);
                }

                /**
                 * CSS filter 方案说明：
                 * - invert(1): 颜色反相（白变黑，黑变白）
                 * - hue-rotate(180deg): 色相旋转，修正反相后的色相偏差
                 * - brightness(intensity): 控制亮度，避免纯黑刺眼
                 *
                 * 对图片和视频的处理：
                 * - img, video, svg 需要再反相一次恢复原始颜色
                 * - [style*="background-image"] 处理内联背景图
                 */
                style.textContent = `
                  html {
                    filter: invert(1) hue-rotate(180deg) brightness(${intensity}) !important;
                  }
                  img, video, svg, canvas,
                  [style*="background-image"] {
                    filter: invert(1) hue-rotate(180deg) !important;
                  }
                `;
              } else {
                // 禁用深色模式：移除样式
                const style = document.getElementById(STYLE_ID);
                if (style) style.remove();
              }

              // 返回当前状态给调用方
              return { enabled, tabUrl: location.href };
            },
            args: [payload.enabled, payload.intensity] // 传递到页面的参数
          });
        })
        .then((results) => {
          /**
           * executeScript 返回的结果是一个数组，每个元素对应一个 frame：
           * [
           *   {
           *     frameId: number,    // iframe 的 ID，0 表示主 frame
           *     result: any          // func 的返回值
           *   }
           * ]
           */
          const result = results?.[0]?.result;
          sendResponse({ success: true, result });
        })
        .catch((err) => {
          sendResponse({ success: false, error: err.message });
        });

      // ⚠️ 必须返回 true！上面使用了 .then() 异步链
      return true;
    }

    default:
      sendResponse({ success: false, error: '未知操作' });
  }
});
```

### popup.js —— 面板交互逻辑

```javascript
/**
 * Popup 脚本 —— 每次点击扩展图标都会重新加载执行
 *
 * 生命周期：
 * 1. 用户点击图标 → popup.html 加载 → popup.js 执行
 * 2. 用户点击页面其他区域 → popup 关闭 → 所有状态丢失
 *
 * 因此所有持久化状态必须通过 chrome.storage 或消息通信获取，
 * 不能依赖内存中的变量。
 */

// DOM 元素引用
const toggleEl = document.getElementById('toggle');
const intensityEl = document.getElementById('intensity');
const intensityValueEl = document.getElementById('intensityValue');
const statusEl = document.getElementById('status');

/**
 * 初始化：从 storage 读取配置并同步 UI
 * 使用 IIFE 是因为顶层不能直接用 await（除非 type="module"）
 */
(async function init() {
  try {
    /**
     * chrome.storage.sync.get(keys) —— 读取同步存储
     * 参数可以是：字符串、字符串数组、对象（用默认值）
     * 返回值是 Promise，resolve 为键值对对象
     */
    const { darkModeEnabled = false, intensity = 0.85 } =
      await chrome.storage.sync.get(['darkModeEnabled', 'intensity']);

    // 同步 UI 状态
    toggleEl.checked = darkModeEnabled;
    intensityEl.value = intensity;
    intensityValueEl.textContent = `${Math.round(intensity * 100)}%`;
    updateStatusUI(darkModeEnabled);
  } catch (err) {
    console.error('初始化失败:', err);
    statusEl.textContent = '加载失败，请重试';
    statusEl.className = 'status off';
  }
})();

// 开关切换事件
toggleEl.addEventListener('change', async (e) => {
  const enabled = e.target.checked;
  const intensity = parseFloat(intensityEl.value);

  // 先更新 UI 响应，提升感知性能
  updateStatusUI(enabled);

  try {
    /**
     * chrome.runtime.sendMessage —— 向 Service Worker 发送消息
     * 消息会被 broadcast 到所有监听 runtime.onMessage 的组件
     *
     * 第一个参数：消息体（任意可序列化对象）
     * 第二个参数（可选）：回调函数，或 await 其返回的 Promise
     */
    const response = await chrome.runtime.sendMessage({
      action: 'TOGGLE_DARK_MODE',
      payload: { enabled, intensity }
    });

    if (response.success) {
      // 持久化状态到 storage
      await chrome.storage.sync.set({ darkModeEnabled: enabled });
    } else {
      // 失败时回滚 UI
      toggleEl.checked = !enabled;
      updateStatusUI(!enabled);
      statusEl.textContent = `失败: ${response.error}`;
    }
  } catch (err) {
    console.error('通信失败:', err);
    toggleEl.checked = !enabled;
    updateStatusUI(!enabled);
    statusEl.textContent = '扩展通信异常';
  }
});

// 强度滑块事件（防抖处理）
let debounceTimer;
intensityEl.addEventListener('input', (e) => {
  const value = parseFloat(e.target.value);
  intensityValueEl.textContent = `${Math.round(value * 100)}%`;

  // 300ms 防抖，避免频繁发送消息
  clearTimeout(debounceTimer);
  debounceTimer = setTimeout(async () => {
    await chrome.storage.sync.set({ intensity: value });

    // 如果深色模式已开启，实时应用新强度
    if (toggleEl.checked) {
      chrome.runtime.sendMessage({
        action: 'TOGGLE_DARK_MODE',
        payload: { enabled: true, intensity: value }
      });
    }
  }, 300);
});

// 更新状态显示 UI
function updateStatusUI(enabled) {
  statusEl.textContent = enabled ? '当前状态：已开启' : '当前状态：已关闭';
  statusEl.className = enabled ? 'status on' : 'status off';
}
```

### 案例要点

| 要点 | 说明 |
|------|------|
| `activeTab` + `scripting` | 最安全的权限组合，仅在用户触发时获得当前页权限 |
| `executeScript.func` | V3 动态注入，函数会被序列化，不能使用外部变量 |
| `chrome.storage.sync` | 跨设备同步的配置存储 |
| `return true` | 异步消息监听必须返回 true |
| Popup 生命周期 | 关闭即销毁，所有状态依赖 storage |

## 案例二：网页内容采集助手

### 功能描述

在任意网页选中文字后，通过右键菜单或快捷键采集到扩展的侧边面板，支持导出为 Markdown。


### background.js —— 右键菜单与快捷键

```javascript
/**
 * 网页采集助手 —— Service Worker
 *
 * 核心职责：
 * 1. 创建和管理右键菜单（contextMenus）
 * 2. 监听快捷键命令（commands）
 * 3. 作为 Content Script 和 Side Panel 的中转站
 */

// 存储采集的数据（内存缓存 + storage 持久化）
// 注意：Service Worker 会休眠，关键数据必须写 storage
const CLIP_STORAGE_KEY = 'clips';

// ============ 初始化 ============

chrome.runtime.onInstalled.addListener(() => {
  /**
   * chrome.contextMenus.create —— 创建右键菜单项
   *
   * contexts 指定在哪些场景下显示菜单：
   * - 'page': 页面空白处右键
   * - 'selection': 选中文本后右键
   * - 'link': 在链接上右键
   * - 'image': 在图片上右键
   * - 'all': 所有场景
   */
  chrome.contextMenus.create({
    id: 'clip-selection',
    title: '✂️ 采集选中文本',
    contexts: ['selection'],  // 仅在有选中文本时显示
    documentUrlPatterns: ['http://*/*', 'https://*/*']
  });

  chrome.contextMenus.create({
    id: 'clip-page',
    title: '📄 采集整页内容',
    contexts: ['page'],       // 仅在页面空白处显示
    documentUrlPatterns: ['http://*/*', 'https://*/*']
  });

  chrome.contextMenus.create({
    id: 'open-sidepanel',
    title: '📋 打开采集面板',
    contexts: ['all']
  });
});

// ============ 右键菜单点击处理 ============

chrome.contextMenus.onClicked.addListener(async (info, tab) => {
  /**
   * info 对象包含：
   * - menuItemId: 点击的菜单项 ID
   * - selectionText: 选中的文本（contexts 包含 'selection' 时）
   * - pageUrl: 当前页面 URL
   * - frameUrl: 如果是 iframe 内的右键
   * - srcUrl: 图片/视频源地址
   * - linkUrl: 链接地址
   *
   * tab 对象包含标签页信息（id, url, title 等）
   */

  switch (info.menuItemId) {
    case 'clip-selection': {
      const clip = createClip({
        type: 'selection',
        content: info.selectionText,
        title: tab.title,
        url: tab.url,
        timestamp: Date.now()
      });
      await saveClip(clip);
      showNotification('已采集选中文本');
      break;
    }

    case 'clip-page': {
      // 向 Content Script 发送消息，获取页面结构化内容
      try {
        const response = await chrome.tabs.sendMessage(tab.id, {
          action: 'EXTRACT_PAGE_CONTENT'
        });

        const clip = createClip({
          type: 'page',
          content: response.content,
          title: tab.title,
          url: tab.url,
          timestamp: Date.now()
        });
        await saveClip(clip);
        showNotification('已采集整页内容');
      } catch (err) {
        console.error('采集失败:', err);
        showNotification('采集失败: ' + err.message);
      }
      break;
    }

    case 'open-sidepanel': {
      /**
       * chrome.sidePanel.open —— 打开侧边面板
       * 需要 'sidePanel' 权限，Chrome 114+
       * 必须在用户手势（如点击）上下文中调用
       */
      await chrome.sidePanel.open({ windowId: tab.windowId });
      break;
    }
  }
});

// ============ 快捷键处理 ============

chrome.commands.onCommand.addListener(async (command, tab) => {
  /**
   * command 是 manifest.json 中定义的命令 ID
   * 系统保留命令：
   * - '_execute_action': 打开 popup（如配置了 action.default_popup）
   * - '_execute_sidebar_action': 打开侧边栏（Firefox）
   */

  if (command === 'quick-clip') {
    // 快捷键采集：获取当前选区并保存
    try {
      const [selection] = await chrome.scripting.executeScript({
        target: { tabId: tab.id },
        func: () => window.getSelection().toString().trim()
      });

      if (selection?.result) {
        const clip = createClip({
          type: 'selection',
          content: selection.result,
          title: tab.title,
          url: tab.url,
          timestamp: Date.now()
        });
        await saveClip(clip);
        showNotification('快速采集成功');
      } else {
        showNotification('请先选中要采集的文本');
      }
    } catch (err) {
      console.error('快速采集失败:', err);
    }
  }
});

// ============ 工具函数 ============

function createClip(data) {
  return {
    id: generateId(),        // 唯一 ID，用于后续删除/编辑
    ...data,
    tags: [],                // 可添加标签
    note: ''                 // 可添加备注
  };
}

function generateId() {
  // 简单 ID 生成：时间戳 + 随机数
  return `${Date.now()}-${Math.random().toString(36).slice(2, 8)}`;
}

async function saveClip(clip) {
  // 读取现有数据
  const { clips = [] } = await chrome.storage.local.get(CLIP_STORAGE_KEY);

  // 新数据添加到头部（最新的在前）
  clips.unshift(clip);

  // 限制存储数量（Chrome storage 有容量限制）
  const MAX_CLIPS = 500;
  if (clips.length > MAX_CLIPS) {
    clips.length = MAX_CLIPS;
  }

  // 保存到 local storage（sync 有容量限制，大量数据用 local）
  await chrome.storage.local.set({ [CLIP_STORAGE_KEY]: clips });

  // 通知 Side Panel 更新（如果已打开）
  try {
    await chrome.runtime.sendMessage({
      target: 'sidepanel',
      action: 'CLIP_ADDED',
      clip
    });
  } catch {
    // Side Panel 可能未打开，忽略错误
  }
}

function showNotification(message) {
  /**
   * chrome.notifications.create —— 系统桌面通知
   * 需要 'notifications' 权限
   *
   * type 选项：
   * - 'basic': 简单通知
   * - 'image': 带图片
   * - 'list': 列表形式
   * - 'progress': 进度条
   */
  chrome.notifications.create({
    type: 'basic',
    iconUrl: 'icons/icon128.png',
    title: '网页采集助手',
    message: message
  });
}
```

### content.js —— 页面内容提取

```javascript
/**
 * Content Script —— 注入到每个匹配的网页中
 *
 * 运行环境：
 * - 可读写页面 DOM
 * - 可监听页面事件（click, keydown, selectionchange 等）
 * - 与页面 JS 隔离（不能互相访问变量）
 * - 与扩展通信通过 chrome.runtime.sendMessage
 */

// 监听来自 Service Worker 的消息
chrome.runtime.onMessage.addListener((request, sender, sendResponse) => {
  const { action } = request;

  switch (action) {
    case 'EXTRACT_PAGE_CONTENT': {
      /**
       * 提取页面正文内容
       * 使用 Readability 算法原理：
       * 1. 移除脚本、样式、导航等噪音元素
       * 2. 计算每个节点的文本密度
       * 3. 选择得分最高的容器作为正文
       */
      const content = extractMainContent();
      sendResponse({ success: true, content });
      break;
    }

    default:
      sendResponse({ success: false, error: '未知操作' });
  }

  // 同步响应，不需要 return true
});

/**
 * 提取页面主要内容
 * 这是一个简化版算法，生产环境建议使用 mozilla/readability 库
 */
function extractMainContent() {
  // 克隆文档，避免修改原始页面
  const doc = document.cloneNode(true);

  // 移除噪音元素
  const noiseSelectors = [
    'script', 'style', 'nav', 'header', 'footer',
    'aside', '.sidebar', '.advertisement', '.comment',
    '#comments', '.social-share', '.related-posts'
  ];
  noiseSelectors.forEach(sel => {
    doc.querySelectorAll(sel).forEach(el => el.remove());
  });

  // 候选容器评分
  const candidates = [];
  const paragraphs = doc.querySelectorAll('p, article, section, div');

  paragraphs.forEach(el => {
    const text = el.textContent.trim();
    const textLen = text.length;
    const commaCount = (text.match(/[,，]/g) || []).length;
    const linkDensity = calculateLinkDensity(el);

    // 文本长度 + 逗号数（段落特征）- 链接密度惩罚
    const score = textLen * 1.0 + commaCount * 10 - linkDensity * 50;

    if (textLen > 100) {
      candidates.push({ element: el, score, textLen });
    }
  });

  // 按分数排序，取最高分
  candidates.sort((a, b) => b.score - a.score);

  if (candidates.length === 0) {
    return { title: document.title, text: '', html: '' };
  }

  const best = candidates[0];

  return {
    title: document.title,
    text: best.element.textContent.trim(),
    html: best.element.innerHTML,
    url: location.href
  };
}

function calculateLinkDensity(element) {
  const textLength = element.textContent.trim().length || 1;
  const linkTextLength = Array.from(element.querySelectorAll('a'))
    .reduce((sum, a) => sum + a.textContent.trim().length, 0);
  return linkTextLength / textLength;
}
```

```javascript
function downloadFile(content, filename, mimeType) {
  const blob = new Blob([content], { type: mimeType });
  const url = URL.createObjectURL(blob);

  // Chrome 扩展中可以使用 chrome.downloads，也可以创建临时 a 标签
  const a = document.createElement('a');
  a.href = url;
  a.download = filename;
  a.click();

  URL.revokeObjectURL(url);
}

function escapeHtml(text) {
  const div = document.createElement('div');
  div.textContent = text;
  return div.innerHTML;
}

function truncate(str, maxLen) {
  if (!str || str.length <= maxLen) return str;
  return str.slice(0, maxLen) + '...';
}

function formatDate(timestamp) {
  return new Date(timestamp).toLocaleString('zh-CN');
}

function formatFileDate() {
  const d = new Date();
  return `${d.getFullYear()}${pad(d.getMonth()+1)}${pad(d.getDate())}-${pad(d.getHours())}${pad(d.getMinutes())}`;
}

function pad(n) {
  return String(n).padStart(2, '0');
}
```

### declarativeNetRequest 规则配置

```javascript
// background.js

/**
 * declarativeNetRequest —— 声明式网络请求修改
 *
 * V3 的设计哲学：网络规则在 manifest 中静态声明，
 * 运行时只能增删规则，不能动态生成规则内容。
 * 这是为了安全和性能，但也限制了灵活性。
 */

const RULE_ID_OFFSET = 1000;

/**
 * 添加动态规则来重定向或修改请求
 * chrome.declarativeNetRequest.updateDynamicRules —— 修改动态规则集
 * chrome.declarativeNetRequest.updateSessionRules —— 修改会话级规则（浏览器关闭后清除）
 */
async function addMockRule(apiPath, mockResponse) {
  // 生成规则 ID（需唯一）
  const ruleId = RULE_ID_OFFSET + Math.floor(Math.random() * 1000);

  // 将 mock 响应编码为 data URL
  const responseBody = btoa(JSON.stringify(mockResponse));
  const dataUrl = `data:application/json;base64,${responseBody}`;

  await chrome.declarativeNetRequest.updateDynamicRules({
    addRules: [
      {
        id: ruleId,
        priority: 1,
        action: {
          type: 'redirect',
          redirect: { url: dataUrl }
        },
        condition: {
          urlFilter: apiPath,
          resourceTypes: ['xmlhttprequest']
        }
      }
    ],
    removeRuleIds: [] // 可同时移除旧规则
  });

  return ruleId;
}

/**
 * 另一种方案：使用 modifyHeaders 修改请求头
 */
async function addHeaderRule(apiPath, headers) {
  const requestHeaders = Object.entries(headers).map(([name, value]) => ({
    header: name,
    operation: 'set',
    value
  }));

  await chrome.declarativeNetRequest.updateDynamicRules({
    addRules: [
      {
        id: 2001,
        priority: 1,
        action: {
          type: 'modifyHeaders',
          requestHeaders
        },
        condition: {
          urlFilter: apiPath,
          resourceTypes: ['xmlhttprequest']
        }
      }
    ],
    removeRuleIds: [2001] // 替换旧规则
  });
}

// 查看当前生效的规则（调试用）
async function logActiveRules() {
  const dynamicRules = await chrome.declarativeNetRequest.getDynamicRules();
  const sessionRules = await chrome.declarativeNetRequest.getSessionRules();
  console.log('动态规则:', dynamicRules);
  console.log('会话规则:', sessionRules);
}

/**
 * 监听请求（仅观察，不能修改）
 * 适用于日志记录、统计等场景
 */
chrome.declarativeNetRequest.onRuleMatchedDebug?.addListener((info) => {
  console.log('规则匹配:', info);
});
```

```javascript
 * panel.js —— DevTools 面板逻辑
 *
 * 关键 API：
 * - chrome.devtools.inspectedWindow: 操作被检查页面
 * - chrome.devtools.network: 监听网络请求
 * - chrome.devtools.performance: 性能分析
 */

const recordBtn = document.getElementById('recordBtn');
let isRecording = false;

// ============ 性能指标采集 ============

/**
 * chrome.devtools.inspectedWindow.eval ——
 * 在被检查页面的上下文中执行 JS 代码
 *
 * 这是 DevTools 扩展最核心的 API，
 * 可以访问页面的所有变量和 API。
 */
function evalInPage(expression, callback) {
  chrome.devtools.inspectedWindow.eval(expression, callback);
}

// 采集 Core Web Vitals
async function collectMetrics() {
  evalInPage(`
    (function() {
      /**
       * PerformanceObserver —— W3C Performance API
       * 监听浏览器性能指标
       *
       * entryType 类型：
       * - 'navigation': 页面导航时间
       * - 'resource': 资源加载时间
       * - 'paint': 渲染时间点（fcp, lcp）
       * - 'layout-shift': 布局偏移（cls）
       * - 'first-input': 首次输入（fid）
       * - 'largest-contentful-paint': LCP
       */
      const paintEntries = performance.getEntriesByType('paint');
      const fcpEntry = paintEntries.find(e => e.name === 'first-contentful-paint');

      const lcpEntries = performance.getEntriesByType('largest-contentful-paint');
      const lcpEntry = lcpEntries[lcpEntries.length - 1];

      const layoutShiftEntries = performance.getEntriesByType('layout-shift');
      const cls = layoutShiftEntries.reduce((sum, e) => sum + e.value, 0);

      const navEntry = performance.getEntriesByType('navigation')[0];

      return {
        fcp: fcpEntry ? Math.round(fcpEntry.startTime) : null,
        lcp: lcpEntry ? Math.round(lcpEntry.startTime) : null,
        cls: Math.round(cls * 1000) / 1000,
        ttfb: navEntry ? Math.round(navEntry.responseStart) : null,
        domComplete: navEntry ? Math.round(navEntry.domComplete) : null,
        resources: performance.getEntriesByType('resource')
          .slice(-10) // 最近 10 个资源
          .map(r => ({
            name: r.name.split('/').pop(),
            duration: Math.round(r.duration),
            size: r.transferSize
          }))
      };
    })()
  `, (result, isException) => {
    if (isException) {
      console.error('性能采集失败:', isException);
      return;
    }
    updateMetricsUI(result);
  });
}

function updateMetricsUI(metrics) {
  // FCP 评分：< 1.8s good, < 3s needs-improvement, else poor
  setMetric('fcp', metrics.fcp, 1800, 3000);
  setMetric('lcp', metrics.lcp, 2500, 4000);
  setMetric('cls', metrics.cls, 0.1, 0.25);

  // 渲染资源加载时间线
  renderTimeline(metrics.resources);
}

function setMetric(id, value, goodThreshold, poorThreshold) {
  const el = document.getElementById(id);
  if (value === null) {
    el.textContent = '--';
    el.className = 'metric-value';
    return;
  }

  const isTime = goodThreshold > 1; // 判断是时间(ms)还是 CLS 分数
  el.textContent = isTime ? `${(value / 1000).toFixed(2)}s` : value;

  if (value <= goodThreshold) el.className = 'metric-value good';
  else if (value <= poorThreshold) el.className = 'metric-value warn';
  else el.className = 'metric-value bad';
}

function renderTimeline(resources) {
  const container = document.getElementById('timeline');
  if (!resources?.length) {
    container.innerHTML = '<p>暂无资源数据</p>';
    return;
  }

  const maxDuration = Math.max(...resources.map(r => r.duration));

  container.innerHTML = resources.map(r => {
    const width = maxDuration > 0 ? (r.duration / maxDuration * 100) : 0;
    const size = r.size > 1024 ? `${(r.size/1024).toFixed(1)}KB` : `${r.size}B`;
    return `
      <div class="timeline-item">
        <div style="width: 120px; overflow: hidden; text-overflow: ellipsis; white-space: nowrap;">
          ${r.name}
        </div>
        <div class="timeline-bar" style="width: ${width}%;"></div>
        <div style="width: 80px; text-align: right;">
          ${r.duration}ms · ${size}
        </div>
      </div>
    `;
  }).join('');
}

// ============ 网络请求监听 ============

/**
 * chrome.devtools.network.onRequestFinished ——
 * 监听被检查页面的网络请求
 */
chrome.devtools.network.onRequestFinished.addListener((request) => {
  /**
   * request 对象包含：
   * - request: { url, method, headers, postData }
   * - response: { status, headers, content }
   * - time: 总耗时
   * - timings: 各阶段耗时 breakdown
   */
  console.log('[Network]', request.request.method, request.request.url);

  // 获取响应内容（可选，大响应可能影响性能）
  request.getContent((content, encoding) => {
    // content: 响应体文本
    // encoding: 编码方式（如 base64）
  });
});

// ============ 事件绑定 ============

recordBtn.addEventListener('click', () => {
  isRecording = !isRecording;
  recordBtn.textContent = isRecording ? '⏹ 停止录制' : '▶ 开始录制性能';

  if (isRecording) {
    collectMetrics();
    // 每 2 秒刷新一次指标
    window._metricInterval = setInterval(collectMetrics, 2000);
  } else {
    clearInterval(window._metricInterval);
  }
});

// 页面重载时自动采集
chrome.devtools.network.onNavigated.addListener(() => {
  if (isRecording) {
    setTimeout(collectMetrics, 1000); // 等待页面稳定
  }
});
```

## 案例五：多扩展协作通信

### 场景

扩展 A 需要调用扩展 B 的功能（例如一个通用登录扩展被多个业务扩展调用）。

### 扩展 A（调用方）

```javascript
// ext-a/background.js

/**
 * 向指定扩展发送消息
 *
 * chrome.runtime.sendMessage 的完整签名：
 * sendMessage(extensionId, message, options?, responseCallback?)
 *
 * extensionId 是被调用扩展的 ID（从 Chrome Web Store URL 或扩展管理中获取）
 */
const AUTH_EXTENSION_ID = 'abcdefghijklmnopqrstuvwxyzabcdef'; // 扩展 B 的 ID

async function requestAuth() {
  try {
    const response = await chrome.runtime.sendMessage(
      AUTH_EXTENSION_ID,
      {
        action: 'GET_AUTH_TOKEN',
        // 可传递额外参数
        scope: 'read write'
      }
    );

    if (response.success) {
      console.log('获取到 Token:', response.token);
      return response.token;
    } else {
      console.error('认证失败:', response.error);
      return null;
    }
  } catch (err) {
    /**
     * 常见错误：
     * - "Could not establish connection"：扩展 B 未安装或未运行
     * - "Access denied"：扩展 B 未声明 externally_connectable
     */
    console.error('无法连接到认证扩展:', err.message);
    // 降级：引导用户安装扩展 B
    chrome.tabs.create({
      url: 'https://chrome.google.com/webstore/detail/' + AUTH_EXTENSION_ID
    });
    return null;
  }
}
```

```javascript
// ext-b/background.js

chrome.runtime.onMessageExternal.addListener(
  (request, sender, sendResponse) => {
    /**
     * sender 包含调用方信息：
     * - sender.id: 调用方扩展 ID
     * - sender.url: 调用方页面 URL
     * - sender.origin: 调用方来源
     *
     * 应该验证 sender.id，防止未授权调用
     */
    const ALLOWED_SENDERS = [
      'aaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaa', // 扩展 A ID
      'bbbbbbbbbbbbbbbbbbbbbbbbbbbbbbbb'  // 扩展 C ID
    ];

    if (!ALLOWED_SENDERS.includes(sender.id)) {
      sendResponse({ success: false, error: '未授权的调用方' });
      return false;
    }

    const { action } = request;

    switch (action) {
      case 'GET_AUTH_TOKEN': {
        getStoredToken()
          .then(token => sendResponse({ success: true, token }))
          .catch(err => sendResponse({ success: false, error: err.message }));
        return true; // 异步响应
      }

      case 'REFRESH_TOKEN': {
        refreshToken()
          .then(token => sendResponse({ success: true, token }))
          .catch(err => sendResponse({ success: false, error: err.message }));
        return true;
      }

      default:
        sendResponse({ success: false, error: '未知操作' });
    }
  }
);

async function getStoredToken() {
  const { token } = await chrome.storage.local.get('token');
  if (!token || isExpired(token)) {
    return refreshToken();
  }
  return token;
}

async function refreshToken() {
  // 使用 Chrome Identity API 获取 OAuth token
  const token = await chrome.identity.getAuthToken({ interactive: true });
  await chrome.storage.local.set({
    token,
    tokenExpiry: Date.now() + 3600000 // 1 小时
  });
  return token;
}
```
