# TypeScript 进阶

## 泛型进阶

### 泛型约束

```typescript
interface HasLength {
  length: number;
}

function logLength<T extends HasLength>(arg: T): T {
  console.log(arg.length);
  return arg;
}

logLength("hello");     // OK: string 有 length 属性
logLength([1, 2, 3]);   // OK: 数组有 length 属性
// logLength(123);      // Error: number 没有 length 属性
```

### 多个泛型参数

```typescript
function map<T, U>(array: T[], fn: (item: T) => U): U[] {
  return array.map(fn);
}

const nums = [1, 2, 3];
const strings = map(nums, (n) => n.toString()); // string[]
```

### 泛型默认值

```typescript
function createArray<T = string>(length: number, value: T): T[] {
  return Array(length).fill(value);
}

const arr1 = createArray(3, "a");   // string[]
const arr2 = createArray(3, 1);      // number[]
```

## 条件类型

### 基本用法

```typescript
type IsString<T> = T extends string ? true : false;

type A = IsString<string>;  // true
type B = IsString<number>;  // false
```

### infer 关键字

```typescript
// 提取函数返回值类型
type ReturnType<T> = T extends (...args: any[]) => infer R ? R : never;

function foo() {
  return { a: 1, b: "hello" };
}

type FooReturn = ReturnType<typeof foo>; // { a: number; b: string }

// 提取数组元素类型
type ElementType<T> = T extends (infer E)[] ? E : never;

type Num = ElementType<number[]>;   // number
type Str = ElementType<string[]>;   // string
```

### 分布式条件类型

```typescript
type ToArray<T> = T extends any ? T[] : never;

// 当传入联合类型时，条件类型会分布到每个成员
type StrOrNumArray = ToArray<string | number>;
// 等价于 string[] | number[]，而不是 (string | number)[]
```

## 映射类型

### 内置映射类型原理

```typescript
// Partial<T> 的实现
type MyPartial<T> = {
  [P in keyof T]?: T[P];
};

// Required<T> 的实现
type MyRequired<T> = {
  [P in keyof T]-?: T[P];  // -? 移除可选修饰符
};

// Readonly<T> 的实现
type MyReadonly<T> = {
  readonly [P in keyof T]: T[P];
};

// Mutable<T> 移除 readonly
type Mutable<T> = {
  -readonly [P in keyof T]: T[P];
};
```

### 键重映射

```typescript
// 修改键名
type Getters<T> = {
  [K in keyof T as `get${Capitalize<string & K>}`]: () => T[K];
};

interface Person {
  name: string;
  age: number;
}

type PersonGetters = Getters<Person>;
// {
//   getName: () => string;
//   getAge: () => number;
// }

// 过滤特定类型的属性
type FilterByType<T, U> = {
  [K in keyof T as T[K] extends U ? K : never]: T[K];
};

interface Example {
  a: string;
  b: number;
  c: string;
  d: boolean;
}

type StringProps = FilterByType<Example, string>;
// { a: string; c: string }
```

## 模板字面量类型

```typescript
type EventName<T extends string> = `on${Capitalize<T>}`;

type ClickEvent = EventName<"click">;     // "onClick"
type ChangeEvent = EventName<"change">;   // "onChange"

// 组合多个模板
type Direction = "top" | "bottom" | "left" | "right";
type Margin = `margin${Capitalize<Direction>}`;
// "marginTop" | "marginBottom" | "marginLeft" | "marginRight"
```

## 类型体操实战

### 深度 Readonly

```typescript
type DeepReadonly<T> = {
  readonly [P in keyof T]: T[P] extends object
    ? T[P] extends Function
      ? T[P]
      : DeepReadonly<T[P]>
    : T[P];
};

interface Nested {
  a: {
    b: {
      c: string;
    };
  };
}

type DeepReadonlyNested = DeepReadonly<Nested>;
// 所有层级都变为 readonly
```

### 元组转联合类型

```typescript
type TupleToUnion<T extends readonly any[]> = T[number];

type Result = TupleToUnion<["a", "b", "c"]>;
// "a" | "b" | "c"
```

### 对象路径类型

```typescript
type Path<T> = T extends object
  ? {
      [K in keyof T]: K extends string
        ? T[K] extends object
          ? K | `${K}.${Path<T[K]>}`
          : K
        : never;
    }[keyof T]
  : never;

interface Data {
  user: {
    name: string;
    address: {
      city: string;
    };
  };
}

type DataPath = Path<Data>;
// "user" | "user.name" | "user.address" | "user.address.city"
```
