## 一、离线消息队列

场景：用户在弱网环境下发送消息，需要本地缓存，网络恢复后按序发送。

```javascript
// ========== 数据库初始化 ==========
const DB_NAME = 'MessageQueueDB';
const DB_VERSION = 1;

function initDB() {
  return new Promise((resolve, reject) => {
    const request = indexedDB.open(DB_NAME, DB_VERSION);

    request.onupgradeneeded = (event) => {
      const db = event.target.result;

      // 待发送消息队列
      const pendingStore = db.createObjectStore('pending', {
        keyPath: 'id'
      });
      pendingStore.createIndex('timestamp', 'timestamp', { unique: false });
      pendingStore.createIndex('priority', 'priority', { unique: false });

      // 发送失败的消息（重试队列）
      const failedStore = db.createObjectStore('failed', {
        keyPath: 'id'
      });
      failedStore.createIndex('retryCount', 'retryCount', { unique: false });
      failedStore.createIndex('nextRetryAt', 'nextRetryAt', { unique: false });
    };

    request.onsuccess = () => resolve(request.result);
    request.onerror = () => reject(request.error);
  });
}

// ========== 消息队列管理器 ==========
class OfflineMessageQueue {
  constructor(db) {
    this.db = db;
    this.isProcessing = false;
    this.maxRetries = 3;
    this.retryDelays = [1000, 5000, 30000]; // 1s, 5s, 30s
  }

  // 添加消息到待发送队列
  async enqueue(message, options = {}) {
    const id = crypto.randomUUID();
    const record = {
      id,
      content: message,
      timestamp: Date.now(),
      priority: options.priority || 0,
      metadata: {
        url: location.href,
        userAgent: navigator.userAgent
      }
    };

    const transaction = this.db.transaction(['pending'], 'readwrite');
    const store = transaction.objectStore('pending');
    await promisifyRequest(store.add(record));

    // 尝试立即发送
    this.processQueue();

    return id;
  }

  // 处理队列
  async processQueue() {
    if (this.isProcessing) return;
    if (!navigator.onLine) return;

    this.isProcessing = true;

    try {
      // 按优先级和时间排序获取待发送消息
      const messages = await this.getPendingMessages();

      for (const msg of messages) {
        try {
          await this.sendMessage(msg);
          await this.removePending(msg.id);
        } catch (error) {
          await this.handleSendFailure(msg, error);
        }
      }

      // 处理失败队列中的重试
      await this.processFailedQueue();
    } finally {
      this.isProcessing = false;
    }
  }

  // 按优先级获取待发送消息
  async getPendingMessages() {
    const transaction = this.db.transaction(['pending'], 'readonly');
    const store = transaction.objectStore('pending');

    // 使用复合索引或内存排序
    const all = await promisifyRequest(store.getAll());
    return all.sort((a, b) => {
      if (a.priority !== b.priority) return b.priority - a.priority;
      return a.timestamp - b.timestamp;
    });
  }

  // 模拟发送消息
  async sendMessage(record) {
    const response = await fetch('/api/messages', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(record.content)
    });

    if (!response.ok) {
      throw new Error(`发送失败: ${response.status}`);
    }

    return response.json();
  }

  // 处理发送失败
  async handleSendFailure(record, error) {
    await this.removePending(record.id);

    const retryCount = (record.retryCount || 0) + 1;

    if (retryCount > this.maxRetries) {
      // 超过重试次数，移动到永久失败队列或上报
      console.error('消息发送最终失败:', record.id, error);
      return;
    }

    const delay = this.retryDelays[retryCount - 1] || 60000;
    const failedRecord = {
      ...record,
      retryCount,
      lastError: error.message,
      nextRetryAt: Date.now() + delay
    };

    const transaction = this.db.transaction(['failed'], 'readwrite');
    const store = transaction.objectStore('failed');
    await promisifyRequest(store.put(failedRecord));

    // 定时重试
    setTimeout(() => this.processQueue(), delay);
  }

  // 处理失败队列
  async processFailedQueue() {
    const now = Date.now();
    const transaction = this.db.transaction(['failed'], 'readonly');
    const store = transaction.objectStore('failed');
    const index = store.index('nextRetryAt');

    const range = IDBKeyRange.upperBound(now);
    const retryable = await promisifyRequest(index.getAll(range));

    for (const record of retryable) {
      // 移回 pending 队列
      await this.moveFailedToPending(record);
    }
  }

  async moveFailedToPending(record) {
    const trans = this.db.transaction(['failed', 'pending'], 'readwrite');
    const failedStore = trans.objectStore('failed');
    const pendingStore = trans.objectStore('pending');

    const { retryCount, lastError, nextRetryAt, ...original } = record;
    await promisifyRequest(pendingStore.put(original));
    await promisifyRequest(failedStore.delete(record.id));
  }

  async removePending(id) {
    const transaction = this.db.transaction(['pending'], 'readwrite');
    const store = transaction.objectStore('pending');
    await promisifyRequest(store.delete(id));
  }

  // 获取队列统计
  async getStats() {
    const trans = this.db.transaction(['pending', 'failed'], 'readonly');
    const pendingStore = trans.objectStore('pending');
    const failedStore = trans.objectStore('failed');

    const [pending, failed] = await Promise.all([
      promisifyRequest(pendingStore.count()),
      promisifyRequest(failedStore.count())
    ]);

    return { pending, failed };
  }
}

// 工具函数：将 IDBRequest 转为 Promise
function promisifyRequest(request) {
  return new Promise((resolve, reject) => {
    request.onsuccess = () => resolve(request.result);
    request.onerror = () => reject(request.error);
  });
}

// 监听网络变化
window.addEventListener('online', () => {
  console.log('网络恢复，开始处理离线队列');
  queue.processQueue();
});

// 初始化使用
const db = await initDB();
const queue = new OfflineMessageQueue(db);

// 发送消息（离线时自动入队）
await queue.enqueue({ text: '你好', roomId: 'room-001' }, { priority: 1 });
```

## 二、大型文件分片存储

场景：需要在前端缓存大量图片或文件，超出单次存储限制时进行分片管理。

```javascript
const FILE_DB_NAME = 'FileCacheDB';
const CHUNK_SIZE = 1024 * 1024; // 1MB 分片

function initFileDB() {
  return new Promise((resolve, reject) => {
    const request = indexedDB.open(FILE_DB_NAME, 1);

    request.onupgradeneeded = (event) => {
      const db = event.target.result;

      // 文件元数据
      const metaStore = db.createObjectStore('fileMeta', { keyPath: 'fileId' });
      metaStore.createIndex('url', 'url', { unique: true });
      metaStore.createIndex('expireAt', 'expireAt', { unique: false });

      // 文件分片
      const chunkStore = db.createObjectStore('chunks', { keyPath: 'chunkId' });
      chunkStore.createIndex('fileId', 'fileId', { unique: false });
    };

    request.onsuccess = () => resolve(request.result);
    request.onerror = () => reject(request.error);
  });
}

class FileCache {
  constructor(db, options = {}) {
    this.db = db;
    this.maxAge = options.maxAge || 7 * 24 * 60 * 60 * 1000; // 默认 7 天
  }

  // 缓存文件
  async cacheFile(url, blob, metadata = {}) {
    const fileId = this.hashUrl(url);
    const chunks = await this.splitBlob(blob);
    const chunkIds = [];

    // 存储分片
    const chunkTrans = this.db.transaction(['chunks'], 'readwrite');
    const chunkStore = chunkTrans.objectStore('chunks');

    for (let i = 0; i < chunks.length; i++) {
      const chunkId = `${fileId}#${i}`;
      chunkIds.push(chunkId);
      await promisifyRequest(chunkStore.put({
        chunkId,
        fileId,
        index: i,
        data: chunks[i],
        size: chunks[i].size
      }));
    }

    // 存储元数据
    const metaTrans = this.db.transaction(['fileMeta'], 'readwrite');
    const metaStore = metaTrans.objectStore('fileMeta');

    await promisifyRequest(metaStore.put({
      fileId,
      url,
      name: metadata.name || url.split('/').pop(),
      mimeType: blob.type,
      totalSize: blob.size,
      chunkCount: chunks.length,
      chunkIds,
      createdAt: Date.now(),
      expireAt: Date.now() + this.maxAge,
      metadata
    }));

    return fileId;
  }

  // 读取文件
  async getFile(url) {
    const fileId = this.hashUrl(url);

    const metaTrans = this.db.transaction(['fileMeta'], 'readonly');
    const metaStore = metaTrans.objectStore('fileMeta');
    const meta = await promisifyRequest(metaStore.get(fileId));

    if (!meta) return null;
    if (meta.expireAt < Date.now()) {
      await this.deleteFile(url);
      return null;
    }

    // 读取所有分片
    const chunkTrans = this.db.transaction(['chunks'], 'readonly');
    const chunkStore = chunkTrans.objectStore('chunks');
    const index = chunkStore.index('fileId');
    const chunks = await promisifyRequest(index.getAll(fileId));

    // 按顺序合并
    chunks.sort((a, b) => a.index - b.index);
    const blob = new Blob(chunks.map(c => c.data), { type: meta.mimeType });

    return {
      blob,
      metadata: meta
    };
  }

  // 分片切割
  async splitBlob(blob) {
    const chunks = [];
    for (let offset = 0; offset < blob.size; offset += CHUNK_SIZE) {
      chunks.push(blob.slice(offset, offset + CHUNK_SIZE));
    }
    return chunks;
  }

  // 删除文件
  async deleteFile(url) {
    const fileId = this.hashUrl(url);

    const trans = this.db.transaction(['fileMeta', 'chunks'], 'readwrite');
    const metaStore = trans.objectStore('fileMeta');
    const chunkStore = trans.objectStore('chunks');
    const index = chunkStore.index('fileId');

    const meta = await promisifyRequest(metaStore.get(fileId));
    if (!meta) return;

    // 删除所有分片
    const chunks = await promisifyRequest(index.getAll(fileId));
    for (const chunk of chunks) {
      await promisifyRequest(chunkStore.delete(chunk.chunkId));
    }

    await promisifyRequest(metaStore.delete(fileId));
  }

  // 清理过期文件
  async cleanExpired() {
    const trans = this.db.transaction(['fileMeta'], 'readonly');
    const store = trans.objectStore('fileMeta');
    const index = store.index('expireAt');
    const range = IDBKeyRange.upperBound(Date.now());

    const expired = await promisifyRequest(index.getAll(range));

    for (const meta of expired) {
      await this.deleteFile(meta.url);
    }

    return expired.length;
  }

  // URL 哈希
  hashUrl(url) {
    let hash = 0;
    for (let i = 0; i < url.length; i++) {
      const char = url.charCodeAt(i);
      hash = ((hash << 5) - hash) + char;
      hash = hash & hash;
    }
    return `f_${Math.abs(hash).toString(36)}`;
  }

  // 获取缓存统计
  async getStats() {
    const trans = this.db.transaction(['fileMeta', 'chunks'], 'readonly');
    const metaStore = trans.objectStore('fileMeta');
    const chunkStore = trans.objectStore('chunks');

    const metas = await promisifyRequest(metaStore.getAll());
    const chunks = await promisifyRequest(chunkStore.getAll());

    const totalFileSize = metas.reduce((sum, m) => sum + m.totalSize, 0);
    const totalChunkSize = chunks.reduce((sum, c) => sum + c.size, 0);

    return {
      fileCount: metas.length,
      chunkCount: chunks.length,
      totalFileSize,
      totalChunkSize,
      expiredCount: metas.filter(m => m.expireAt < Date.now()).length
    };
  }
}

// 配合 Service Worker 使用
// sw.js
self.addEventListener('fetch', (event) => {
  const url = event.request.url;

  // 只处理图片请求
  if (!url.match(/\.(png|jpg|jpeg|gif|webp)$/)) return;

  event.respondWith(
    (async () => {
      // 先查 IndexedDB 缓存
      const cache = new FileCache(db);
      const cached = await cache.getFile(url);

      if (cached) {
        return new Response(cached.blob, {
          headers: {
            'Content-Type': cached.metadata.mimeType,
            'X-Cache': 'indexeddb'
          }
        });
      }

      // 网络请求并缓存
      const response = await fetch(event.request);
      const blob = await response.clone().blob();
      await cache.cacheFile(url, blob);

      return response;
    })()
  );
});
```

## 四、实时同步方案

场景：多个标签页共享 IndexedDB 数据，需要实时同步状态。

```javascript
class IndexedDBSync {
  constructor(db, storeName) {
    this.db = db;
    this.storeName = storeName;
    this.channel = new BroadcastChannel(`idb_sync_${db.name}_${storeName}`);
    this.listeners = new Set();

    this.channel.onmessage = (event) => {
      const { type, key, data } = event.data;
      this.listeners.forEach(cb => cb(type, key, data));
    };
  }

  onChange(callback) {
    this.listeners.add(callback);
    return () => this.listeners.delete(callback);
  }

  async put(key, data) {
    const transaction = this.db.transaction([this.storeName], 'readwrite');
    const store = transaction.objectStore(this.storeName);
    await promisifyRequest(store.put(data, key));

    // 广播变更
    this.channel.postMessage({ type: 'put', key, data });
  }

  async delete(key) {
    const transaction = this.db.transaction([this.storeName], 'readwrite');
    const store = transaction.objectStore(this.storeName);
    await promisifyRequest(store.delete(key));

    this.channel.postMessage({ type: 'delete', key });
  }

  async get(key) {
    const transaction = this.db.transaction([this.storeName], 'readonly');
    const store = transaction.objectStore(this.storeName);
    return promisifyRequest(store.get(key));
  }
}

// 使用：跨标签页状态同步
const sync = new IndexedDBSync(db, 'appState');

// 标签页 A
await sync.put('uiConfig', { theme: 'dark', sidebar: true });

// 标签页 B 自动收到通知
sync.onChange((type, key, data) => {
  if (key === 'uiConfig') {
    applyTheme(data.theme);
  }
});
```

## 五、性能优化清单

| 优化项 | 做法 |
|--------|------|
| 批量操作 | 使用 `bulkAdd` / `bulkPut` 替代循环单条插入 |
| 索引设计 | 只为查询频繁的字段创建索引，索引会增加写入开销 |
| 事务合并 | 将多个操作放在同一个事务中 |
| 分页加载 | 大数据集使用 `offset` + `limit` 或游标 |
| 数据压缩 | 存储前用 `CompressionStream` 压缩大文本 |
| 定期清理 | 设置 TTL，定时清理过期数据 |
| 分片存储 | 大文件分片存储，避免单条记录过大 |

```javascript
// 批量插入优化
async function batchInsert(storeName, items, batchSize = 1000) {
  for (let i = 0; i < items.length; i += batchSize) {
    const batch = items.slice(i, i + batchSize);
    const transaction = db.transaction([storeName], 'readwrite');
    const store = transaction.objectStore(storeName);

    for (const item of batch) {
      store.put(item);
    }

    await promisifyRequest(transaction); // 等待事务完成
  }
}

async function compressAndStore(store, key, data) {

  const stream = new Blob([json]).stream()
    .pipeThrough(new CompressionStream('gzip'));
  const compressed = await new Response(stream).blob();
```

### 数据压缩：CompressionStream API

`CompressionStream` 是 Web Streams API 的一部分，提供基于流的**原生压缩/解压缩能力**。它允许你将任意可读流通过转换流进行压缩，无需引入第三方库即可实现 `gzip`、`deflate` 等格式的高效压缩。

#### 支持的压缩格式

| 格式 | 说明 | 浏览器支持 | Node.js 支持 |
|------|------|-----------|-------------|
| `gzip` | 最常用的压缩格式，兼容性好 | Chrome/Edge 80+、Firefox 113+、Safari 16.4+ | v17.0.0+，v18.0.0+ 全局可用 |
| `deflate` | 标准 deflate 压缩 | Chrome/Edge 80+、Firefox 113+、Safari 16.4+ | v17.0.0+ |
| `deflate-raw` | 无 zlib 包头的原始 deflate | Chrome/Edge 80+、Firefox 113+、Safari 16.4+ | v21.2.0、v20.12.0+ |
| `brotli` | 高压缩率，适合文本 | Chrome/Edge、Firefox 146+、Safari 17.4+ | v24.7.0、v22.20.0+ |

#### 浏览器环境

现代浏览器已将 `CompressionStream` / `DecompressionStream` 作为全局对象提供。使用前建议做特性检测：

```javascript
if (typeof CompressionStream === 'undefined') {
  throw new Error('当前浏览器不支持 CompressionStream，请降级到 pako/fflate 等库');
}
```

#### Node.js 环境

`CompressionStream` 同样可以在 Node.js 中使用：

- **Node.js v17.0.0**：首次引入 `CompressionStream` 类；
- **Node.js v18.0.0**：该类被暴露到全局对象（`global.CompressionStream`），无需额外导入即可使用；
- **Node.js 22.5.0+**：内置了 `node:sqlite` 等模块，同时也完整支持 Web Streams API。

因此，在 Node.js 中你既可以写浏览器同构的压缩代码，也可以使用更成熟的 `node:zlib` 模块：

```javascript
// Node.js 中使用原生 CompressionStream（v18+）
const compressed = await new Response(
  new Blob(['hello world']).stream().pipeThrough(new CompressionStream('gzip'))
).bytes();

// 或者使用 node:zlib（更推荐在纯 Node 场景使用）
import { gzipSync, gunzipSync } from 'node:zlib';
const buf = gzipSync(Buffer.from('hello world'));
```

#### 基础用法

`CompressionStream` 是一个转换流（TransformStream），具有 `writable`（写入原始数据）和 `readable`（读取压缩数据）两个端口：

```javascript
const stream = new CompressionStream('gzip');

// 写入待压缩数据
const writer = stream.writable.getWriter();
const encoder = new TextEncoder();
writer.write(encoder.encode('Hello, IndexedDB!'));
writer.close();

// 读取压缩后的数据
const reader = stream.readable.getReader();
const chunks = [];
while (true) {
  const { done, value } = await reader.read();
  if (done) break;
  chunks.push(value);
}
const compressed = new Uint8Array(chunks.flatMap(c => Array.from(c)));
```

更简洁的写法是借助 `Blob.stream()` 和 `Response`：

```javascript
async function gzip(data) {
  const stream = new Blob([data]).stream().pipeThrough(new CompressionStream('gzip'));
  return await new Response(stream).bytes();
}

async function gunzip(compressed) {
  const stream = new Blob([compressed]).stream().pipeThrough(new DecompressionStream('gzip'));
  return await new Response(stream).text();
}

// 使用
const original = JSON.stringify({ /* 大量数据 */ });
const compressed = await gzip(original);
const restored = await gunzip(compressed);
```

#### 与 IndexedDB 结合的完整示例

```javascript
async function compressAndStore(db, storeName, key, data) {
  const json = JSON.stringify(data);
  const compressed = await new Response(
    new Blob([json]).stream().pipeThrough(new CompressionStream('gzip'))
  ).blob();

  const transaction = db.transaction([storeName], 'readwrite');
  const store = transaction.objectStore(storeName);
  await promisifyRequest(store.put(compressed, key));
}

async function readAndDecompress(db, storeName, key) {
  const transaction = db.transaction([storeName], 'readonly');
  const store = transaction.objectStore(storeName);
  const compressed = await promisifyRequest(store.get(key));

  if (!compressed) return null;

  const text = await new Response(
    compressed.stream().pipeThrough(new DecompressionStream('gzip'))
  ).text();

  return JSON.parse(text);
}
```

#### 使用建议

- **文本类大数据**（日志、JSON、配置）压缩收益最高，通常可减少 50% ~ 90% 体积；
- **二进制数据**（图片、视频）本身已压缩，再使用 `gzip` 收益甚微，不建议压缩；
- 在**浏览器端**优先使用 `gzip`，兼容性最好；
- 在**Node.js 端**如果追求压缩率，可尝试 `brotli`；如果追求速度和兼容性，使用 `gzip`；
- 压缩/解压缩是 CPU 密集型操作，大数据量建议放在 Web Worker 中执行，避免阻塞主线程。
