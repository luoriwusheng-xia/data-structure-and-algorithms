# 06 - 最小可落地实战

前面几篇讲的是原则，这一篇给一套最小可运行的思路：做一个企业调用的订单查询接口，要求具备下面几项能力：

- Node.js 26
- ESM
- 应用身份识别
- HMAC 签名校验
- 请求日志
- 统一响应结构

## 一、目标接口

```text
POST /partner/v1/orders/query
```

请求头：

```text
Content-Type: application/json
X-App-Id: app_live_demo
X-Timestamp: 1720000000000
X-Nonce: nonce-001
X-Signature: <hmac signature>
```

请求体：

```json
{
  "orderNo": "ORDER_10001"
}
```

### `src/core/signature.js`

```js
import { createHash, createHmac, timingSafeEqual } from 'node:crypto';

function sha256Hex(text) {
  return createHash('sha256').update(text).digest('hex');
}

export function buildCanonicalString({ method, path, timestamp, nonce, bodyText }) {
  const bodyHash = sha256Hex(bodyText);
  return [method.toUpperCase(), path, timestamp, nonce, bodyHash].join('\n');
}

export function sign({ method, path, timestamp, nonce, bodyText }, secret) {
  const canonical = buildCanonicalString({ method, path, timestamp, nonce, bodyText });
  return createHmac('sha256', secret).update(canonical).digest('hex');
}

export function verify(input, secret, signature) {
  const expected = sign(input, secret);
  const left = Buffer.from(expected, 'utf8');
  const right = Buffer.from(signature, 'utf8');

  if (left.length !== right.length) {
    return false;
  }

  return timingSafeEqual(left, right);
}
```

### `src/middleware/request-log.js`

```js
import { randomUUID } from 'node:crypto';
import { logger } from '../infra/logger.js';

export function requestLogMiddleware(req, res, next) {
  const requestId = req.headers['x-request-id'] ?? randomUUID();
  const startedAt = Date.now();

  req.requestId = requestId;
  res.setHeader('x-request-id', requestId);

  res.on('finish', () => {
    logger.info('request completed', {
      requestId,
      method: req.method,
      path: req.originalUrl,
      statusCode: res.statusCode,
      durationMs: Date.now() - startedAt
    });
  });

  next();
}
```

### `src/middleware/auth.js`

```js
import { AppError } from '../core/errors.js';

const appStore = new Map([
  ['app_live_demo', {
    appId: 'app_live_demo',
    tenantId: 'tenant_demo',
    secret: 'platform-demo-secret'
  }]
]);

export function authMiddleware(req, res, next) {
  const appId = req.headers['x-app-id'];
  if (!appId) {
    return next(new AppError('missing x-app-id', {
      code: 'APP_ID_MISSING',
      status: 401
    }));
  }

  const app = appStore.get(appId);
  if (!app) {
    return next(new AppError('app not found', {
      code: 'APP_NOT_FOUND',
      status: 401
    }));
  }

  req.appAuth = app;
  next();
}
```

### `src/middleware/signature.js`

```js
import { AppError } from '../core/errors.js';
import { verify } from '../core/signature.js';

const ALLOWED_SKEW_MS = 5 * 60 * 1000;

export function signatureMiddleware(req, res, next) {
  const timestamp = req.headers['x-timestamp'];
  const nonce = req.headers['x-nonce'];
  const signature = req.headers['x-signature'];

  if (!timestamp || !nonce || !signature) {
    return next(new AppError('missing signature headers', {
      code: 'SIGNATURE_HEADERS_MISSING',
      status: 400
    }));
  }

  const drift = Math.abs(Date.now() - Number(timestamp));
  if (drift > ALLOWED_SKEW_MS) {
    return next(new AppError('timestamp expired', {
      code: 'TIMESTAMP_EXPIRED',
      status: 401
    }));
  }

  const bodyText = JSON.stringify(req.body ?? {});
  const valid = verify(
    {
      method: req.method,
      path: req.path,
      timestamp,
      nonce,
      bodyText
    },
    req.appAuth.secret,
    signature
  );

  if (!valid) {
    return next(new AppError('signature invalid', {
      code: 'SIGNATURE_INVALID',
      status: 401
    }));
  }

  next();
}
```

### `src/app.js`

```js
import express from 'express';
import { AppError } from './core/errors.js';
import { fail } from './core/response.js';
import { requestLogMiddleware } from './middleware/request-log.js';
import { authMiddleware } from './middleware/auth.js';
import { signatureMiddleware } from './middleware/signature.js';
import { partnerRoutes } from './routes/partner.routes.js';

export function createApp() {
  const app = express();

  app.use(express.json());
  app.use(requestLogMiddleware);
  app.use('/partner', authMiddleware, signatureMiddleware, partnerRoutes);

  app.use((err, req, res, next) => {
    if (err instanceof AppError) {
      return res.status(err.status).json(fail(err));
    }

    const unknown = new AppError('internal error');
    return res.status(500).json(fail(unknown));
  });

  return app;
}
```

## 五、客户端签名示例

第三方调用方需要先生成签名：

```js
import { createHash, createHmac } from 'node:crypto';

function sha256Hex(text) {
  return createHash('sha256').update(text).digest('hex');
}

function signRequest({ method, path, timestamp, nonce, body }, secret) {
  const bodyText = JSON.stringify(body);
  const bodyHash = sha256Hex(bodyText);
  const canonical = [method.toUpperCase(), path, timestamp, nonce, bodyHash].join('\n');
  return createHmac('sha256', secret).update(canonical).digest('hex');
}

const body = { orderNo: 'ORDER_10001' };
const timestamp = String(Date.now());
const nonce = 'nonce-001';

const signature = signRequest({
  method: 'POST',
  path: '/v1/orders/query',
  timestamp,
  nonce,
  body
}, 'platform-demo-secret');

console.log(signature);
```

## 七、这个案例故意没有展开的内容

为了保持最小可落地，这里还没有展开：

- Redis nonce 防重放
- 幂等键
- 租户权限表
- 数据库仓储
- Webhook 回调
- OpenTelemetry / tracing

但骨架已经具备扩展这些能力的位置。

## 八、本篇结论

一个对外 API 平台的第一版，不需要先做成巨型框架；但必须从第一天开始把这些东西摆对位置：

- 身份识别
- 签名校验
- 统一响应
- 结构化日志
- 清晰目录结构

这比单纯“能返回 JSON”重要得多。
