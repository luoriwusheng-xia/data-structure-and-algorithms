# Express 5.x 项目实战：从零构建企业级 API 服务

## 一、Express 5.x 核心变化

Express 5.x 在错误处理层面做了根本性的改进，这是从 4.x 迁移时最需要关注的点。

### 1.1 Promise 自动捕获

Express 5 会自动捕获中间件返回的 rejected promise，无需再手动调用 `next(err)`。

```javascript
// Express 4.x - 必须包裹 try/catch 或调用 next(err)
app.get('/users', async (req, res, next) => {
  try {
    const users = await User.findAll();
    res.json(users);
  } catch (err) {
    next(err); // 不能遗漏
  }
});

// Express 5.x - 自动传播错误
app.get('/users', async (req, res) => {
  const users = await User.findAll(); // reject 自动走到 error handler
  res.json(users);
});
```

```javascript
// 空 reject 也会被捕获，Express 会提供默认错误消息
app.use((req, res) => {
  return Promise.reject(); // err.message = 'Rejected promise'
});
```

### 1.2 移除 path-to-regexp

Express 5 移除了 `path-to-regexp` 依赖，路由匹配行为更加确定，不再需要担心正则表达式特殊字符的转义问题。

```javascript
// Express 5 中以下写法不再支持，需要改用参数化路由
// app.get('/files/*.txt', handler) // ❌ 不再支持通配符

// 正确写法
app.get('/files/:name.txt', (req, res) => {
  // req.params.name
});
```

### 2.2 入口文件设计

```javascript
// src/app.js
const express = require('express');
const helmet = require('helmet');
const cors = require('cors');
const compression = require('compression');
const routes = require('./routes');
const { errorHandler, notFound } = require('./middlewares/error');

const app = express();

// 安全与性能中间件
app.use(helmet());
app.use(cors({
  origin: process.env.CORS_ORIGIN?.split(',') || [],
  credentials: true,
}));
app.use(compression());

// 解析层
app.use(express.json({ limit: '10kb' }));
app.use(express.urlencoded({ extended: true, limit: '10kb' }));

// 健康检查（负载均衡器使用）
app.get('/health', (req, res) => {
  res.status(200).json({ status: 'ok', timestamp: new Date().toISOString() });
});

// API 路由
app.use('/api/v1', routes);

// 404 与错误处理
app.use(notFound);
app.use(errorHandler);

module.exports = app;
```

```javascript
// bin/server.js
const app = require('../src/app');
const { connectDB } = require('../src/config/database');

const PORT = process.env.PORT || 3000;

// 优雅关闭
const server = app.listen(PORT, async () => {
  await connectDB();
  console.log(`Server running on port ${PORT}`);
});

const gracefulShutdown = (signal) => {
  console.log(`Received ${signal}. Shutting down gracefully...`);
  server.close(async () => {
    await disconnectDB();
    process.exit(0);
  });
};

process.on('SIGTERM', () => gracefulShutdown('SIGTERM'));
process.on('SIGINT', () => gracefulShutdown('SIGINT'));
```

### 2.4 控制器层（薄控制器原则）

```javascript
// src/controllers/user.controller.js
const userService = require('../services/user.service');
const { asyncHandler } = require('../utils/async');

exports.create = async (req, res) => {
  // Express 5: async 函数抛出的错误自动进入 error handler
  const user = await userService.create(req.body);
  res.status(201).json({ data: user });
};

exports.getById = async (req, res) => {
  const user = await userService.findById(req.params.id);
  if (!user) {
    return res.status(404).json({ error: 'User not found' });
  }
  res.json({ data: user });
};

exports.update = async (req, res) => {
  const user = await userService.update(req.params.id, req.body);
  res.json({ data: user });
};
```

### 2.5 Service 层（业务逻辑核心）

```javascript
// src/services/user.service.js
const User = require('../models/user.model');
const { hashPassword } = require('../utils/crypto');
const { ConflictError } = require('../errors');

class UserService {
  async create(data) {
    const exists = await User.findByEmail(data.email);
    if (exists) {
      throw new ConflictError('Email already registered');
    }

    const hashedPassword = await hashPassword(data.password);
    return User.create({
      ...data,
      password: hashedPassword,
    });
  }

  async findById(id) {
    return User.findById(id, {
      exclude: ['password', 'deleted_at'],
    });
  }

  async update(id, data) {
    // 不允许通过此接口更新敏感字段
    const safeFields = ['nickname', 'avatar', 'phone'];
    const updates = Object.keys(data)
      .filter((key) => safeFields.includes(key))
      .reduce((obj, key) => ({ ...obj, [key]: data[key] }), {});

    return User.update(id, updates);
  }
}

module.exports = new UserService();
```

## 三、自定义错误体系

```javascript
// src/errors/index.js
class AppError extends Error {
  constructor(message, statusCode = 500, code = 'INTERNAL_ERROR') {
    super(message);
    this.statusCode = statusCode;
    this.code = code;
    this.isOperational = true;
    Error.captureStackTrace(this, this.constructor);
  }
}

class ValidationError extends AppError {
  constructor(message, fields = []) {
    super(message, 400, 'VALIDATION_ERROR');
    this.fields = fields;
  }
}

class AuthenticationError extends AppError {
  constructor(message = 'Unauthorized') {
    super(message, 401, 'UNAUTHORIZED');
  }
}

class ForbiddenError extends AppError {
  constructor(message = 'Forbidden') {
    super(message, 403, 'FORBIDDEN');
  }
}

class NotFoundError extends AppError {
  constructor(resource = 'Resource') {
    super(`${resource} not found`, 404, 'NOT_FOUND');
  }
}

class ConflictError extends AppError {
  constructor(message) {
    super(message, 409, 'CONFLICT');
  }
}

module.exports = {
  AppError,
  ValidationError,
  AuthenticationError,
  ForbiddenError,
  NotFoundError,
  ConflictError,
};
```

```javascript
// src/middlewares/error.js
const { AppError } = require('../errors');

const notFound = (req, res) => {
  res.status(404).json({
    error: 'Not found',
    path: req.url,
  });
};

const errorHandler = (err, req, res, next) => {
  // 开发环境输出完整堆栈
  const isDev = process.env.NODE_ENV === 'development';

  if (err instanceof AppError && err.isOperational) {
    return res.status(err.statusCode).json({
      error: err.message,
      code: err.code,
      ...(err.fields && { fields: err.fields }),
      ...(isDev && { stack: err.stack }),
    });
  }

  // 未预期的错误，记录后返回通用消息
  console.error('Unexpected error:', err);
  res.status(500).json({
    error: isDev ? err.message : 'Internal server error',
    ...(isDev && { stack: err.stack }),
  });
};

module.exports = { notFound, errorHandler };
```

## 四、请求校验

```javascript
// src/validators/user.validator.js
const { ValidationError } = require('../errors');

const validateCreateUser = (req, res, next) => {
  const { email, password, nickname } = req.body;
  const errors = [];

  if (!email || !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) {
    errors.push({ field: 'email', message: 'Valid email is required' });
  }

  if (!password || password.length < 8) {
    errors.push({ field: 'password', message: 'Password must be at least 8 characters' });
  }

  if (!nickname || nickname.trim().length < 2) {
    errors.push({ field: 'nickname', message: 'Nickname must be at least 2 characters' });
  }

  if (errors.length > 0) {
    return next(new ValidationError('Validation failed', errors));
  }

  next();
};

module.exports = { validateCreateUser };
```
