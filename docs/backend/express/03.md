# Express 5.x + MySQL 8.x 数据库实战

## 一、MySQL 8.x 核心新特性

MySQL 8.0 相比 5.7 有显著的性能和功能提升，在实战中有几个必须掌握的特性：

### 1.1 JSON 原生支持

MySQL 8 提供了原生的 JSON 数据类型，支持索引和高效查询。

```sql
-- 创建含 JSON 列的表
CREATE TABLE products (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    attributes JSON,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_name (name)
);

-- 插入 JSON 数据
INSERT INTO products (name, attributes) VALUES
('iPhone 15', '{"color": "blue", "storage": "256GB", "weight": 171}'),
('MacBook Pro', '{"color": "space_gray", "ram": "16GB", "cpu": "M3"}');

-- JSON 路径查询
SELECT name,
       JSON_UNQUOTE(JSON_EXTRACT(attributes, '$.color')) AS color,
       attributes->>'$.storage' AS storage
FROM products
WHERE JSON_CONTAINS(attributes, '"blue"', '$.color');

-- 多值索引（MySQL 8.0.17+）
CREATE TABLE events (
    id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255),
    tags JSON,
    INDEX idx_tags ((CAST(tags AS CHAR(64) ARRAY)))
);
```

### 1.2 窗口函数（Window Functions）

窗口函数避免了自连接和子查询，显著提升复杂报表性能。

```sql
-- 用户订单排名（每个用户的订单按金额排名）
SELECT
    user_id,
    order_id,
    amount,
    RANK() OVER (PARTITION BY user_id ORDER BY amount DESC) AS rank_num,
    DENSE_RANK() OVER (PARTITION BY user_id ORDER BY amount DESC) AS dense_rank_num,
    ROW_NUMBER() OVER (PARTITION BY user_id ORDER BY created_at DESC) AS row_num
FROM orders;

-- 累计销售额（滚动计算）
SELECT
    date,
    daily_amount,
    SUM(daily_amount) OVER (ORDER BY date ROWS UNBOUNDED PRECEDING) AS cumulative_sum,
    AVG(daily_amount) OVER (ORDER BY date ROWS 6 PRECEDING) AS rolling_7day_avg
FROM daily_sales;

-- 首尾值分析
SELECT
    user_id,
    login_time,
    FIRST_VALUE(login_time) OVER (PARTITION BY user_id ORDER BY login_time) AS first_login,
    LAG(login_time, 1) OVER (PARTITION BY user_id ORDER BY login_time) AS prev_login
FROM user_logins;
```

### 1.3 CTE（公用表表达式）

CTE 使复杂查询更易读，支持递归查询树形结构。

```sql
-- 非递归 CTE：每个商品分类下最贵的商品
WITH ranked_products AS (
    SELECT
        category_id,
        name,
        price,
        RANK() OVER (PARTITION BY category_id ORDER BY price DESC) AS rnk
    FROM products
)
SELECT category_id, name, price
FROM ranked_products
WHERE rnk = 1;

-- 递归 CTE：组织架构树
WITH RECURSIVE org_tree AS (
    -- 锚定成员：顶层节点
    SELECT id, name, parent_id, 0 AS depth
    FROM departments
    WHERE parent_id IS NULL

    UNION ALL

    -- 递归成员
    SELECT d.id, d.name, d.parent_id, ot.depth + 1
    FROM departments d
    INNER JOIN org_tree ot ON d.parent_id = ot.id
)
SELECT * FROM org_tree ORDER BY depth, id;

-- 递归 CTE：无限级分类路径
WITH RECURSIVE category_path AS (
    SELECT id, name, parent_id, CAST(name AS CHAR(255)) AS path
    FROM categories
    WHERE parent_id IS NULL

    UNION ALL

    SELECT c.id, c.name, c.parent_id, CONCAT(cp.path, ' > ', c.name)
    FROM categories c
    INNER JOIN category_path cp ON c.parent_id = cp.id
)
SELECT id, name, path FROM category_path;
```

### 2.2 查询构建器封装

```javascript
// src/utils/db.js
const pool = require('../config/database');

class QueryBuilder {
  constructor(table) {
    this.table = table;
    this.whereConditions = [];
    this.whereValues = [];
    this.orderClause = '';
    this.limitClause = '';
  }

  where(column, operator, value) {
    if (value === undefined) {
      value = operator;
      operator = '=';
    }
    this.whereConditions.push(`${column} ${operator} ?`);
    this.whereValues.push(value);
    return this;
  }

  whereIn(column, values) {
    const placeholders = values.map(() => '?').join(',');
    this.whereConditions.push(`${column} IN (${placeholders})`);
    this.whereValues.push(...values);
    return this;
  }

  whereJson(column, path, value) {
    this.whereConditions.push(`JSON_UNQUOTE(JSON_EXTRACT(${column}, ?)) = ?`);
    this.whereValues.push(path, value);
    return this;
  }

  orderBy(column, direction = 'ASC') {
    this.orderClause = `ORDER BY ${column} ${direction}`;
    return this;
  }

  limit(count, offset) {
    this.limitClause = offset
      ? `LIMIT ${offset}, ${count}`
      : `LIMIT ${count}`;
    return this;
  }

  async select(columns = '*') {
    const colStr = Array.isArray(columns) ? columns.join(', ') : columns;
    const where = this.whereConditions.length
      ? `WHERE ${this.whereConditions.join(' AND ')}`
      : '';

    const sql = `SELECT ${colStr} FROM ${this.table} ${where} ${this.orderClause} ${this.limitClause}`;
    const [rows] = await pool.execute(sql, this.whereValues);
    return rows;
  }

  async first(columns = '*') {
    this.limit(1);
    const rows = await this.select(columns);
    return rows[0] || null;
  }

  async insert(data) {
    const columns = Object.keys(data);
    const values = Object.values(data);
    const placeholders = columns.map(() => '?').join(', ');

    const sql = `INSERT INTO ${this.table} (${columns.join(', ')}) VALUES (${placeholders})`;
    const [result] = await pool.execute(sql, values);
    return { id: result.insertId, ...data };
  }

  async update(data) {
    const setClause = Object.keys(data).map((k) => `${k} = ?`).join(', ');
    const where = this.whereConditions.length
      ? `WHERE ${this.whereConditions.join(' AND ')}`
      : '';

    const sql = `UPDATE ${this.table} SET ${setClause} ${where}`;
    const [result] = await pool.execute(sql, [...Object.values(data), ...this.whereValues]);
    return result.affectedRows;
  }

  async delete() {
    const where = this.whereConditions.length
      ? `WHERE ${this.whereConditions.join(' AND ')}`
      : '';

    const sql = `DELETE FROM ${this.table} ${where}`;
    const [result] = await pool.execute(sql, this.whereValues);
    return result.affectedRows;
  }
}

module.exports = (table) => new QueryBuilder(table);
```

### 2.3 Model 层实现

```javascript
// src/models/user.model.js
const db = require('../utils/db');
const pool = require('../config/database');

class UserModel {
  static table = 'users';

  static async findById(id, options = {}) {
    const { exclude = [] } = options;
    const columns = exclude.length
      ? this.getColumns(exclude)
      : '*';

    return db(this.table)
      .where('id', id)
      .where('deleted_at', 'IS', null)
      .first(columns);
  }

  static async findByEmail(email) {
    return db(this.table)
      .where('email', email)
      .where('deleted_at', 'IS', null)
      .first();
  }

  static async findAll({ page = 1, pageSize = 20, search = '' } = {}) {
    const offset = (page - 1) * pageSize;
    const query = db(this.table)
      .where('deleted_at', 'IS', null)
      .orderBy('created_at', 'DESC')
      .limit(pageSize, offset);

    if (search) {
      query.where('email', 'LIKE', `%${search}%`);
    }

    const [rows, countResult] = await Promise.all([
      query.select(['id', 'email', 'nickname', 'status', 'created_at']),
      pool.execute(
        `SELECT COUNT(*) as total FROM ${this.table} WHERE deleted_at IS NULL`,
      ),
    ]);

    return {
      list: rows,
      pagination: {
        page,
        pageSize,
        total: countResult[0][0].total,
        totalPages: Math.ceil(countResult[0][0].total / pageSize),
      },
    };
  }

  static async create(data) {
    return db(this.table).insert({
      ...data,
      created_at: new Date(),
      updated_at: new Date(),
    });
  }

  static async update(id, data) {
    return db(this.table)
      .where('id', id)
      .update({
        ...data,
        updated_at: new Date(),
      });
  }

  // 软删除
  static async softDelete(id) {
    return db(this.table)
      .where('id', id)
      .update({ deleted_at: new Date() });
  }

  static getColumns(exclude) {
    const all = [
      'id', 'email', 'nickname', 'avatar', 'phone',
      'status', 'created_at', 'updated_at',
    ];
    return all.filter((c) => !exclude.includes(c)).join(', ');
  }
}

module.exports = UserModel;
```

### 2.4 事务封装

```javascript
// src/utils/transaction.js
const pool = require('../config/database');

/**
 * 事务包装器
 * 自动提交/回滚，支持嵌套事务检测
 */
async function withTransaction(fn) {
  const connection = await pool.getConnection();

  try {
    await connection.beginTransaction();

    const result = await fn(connection);

    await connection.commit();
    return result;
  } catch (err) {
    await connection.rollback();
    throw err;
  } finally {
    connection.release();
  }
}

module.exports = { withTransaction };
```

```javascript
// 实战：订单创建（涉及库存扣减、订单写入、日志记录）
const { withTransaction } = require('../utils/transaction');

async function createOrder(userId, items) {
  return withTransaction(async (conn) => {
    // 1. 校验库存并锁定
    for (const item of items) {
      const [rows] = await conn.execute(
        `SELECT stock FROM products WHERE id = ? AND stock >= ? FOR UPDATE`,
        [item.productId, item.quantity],
      );

      if (rows.length === 0) {
        throw new Error(`Insufficient stock for product ${item.productId}`);
      }
    }

    // 2. 扣减库存
    for (const item of items) {
      await conn.execute(
        `UPDATE products SET stock = stock - ? WHERE id = ?`,
        [item.quantity, item.productId],
      );
    }

    // 3. 创建订单
    const totalAmount = items.reduce((sum, i) => sum + i.price * i.quantity, 0);
    const [orderResult] = await conn.execute(
      `INSERT INTO orders (user_id, total_amount, status, created_at) VALUES (?, ?, 'pending', NOW())`,
      [userId, totalAmount],
    );

    const orderId = orderResult.insertId;

    // 4. 写入订单明细
    for (const item of items) {
      await conn.execute(
        `INSERT INTO order_items (order_id, product_id, quantity, unit_price) VALUES (?, ?, ?, ?)`,
        [orderId, item.productId, item.quantity, item.price],
      );
    }

    // 5. 记录操作日志
    await conn.execute(
      `INSERT INTO operation_logs (user_id, action, target, detail, created_at) VALUES (?, 'CREATE_ORDER', ?, ?, NOW())`,
      [userId, orderId, JSON.stringify(items)],
    );

    return { orderId, totalAmount };
  });
}
```

```javascript
async function getColorDistribution() {
```

```javascript
// 使用窗口函数实现高效深分页
async function getOrdersWithRanking(userId, page = 1, pageSize = 20) {
  const [rows] = await pool.execute(`
    WITH ranked_orders AS (
      SELECT
        id,
        total_amount,
        status,
        created_at,
        ROW_NUMBER() OVER (ORDER BY created_at DESC) AS row_num,
        COUNT(*) OVER() AS total_count
      FROM orders
      WHERE user_id = ?
    )
    SELECT *
    FROM ranked_orders
    WHERE row_num BETWEEN ? AND ?
  `, [userId, (page - 1) * pageSize + 1, page * pageSize]);

  return {
    list: rows.map(({ row_num, total_count, ...rest }) => rest),
    total: rows[0]?.total_count || 0,
  };
}
```

```javascript
// 获取部门及其所有下属
async function getDepartmentTree(departmentId) {
  const [rows] = await pool.execute(`
    WITH RECURSIVE sub_departments AS (
      SELECT id, name, parent_id, 0 AS depth
      FROM departments
      WHERE id = ?

      UNION ALL

      SELECT d.id, d.name, d.parent_id, sd.depth + 1
      FROM departments d
      INNER JOIN sub_departments sd ON d.parent_id = sd.id
    )
    SELECT id, name, parent_id, depth
    FROM sub_departments
    ORDER BY depth, id
  `, [departmentId]);

  return rows;
}
```

## 五、数据迁移方案

```javascript
// scripts/migrate.js
const fs = require('fs');
const path = require('path');
const pool = require('../src/config/database');

async function migrate() {
  const migrationsDir = path.join(__dirname, '../migrations');
  const files = fs.readdirSync(migrationsDir).sort();

  const [applied] = await pool.execute(
    'SELECT filename FROM migrations ORDER BY applied_at'
  );
  const appliedSet = new Set(applied.map((r) => r.filename));

  for (const file of files) {
    if (appliedSet.has(file)) continue;

    const sql = fs.readFileSync(path.join(migrationsDir, file), 'utf8');

    await pool.execute(sql);
    await pool.execute(
      'INSERT INTO migrations (filename, applied_at) VALUES (?, NOW())',
      [file],
    );

    console.log(`Applied: ${file}`);
  }
}

migrate().catch(console.error).finally(() => pool.end());
```
