# Express 5.x 中间件：从原理到大型项目实战

## 一、中间件核心机制

Express 的请求处理流水线本质上是一个**中间件栈（middleware stack）**，每个中间件函数接收 `(req, res, next)` 参数，决定是终止响应还是将控制权移交下一个中间件。

```
请求 → 中间件1 → 中间件2 → 路由处理器 → 响应
              ↓
            next() 推进执行
```

### 1.1 执行顺序与挂载方式

```javascript
const app = express();

// 1. 应用级中间件（全局）
app.use(express.json());

// 2. 路由级中间件（路径限定）
app.use('/api', apiRouter);

// 3. 错误处理中间件（必须接收 4 个参数）
app.use((err, req, res, next) => {
  res.status(500).json({ error: err.message });
});
```

挂载顺序决定执行顺序，以下是一个典型的大型项目加载序列：

```javascript
// src/loaders/express.js
module.exports = (app) => {
  // 第 1 层：安全与性能
  app.use(helmet());
  app.use(cors());
  app.use(compression());

  // 第 2 层：请求解析
  app.use(express.json());

  // 第 3 层：上下文注入
  app.use(requestContext()); // traceId, 请求计时

  // 第 4 层：认证
  app.use('/api', authenticate);

  // 第 5 层：业务路由
  app.use('/api/v1', routes);

  // 第 6 层：错误处理
  app.use(notFound);
  app.use(errorHandler);
};
```

## 二、中间件类型详解

### 2.1 应用级中间件

```javascript
// 全局生效
app.use((req, res, next) => {
  console.log(`${req.method} ${req.url}`);
  next();
});

// 路径限定
app.use('/api', (req, res, next) => {
  req.apiVersion = 'v1';
  next();
});

// HTTP 方法限定
app.post('/webhook', verifySignature, handleWebhook);
```

### 2.2 路由级中间件

```javascript
const { Router } = require('express');
const router = Router();

// 路由级别中间件
router.use('/admin', requireAdmin);

// 特定路由链
router.route('/users')
  .get(listUsers)
  .post(validateUser, createUser);

// 参数处理中间件（路由参数预处理）
router.param('userId', async (req, res, next, id) => {
  const user = await User.findById(id);
  if (!user) return next(new NotFoundError('User'));
  req.targetUser = user;
  next();
});

// 使用参数预处理后的路由
router.get('/users/:userId', (req, res) => {
  // req.targetUser 已由 param 中间件注入
  res.json({ data: req.targetUser });
});
```

### 2.3 错误处理中间件

Express 5 最重要的改进：异步函数抛出的错误会自动进入错误处理中间件。

```javascript
// 统一错误处理中间件
const errorHandler = (err, req, res, next) => {
  // 1. 记录错误（带请求上下文）
  logger.error({
    message: err.message,
    stack: err.stack,
    path: req.path,
    method: req.method,
    traceId: req.traceId,
    userId: req.user?.id,
  });

  // 2. 内容协商
  res.status(err.status || 500);

  res.format({
    json: () => {
      res.json({
        error: err.message,
        ...(process.env.NODE_ENV === 'development' && { stack: err.stack }),
      });
    },
    html: () => {
      res.render('error', { error: err });
    },
    default: () => {
      res.type('txt').send(err.message);
    },
  });
};
```

## 三、大型项目中间件实战

### 3.1 分布式追踪中间件（Trace ID）

在微服务架构中，追踪请求链路是排查问题的关键。

```javascript
// src/middlewares/trace.js
const { v4: uuidv4 } = require('uuid');

const traceMiddleware = (req, res, next) => {
  // 优先从上游服务获取（微服务间透传）
  const traceId = req.get('X-Trace-Id') || uuidv4();
  req.traceId = traceId;

  // 响应头中返回，便于客户端关联日志
  res.setHeader('X-Trace-Id', traceId);

  // 记录请求时间
  req.startTime = Date.now();

  res.on('finish', () => {
    const duration = Date.now() - req.startTime;
    console.log(`[${traceId}] ${req.method} ${req.url} ${res.statusCode} ${duration}ms`);
  });

  next();
};

module.exports = traceMiddleware;
```

### 3.2 请求限流中间件（滑动窗口）

```javascript
// src/middlewares/rateLimiter.js
const Redis = require('ioredis');
const redis = new Redis();

/**
 * 滑动窗口限流
 * @param {Object} options
 * @param {number} options.windowMs 窗口大小（毫秒）
 * @param {number} options.max 窗口内最大请求数
 * @param {Function} options.keyGenerator 生成限流键的函数
 */
const slidingWindowRateLimit = (options) => {
  const { windowMs = 60000, max = 100, keyGenerator = (req) => req.ip } = options;

  return async (req, res, next) => {
    const key = `rate_limit:${keyGenerator(req)}`;
    const now = Date.now();
    const windowStart = now - windowMs;

    try {
      const pipeline = redis.pipeline();

      // 移除窗口外的请求记录
      pipeline.zremrangebyscore(key, 0, windowStart);
      // 统计当前窗口内的请求数
      pipeline.zcard(key);
      // 添加当前请求
      pipeline.zadd(key, now, `${now}-${Math.random()}`);
      // 设置 key 过期时间
      pipeline.pexpire(key, windowMs);

      const results = await pipeline.exec();
      const currentCount = results[1][1];

      // 设置响应头（客户端可见）
      res.setHeader('X-RateLimit-Limit', max);
      res.setHeader('X-RateLimit-Remaining', Math.max(0, max - currentCount));

      if (currentCount > max) {
        res.setHeader('Retry-After', Math.ceil(windowMs / 1000));
        return res.status(429).json({
          error: 'Too many requests',
          code: 'RATE_LIMITED',
        });
      }

      next();
    } catch (err) {
      // Redis 故障时，选择放行或降级（根据业务要求）
      next();
    }
  };
};

// 使用：不同接口不同限流策略
app.use('/api/login', slidingWindowRateLimit({ windowMs: 15 * 60 * 1000, max: 5 }));
app.use('/api', slidingWindowRateLimit({ windowMs: 60 * 1000, max: 100 }));
```

### 3.3 多租户上下文中间件（SaaS 场景）

```javascript
// src/middlewares/tenant.js
const { AuthenticationError } = require('../errors');

/**
 * 多租户上下文隔离
 * 从子域名或 header 中解析租户标识，注入到请求上下文
 */
const tenantMiddleware = async (req, res, next) => {
  // 方式 1：从子域名解析（tenant1.example.com）
  // const host = req.get('host');
  // const tenantId = host.split('.')[0];

  // 方式 2：从 Header 解析
  const tenantId = req.get('X-Tenant-Id');

  if (!tenantId) {
    return next(new AuthenticationError('Tenant ID required'));
  }

  // 校验租户存在且有效
  const tenant = await Tenant.findById(tenantId);
  if (!tenant || tenant.status !== 'active') {
    return next(new AuthenticationError('Invalid or inactive tenant'));
  }

  // 注入租户上下文（后续所有数据库查询自动附加 tenant_id 过滤）
  req.tenant = {
    id: tenant.id,
    name: tenant.name,
    dbSchema: tenant.schema_name, // 不同租户不同 schema
    plan: tenant.plan, // 套餐级别影响功能权限
  };

  // 数据库连接池注入（动态 schema 切换）
  req.db = getConnectionForTenant(tenant.id);

  next();
};

module.exports = tenantMiddleware;
```

### 3.4 请求幂等中间件（防止重复提交）

```javascript
// src/middlewares/idempotency.js
const crypto = require('crypto');

/**
 * 幂等性中间件
 * 通过 Idempotency-Key 头保证同一请求只执行一次
 * 适用于支付、下单等关键操作
 */
const idempotencyMiddleware = (options = {}) => {
  const { ttl = 24 * 60 * 60 } = options; // 默认 24 小时

  return async (req, res, next) => {
    const key = req.get('Idempotency-Key');

    // 非幂等请求直接放行
    if (!key) return next();

    const cacheKey = `idempotency:${req.user?.id || req.ip}:${key}`;

    // 检查是否已有结果
    const cached = await redis.get(cacheKey);
    if (cached) {
      const response = JSON.parse(cached);
      return res.status(response.status).json(response.body);
    }

    // 拦截 res.send/res.json 缓存响应
    const originalJson = res.json.bind(res);
    res.json = (body) => {
      if (res.statusCode < 400) {
        redis.setex(cacheKey, ttl, JSON.stringify({
          status: res.statusCode,
          body,
        }));
      }
      return originalJson(body);
    };

    next();
  };
};

// 使用
app.post('/api/orders', authenticate, idempotencyMiddleware(), createOrder);
```

### 3.5 请求审计日志中间件

```javascript
// src/middlewares/audit.js
/**
 * 审计日志中间件
 * 记录敏感操作的变更前/后状态，满足合规要求
 */
const auditLog = (action, options = {}) => {
  const { logRequestBody = false, logResponseBody = false } = options;

  return (req, res, next) => {
    const logEntry = {
      timestamp: new Date().toISOString(),
      action,
      userId: req.user?.id,
      tenantId: req.tenant?.id,
      traceId: req.traceId,
      method: req.method,
      path: req.path,
      ip: req.ip,
      userAgent: req.get('user-agent'),
    };

    if (logRequestBody) {
      logEntry.requestBody = sanitizeBody(req.body);
    }

    // 记录原始状态（用于更新/删除操作）
    if (req.targetUser) {
      logEntry.beforeState = { ...req.targetUser };
    }

    // 响应完成后记录
    res.on('finish', () => {
      logEntry.statusCode = res.statusCode;
      logEntry.duration = Date.now() - req.startTime;

      // 异步写入审计日志（不阻塞响应）
      auditService.write(logEntry).catch(console.error);
    });

    next();
  };
};

// 使用
app.delete('/api/users/:userId', authenticate, requireAdmin, auditLog('DELETE_USER'), deleteUser);
```

### 3.6 缓存中间件（响应缓存）

```javascript
// src/middlewares/cache.js
/**
 * 响应缓存中间件
 * 适用于读多写少、数据变化不频繁的场景（配置、分类列表等）
 */
const cacheMiddleware = (options = {}) => {
  const { ttl = 300, keyGenerator = (req) => req.originalUrl } = options;

  return async (req, res, next) => {
    // POST/PUT/DELETE 不缓存
    if (!['GET', 'HEAD'].includes(req.method)) {
      return next();
    }

    const cacheKey = `cache:${keyGenerator(req)}`;
    const cached = await redis.get(cacheKey);

    if (cached) {
      res.setHeader('X-Cache', 'HIT');
      return res.json(JSON.parse(cached));
    }

    // 拦截响应并缓存
    const originalJson = res.json.bind(res);
    res.json = (body) => {
      res.setHeader('X-Cache', 'MISS');
      redis.setex(cacheKey, ttl, JSON.stringify(body));
      return originalJson(body);
    };

    next();
  };
};

// 使用：商品分类列表缓存 10 分钟
app.get('/api/categories', cacheMiddleware({ ttl: 600 }), listCategories);
```

### 3.7 请求大小限制与文件上传中间件

```javascript
// src/middlewares/upload.js
const multer = require('multer');

// 分片上传场景（大文件）
const uploadConfig = multer({
  storage: multer.memoryStorage(), // 或自定义 S3/OSS 存储
  limits: {
    fileSize: 10 * 1024 * 1024, // 10MB
    files: 5,
  },
  fileFilter: (req, file, cb) => {
    // 白名单校验
    const allowed = ['image/jpeg', 'image/png', 'application/pdf'];
    if (!allowed.includes(file.mimetype)) {
      return cb(new Error('File type not allowed'), false);
    }
    cb(null, true);
  },
});

// 路由中使用
app.post('/api/upload', authenticate, uploadConfig.array('files', 5), handleUpload);
```

## 四、中间件编写最佳实践

### 4.1 错误处理模式

```javascript
// ❌ 错误：Promise reject 未被捕获（Express 4 的问题）
app.use(async (req, res, next) => {
  const data = await fetchData(); // reject 会导致请求挂起
  req.data = data;
  next();
});

// ✅ 正确：Express 5 自动捕获
app.use(async (req, res, next) => {
  const data = await fetchData();
  req.data = data;
  next();
});

// ✅ 兼容模式：显式 try/catch（跨版本兼容）
app.use(async (req, res, next) => {
  try {
    const data = await fetchData();
    req.data = data;
    next();
  } catch (err) {
    next(err);
  }
});
```

### 4.2 条件短路模式

```javascript
// 条件中间件：根据配置动态启用
const conditionalMiddleware = (condition, middleware) => {
  return (req, res, next) => {
    if (condition(req)) {
      return middleware(req, res, next);
    }
    next();
  };
};

// 使用：仅对生产环境启用详细日志
app.use(conditionalMiddleware(
  () => process.env.NODE_ENV === 'production',
  detailedLogger
));
```

### 4.3 中间件组合（可复用链）

```javascript
// 将常用中间件组合为可复用的管道
const compose = (...middlewares) => {
  return (req, res, next) => {
    let index = 0;

    const dispatch = (i) => {
      if (i >= middlewares.length) return next();
      const mw = middlewares[i];
      return mw(req, res, () => dispatch(i + 1));
    };

    dispatch(0);
  };
};

// 定义标准 API 管道
const apiPipeline = compose(
  authenticate,
  requireActiveUser,
  rateLimit({ max: 100 })
);

// 应用到路由
app.use('/api/orders', apiPipeline, orderRoutes);
app.use('/api/invoices', apiPipeline, invoiceRoutes);
```

## 五、Express 5 中间件注意事项

| 场景 | Express 4 | Express 5 |
|------|-----------|-----------|
| async 函数抛错 | 请求挂起 | 自动进入 error handler |
| Promise.reject() | 未处理 | 默认错误消息 "Rejected promise" |
| 通配符路由 | `/*` 支持 | 不再支持，改用参数化 |
| 路径中的 `?` | 作为正则量词 | 字面量匹配 |
