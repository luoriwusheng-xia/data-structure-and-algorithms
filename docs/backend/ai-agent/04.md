# 实战：自动化数据处理 Agent

本章构建一个能够自主执行数据处理任务的 Agent，典型场景：日报自动生成、数据清洗、报表汇总。

## 需求场景

每天早上 9 点，Agent 需要：

1. 从多个数据源（数据库、API、Excel）拉取昨日数据
2. 进行数据清洗和计算
3. 生成图表和文字分析
4. 发送到企业微信群

## 架构设计

```
┌─────────────┐    ┌─────────────┐    ┌─────────────┐

```

```typescript
// src/tools/database.ts
import { tool } from '@langchain/core/tools'
import { z } from 'zod'
import { Pool } from 'pg'

const pgPool = new Pool({ connectionString: process.env.DATABASE_URL })

export const queryDatabaseTool = tool(
  async ({ sql }) => {
    try {
      const result = await pgPool.query(sql)
      return JSON.stringify(result.rows.slice(0, 100)) // 限制返回量
    } catch (e: any) {
      return `查询失败：${e.message}`
    }
  },
  {
    name: 'query_database',
    description: '执行 SQL 查询，返回 JSON 格式结果',
    schema: z.object({
      sql: z.string().describe('SELECT 语句，禁止 UPDATE/DELETE'),
    }),
  }

// src/tools/file.ts
import { readFileSync, writeFileSync } from 'fs'

export const readExcelTool = tool(
  async ({ filePath, sheetName }) => {
    const xlsx = await import('xlsx')
    const workbook = xlsx.readFile(filePath)
    const sheet = workbook.Sheets[sheetName || workbook.SheetNames[0]]
    const data = xlsx.utils.sheet_to_json(sheet)
    return JSON.stringify(data.slice(0, 50))
  },
  {
    name: 'read_excel',
    description: '读取 Excel 文件',
    schema: z.object({
      filePath: z.string(),
      sheetName: z.string().optional(),
    }),
  }
)

export const writeCsvTool = tool(
  async ({ data, filePath }) => {
    const { stringify } = await import('csv-stringify/sync')
    const csv = stringify(JSON.parse(data), { header: true })
    writeFileSync(filePath, csv)
    return `已保存到 ${filePath}`
  },
  {
    name: 'write_csv',
    description: '将数据写入 CSV 文件',
    schema: z.object({
      data: z.string().describe('JSON 数组字符串'),
      filePath: z.string(),
    }),
  }
)

// src/tools/chart.ts
import { createCanvas } from 'canvas'

export const generateChartTool = tool(
  async ({ data, type, title, outputPath }) => {
    const canvas = createCanvas(800, 400)
    const ctx = canvas.getContext('2d')

    // 简化版柱状图绘制
    const parsed = JSON.parse(data)
    const values = parsed.map((d: any) => Object.values(d)[1] as number)
    const labels = parsed.map((d: any) => Object.values(d)[0] as string)
    const max = Math.max(...values)

    ctx.fillStyle = '#fff'
    ctx.fillRect(0, 0, 800, 400)

    const barWidth = 700 / values.length
    values.forEach((v, i) => {
      const height = (v / max) * 300
      ctx.fillStyle = '#3b82f6'
      ctx.fillRect(50 + i * barWidth, 350 - height, barWidth - 10, height)

      ctx.fillStyle = '#333'
      ctx.font = '12px sans-serif'
      ctx.fillText(labels[i], 50 + i * barWidth, 370)
    })

    ctx.fillStyle = '#333'
    ctx.font = 'bold 16px sans-serif'
    ctx.fillText(title, 50, 30)

    const buffer = canvas.toBuffer('image/png')
    writeFileSync(outputPath, buffer)
    return `图表已保存：${outputPath}`
  },
  {
    name: 'generate_chart',
    description: '生成柱状图',
    schema: z.object({
      data: z.string().describe('JSON 数组，每项包含 label 和 value'),
      type: z.enum(['bar', 'line', 'pie']).default('bar'),
      title: z.string(),
      outputPath: z.string(),
    }),
  }
)

// src/tools/wechat.ts
export const sendWechatMessageTool = tool(
  async ({ webhook, content, imagePath }) => {
    const formData = new FormData()
    formData.append('msgtype', imagePath ? 'image' : 'text')

    if (imagePath) {
      // 先上传图片获取 media_id
      const imageBuffer = readFileSync(imagePath)
      // 调用企业微信上传接口...
      formData.append('image', new Blob([imageBuffer]))
    } else {
      formData.append('text', JSON.stringify({ content }))
    }

    await fetch(webhook, {
      method: 'POST',
      body: formData,
    })

    return '发送成功'
  },
  {
    name: 'send_wechat',
    description: '发送企业微信机器人消息',
    schema: z.object({
      webhook: z.string().describe('企业微信 webhook 地址'),
      content: z.string().optional(),
      imagePath: z.string().optional(),
    }),
  }
)
```

### 2. Agent 编排

```typescript
// src/agent/daily-report.ts
import { ChatOpenAI } from '@langchain/openai'
import { createReactAgent } from '@langchain/langgraph/prebuilt'
import { HumanMessage } from '@langchain/core/messages'
import {
  queryDatabaseTool,
  readExcelTool,
  writeCsvTool,
  generateChartTool,
  sendWechatMessageTool,
} from '../tools'
import { config } from '../config'

const model = new ChatOpenAI({
  modelName: config.llm.modelName,
  temperature: 0.2,
  openAIApiKey: config.llm.apiKey,
  configuration: { baseURL: config.llm.baseURL },
})

const agent = createReactAgent({
  llm: model,
  tools: [
    queryDatabaseTool,
    readExcelTool,
    writeCsvTool,
    generateChartTool,
    sendWechatMessageTool,
  ],
})

export async function generateDailyReport(date: string) {
  const prompt = `
请完成 ${date} 的日报生成任务，按以下步骤执行：

1. 查询数据库获取昨日订单数据：
   SELECT date_trunc('hour', created_at) as hour, count(*) as order_count, sum(amount) as total_amount
   FROM orders
   WHERE created_at >= '${date}'::date AND created_at < '${date}'::date + interval '1 day'
   GROUP BY 1 ORDER BY 1

2. 将结果保存为 CSV 到 /tmp/daily_${date}.csv

3. 生成柱状图，展示每小时的订单金额，保存到 /tmp/daily_${date}.png

4. 分析数据特点（峰值时段、总单量、总金额），编写 100 字以内的文字总结

5. 发送企业微信消息，包含文字总结和图表
   webhook: ${process.env.WECHAT_WEBHOOK}

请逐步执行，遇到错误请重试一次。
`

  const result = await agent.invoke({
    messages: [new HumanMessage(prompt)],
  })

  return result.messages.at(-1)?.content
}
```

### 3. 调度与重试

```typescript
// src/scheduler.ts
import { CronJob } from 'cron'
import { generateDailyReport } from './agent/daily-report'

// 每天早上 9:00 执行
const dailyJob = new CronJob(
  '0 9 * * *',
  async () => {
    const yesterday = new Date()
    yesterday.setDate(yesterday.getDate() - 1)
    const dateStr = yesterday.toISOString().split('T')[0]

    console.log(`[${new Date().toISOString()}] 开始生成日报：${dateStr}`)

    try {
      const result = await generateDailyReport(dateStr)
      console.log('日报生成完成：', result)
    } catch (error) {
      console.error('日报生成失败：', error)
      // 发送告警通知
      await sendAlert(`日报生成失败：${error}`)
    }
  },
  null,
  true,
  'Asia/Shanghai'
)

async function sendAlert(message: string) {
  await fetch(process.env.WECHAT_WEBHOOK!, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      msgtype: 'text',
      text: { content: `⚠️ Agent 告警\n${message}` },
    }),
  })
}

console.log('调度器已启动，每日 9:00 执行日报任务')
```

## 错误处理与容错

```typescript
// src/utils/retry.ts
export async function withRetry<T>(
  fn: () => Promise<T>,
  options: { maxRetries?: number; delay?: number } = {}
): Promise<T> {
  const { maxRetries = 3, delay = 1000 } = options

  for (let i = 0; i < maxRetries; i++) {
    try {
      return await fn()
    } catch (error) {
      if (i === maxRetries - 1) throw error
      console.warn(`第 ${i + 1} 次重试...`)
      await new Promise((r) => setTimeout(r, delay * (i + 1)))
    }
  }

  throw new Error('unreachable')
}

// 工具层包装
export const safeQueryDatabase = tool(
  async ({ sql }) => {
    return withRetry(async () => {
      // 实际查询逻辑
      const result = await pgPool.query(sql)
      return JSON.stringify(result.rows.slice(0, 100))
    }, { maxRetries: 2 })
  },
  { name: 'safe_query_database', description: '...', schema: z.object({ sql: z.string() }) }
)
```

## 幂等性保证

Agent 任务可能因为各种原因被重复触发，需要保证幂等：

```typescript
// 使用 Redis 分布式锁 + 任务状态记录
async function executeWithLock(taskId: string, fn: () => Promise<void>) {
  const lockKey = `lock:task:${taskId}`
  const statusKey = `task:status:${taskId}`

  // 1. 检查是否已执行成功
  const status = await redis.get(statusKey)
  if (status === 'success') {
    console.log(`任务 ${taskId} 已执行过，跳过`)
    return
  }

  // 2. 获取分布式锁
  const lock = await redis.set(lockKey, '1', 'EX', 300, 'NX')
  if (!lock) {
    console.log(`任务 ${taskId} 正在执行中，跳过`)
    return
  }

  try {
    await fn()
    await redis.set(statusKey, 'success', 'EX', 86400 * 7) // 保留 7 天
  } finally {
    await redis.del(lockKey)
  }
}
```
