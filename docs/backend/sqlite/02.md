## 1. 索引原理与优化

### 1.1 B-tree 索引

SQLite 使用 B-tree（平衡树）存储索引，所有索引都是 B-tree 结构。

```sql
-- 单列索引
CREATE INDEX idx_name ON users(name);

-- 复合索引（最左前缀原则）
CREATE INDEX idx_age_name ON users(age, name);
-- 有效：WHERE age = 20
-- 有效：WHERE age = 20 AND name = '张三'
-- 无效：WHERE name = '张三'（跳过了最左列）

-- 唯一索引
CREATE UNIQUE INDEX idx_email ON users(email);

-- 部分索引（3.8.0+）
CREATE INDEX idx_active ON users(last_login) WHERE is_active = 1;
-- 只给活跃用户建索引，节省空间

-- 表达式索引（3.9.0+）
CREATE INDEX idx_lower_name ON users(lower(name));
-- 支持大小写不敏感的查询
SELECT * FROM users WHERE lower(name) = 'zhangsan';

-- 覆盖索引
CREATE INDEX idx_cover ON orders(user_id, status, created_at);
-- 如果查询只需要这三列，SQLite 可以直接从索引获取，不回表
```

### 1.2 索引选择与优化

```sql
-- 查看查询是否使用索引
EXPLAIN QUERY PLAN
SELECT * FROM users WHERE name = '张三';
-- 输出：SEARCH users USING INDEX idx_name (name=?)

EXPLAIN QUERY PLAN
SELECT * FROM users WHERE age > 20;
-- 输出可能是：SCAN users（全表扫描）
-- 因为 age > 20 可能匹配大量行，全表扫描更快

-- 强制使用/不使用索引（调试用，生产环境慎用）
SELECT * FROM users INDEXED BY idx_name WHERE name = '张三';
SELECT * FROM users NOT INDEXED WHERE name = '张三';
```

### 1.3 索引失效的场景

```sql
-- 1. 对列进行函数操作
SELECT * FROM users WHERE upper(name) = 'ZHANGSAN';  -- 失效
-- 解决：创建表达式索引
CREATE INDEX idx_upper_name ON users(upper(name));

-- 2. 类型隐式转换
SELECT * FROM users WHERE phone = 13800138000;  -- phone 是 TEXT，失效
-- 解决：保持类型一致
SELECT * FROM users WHERE phone = '13800138000';

-- 3. LIKE 以通配符开头
SELECT * FROM users WHERE name LIKE '%张%';  -- 失效
-- 解决：使用 FTS5 全文搜索

-- 4. OR 条件（部分情况）
SELECT * FROM users WHERE name = '张三' OR age = 20;
-- 可能使用 idx_name，但 age 列无法使用 idx_age
-- 解决：拆分为 UNION
SELECT * FROM users WHERE name = '张三'
UNION
SELECT * FROM users WHERE age = 20;

-- 5. 复合索引未遵循最左前缀
CREATE INDEX idx_a_b_c ON t(a, b, c);
WHERE b = 2 AND c = 3;  -- 无法使用
```

### 1.4 索引维护

```sql
-- 重建索引（整理碎片）
REINDEX;
REINDEX idx_name;  -- 重建指定索引

-- 分析表（更新统计信息，帮助优化器选择索引）
ANALYZE;
ANALYZE users;     -- 只分析 users 表

-- 查看索引统计信息
SELECT * FROM sqlite_stat1;  -- 需要先执行 ANALYZE

-- 删除无用索引
DROP INDEX IF EXISTS idx_unused;

-- 查看所有索引
SELECT name, sql FROM sqlite_schema WHERE type = 'index';
```

## 2. 查询计划与优化

### 2.1 EXPLAIN 命令

```sql
-- EXPLAIN QUERY PLAN：查看执行计划（推荐）
EXPLAIN QUERY PLAN
SELECT u.name, o.total
FROM users u
JOIN orders o ON u.id = o.user_id
WHERE u.age > 18
ORDER BY o.total DESC
LIMIT 10;

-- 输出示例：
-- QUERY PLAN
-- |--SCAN u                    -- 全表扫描 users
-- |--SEARCH o USING INDEX idx_user_id (user_id=?)  -- 索引查找 orders
-- `--USE TEMP B-TREE FOR ORDER BY  -- 需要排序（可以考虑加索引避免）

-- 优化后：给 age 和 total 加索引
CREATE INDEX idx_users_age ON users(age);
CREATE INDEX idx_orders_total ON orders(total);
-- 再看执行计划，SCAN 变成 SEARCH
```

### 2.2 优化技巧

```sql
-- 1. 避免 SELECT *，只取需要的列
SELECT id, name FROM users;  -- 好
SELECT * FROM users;          -- 差，尤其有 BLOB 时

-- 2. 用 EXISTS 替代 IN（子查询）
SELECT * FROM users WHERE id IN (SELECT user_id FROM orders);
-- 改：
SELECT * FROM users u WHERE EXISTS (
    SELECT 1 FROM orders o WHERE o.user_id = u.id
);

-- 3. 用 UNION ALL 替代 OR（索引更友好）
SELECT * FROM users WHERE age = 20 OR status = 1;
-- 改：
SELECT * FROM users WHERE age = 20
UNION ALL
SELECT * FROM users WHERE status = 1 AND age != 20;

-- 4. 批量操作替代循环
-- 差：循环执行 INSERT
-- 好：
INSERT INTO users (name) VALUES ('a'), ('b'), ('c');

-- 5. 合理使用 LIMIT
SELECT * FROM logs ORDER BY created_at DESC LIMIT 100;
-- 避免返回百万行数据

-- 6. 避免隐式排序
-- GROUP BY 会自动排序，如果不需要可以加 ORDER BY NULL
SELECT status, COUNT(*) FROM users GROUP BY status ORDER BY NULL;
```

## 3. 事务与 WAL 模式

### 3.1 事务隔离级别

SQLite 的默认隔离级别是 **SERIALIZABLE**（串行化），但支持 PRAGMA 调整：

```sql
-- 查看当前日志模式
PRAGMA journal_mode;

-- 切换为 WAL 模式（强烈推荐）
PRAGMA journal_mode = WAL;
-- 输出：wal

-- WAL 模式的优势：
-- 1. 读操作不会阻塞写操作
-- 2. 写操作不会阻塞读操作
-- 3. 并发性能大幅提升

-- 其他模式（不建议日常使用）：
PRAGMA journal_mode = DELETE;   -- 默认，事务后删除日志文件
PRAGMA journal_mode = TRUNCATE; -- 事务后截断日志文件
PRAGMA journal_mode = PERSIST;  -- 保留日志文件（归零）
PRAGMA journal_mode = MEMORY;   -- 日志存在内存中（快但不安全）
PRAGMA journal_mode = OFF;      -- 无日志（最快但无回滚能力）
```

### 3.2 WAL 模式的工作原理

```
传统模式（DELETE）：
  写入 → 拷贝原页到 journal → 修改数据库文件 → 提交 → 删除 journal
  读和写互相阻塞

WAL 模式：
  写入 → 追加到 -wal 文件 → 提交
  读取直接从主库文件（不受写入影响）
  定期 checkpoint 将 wal 合并回主库
```

```sql
-- WAL 模式相关配置
PRAGMA wal_autocheckpoint;      -- 查看自动 checkpoint 阈值（默认 1000 页）
PRAGMA wal_autocheckpoint = 500; -- 设置阈值

-- 手动 checkpoint
PRAGMA wal_checkpoint;          -- PASSIVE：不阻塞
PRAGMA wal_checkpoint(RESTART); -- 重启 WAL 文件
PRAGMA wal_checkpoint(TRUNCATE);-- 截断 WAL 文件

-- WAL 文件大小
-- 如果 WAL 文件过大，可以手动 checkpoint
```

### 3.3 事务高级用法

```sql
-- 立即事务（BEGIN IMMEDIATE）
-- 立即获取 RESERVED 锁，避免等待，适合并发环境
BEGIN IMMEDIATE;
    UPDATE accounts SET balance = balance - 100 WHERE id = 1;
    UPDATE accounts SET balance = balance + 100 WHERE id = 2;
COMMIT;

-- 独占事务（BEGIN EXCLUSIVE）
-- 立即获取 EXCLUSIVE 锁，完全独占数据库
BEGIN EXCLUSIVE;
    -- 执行大量 DDL
    ALTER TABLE ...;
    CREATE INDEX ...;
COMMIT;

-- 延迟事务（BEGIN DEFERRED，默认）
-- 第一次访问数据时才获取锁
BEGIN DEFERRED;
    -- 这里还没有锁
    SELECT * FROM users;  -- 现在获取 SHARED 锁
    -- UPDATE ... 会尝试获取 RESERVED 锁
COMMIT;
```

### 3.4 锁机制详解

```
锁状态转换：
UNLOCKED → SHARED → RESERVED → PENDING → EXCLUSIVE

- UNLOCKED：未持有任何锁
- SHARED：可以读，多个连接可以同时持有
- RESERVED：准备写入，只有一个连接能持有
- PENDING：等待写入，阻塞新的 SHARED 锁
- EXCLUSIVE：独占，完全排他
```

```javascript
// Node.js 处理并发写入
const Database = require('better-sqlite3');
const db = new Database('app.db');

// 启用 WAL 模式
db.pragma('journal_mode = WAL');

// 并发写入示例：使用 busyTimeout 自动重试
db.pragma('busy_timeout = 5000');  // 等待 5 秒

// 对于极高并发，考虑队列
try {
    const update = db.prepare('UPDATE counters SET value = value + 1 WHERE id = 1');
    update.run();
} catch (e) {
    if (e.message.includes('busy')) {
        console.log('数据库忙，稍后重试');
    }
}
```

## 4. JSON 支持

SQLite 3.38+ 内置了完整的 JSON 支持，3.53 新增了 `json_array_insert()` 和 `jsonb_array_insert()`。

### 4.1 基本 JSON 函数

```sql
-- 创建存储 JSON 的表
CREATE TABLE users_json (
    id INTEGER PRIMARY KEY,
    profile TEXT  -- 存储 JSON 字符串
);

-- 插入 JSON 数据
INSERT INTO users_json (profile) VALUES ('{
    "name": "张三",
    "age": 25,
    "tags": ["developer", "blogger"],
    "address": {
        "city": "北京",
        "zip": "100000"
    }
}');

-- 提取 JSON 字段
SELECT json_extract(profile, '$.name') AS name FROM users_json;
-- 输出：张三

SELECT json_extract(profile, '$.address.city') AS city FROM users_json;
-- 输出：北京

SELECT json_extract(profile, '$.tags[0]') AS first_tag FROM users_json;
-- 输出：developer

-- 修改 JSON 字段
UPDATE users_json SET profile = json_set(profile, '$.age', 26);
UPDATE users_json SET profile = json_insert(profile, '$.email', 'zs@example.com');  -- 仅当不存在时插入
UPDATE users_json SET profile = json_replace(profile, '$.name', '张三丰');  -- 仅替换已存在的键
UPDATE users_json SET profile = json_remove(profile, '$.address.zip');  -- 删除键

-- 3.53+ 新增：在 JSON 数组中插入元素
UPDATE users_json SET profile = json_array_insert(profile, '$.tags', 'reader');
-- tags 变为：["developer", "blogger", "reader"]
UPDATE users_json SET profile = json_array_insert(profile, '$.tags[#]', 'vlogger');
-- # 表示末尾，tags 变为：["developer", "blogger", "reader", "vlogger"]
```

### 4.2 JSON 查询

```sql
-- JSON 路径表达式
-- $          根对象
-- .key       对象属性
-- [index]    数组索引
-- [*]        所有数组元素
-- [**]       递归搜索

-- 查询数组长度
SELECT json_array_length(profile, '$.tags') FROM users_json;

-- 展开 JSON 数组（每条记录变成多行）
SELECT
    id,
    json_each.value AS tag
FROM users_json, json_each(profile, '$.tags');
-- 结果：
-- id | tag
-- 1  | developer
-- 1  | blogger

-- JSON 类型检查
SELECT json_type(profile, '$.age') FROM users_json;  -- integer
SELECT json_type(profile, '$.name') FROM users_json; -- text

-- JSON 有效性检查
SELECT json_valid('{"a": 1}');  -- 1（true）
SELECT json_valid('{invalid}');  -- 0（false）

-- JSON 合并
SELECT json_patch('{"a": 1, "b": 2}', '{"b": 3, "c": 4}');
-- 结果：{"a": 1, "b": 3, "c": 4}
```

### 4.3 JSONB 二进制格式

```sql
-- 3.45+ 支持 JSONB（二进制 JSON，更高效）
-- jsonb() 将 JSON 文本转换为二进制格式存储

INSERT INTO users_json (profile) VALUES (jsonb('{"name": "李四"}'));

-- jsonb_extract 比 json_extract 更快
SELECT jsonb_extract(profile, '$.name') FROM users_json;

-- jsonb_array_insert（3.53+）
UPDATE users_json SET profile = jsonb_array_insert(profile, '$.tags', 'newbie');
```

### 4.4 在 Node.js 中使用 JSON

```javascript
const db = require('better-sqlite3')('app.db');

// 存储对象
db.exec(`CREATE TABLE IF NOT EXISTS events (
    id INTEGER PRIMARY KEY,
    data TEXT
)`);

const insert = db.prepare('INSERT INTO events (data) VALUES (?)');
insert.run(JSON.stringify({ type: 'click', x: 100, y: 200 }));

// 查询并解析
const rows = db.prepare('SELECT * FROM events').all();
for (const row of rows) {
    const data = JSON.parse(row.data);
    console.log(data.type);  // 'click'
}

// 使用 json_extract 在 SQL 层过滤
const clicks = db.prepare(`
    SELECT * FROM events
    WHERE json_extract(data, '$.type') = ?
`).all('click');

// 配合 json_each 展开数组
const tags = db.prepare(`
    SELECT DISTINCT json_each.value AS tag
    FROM users_json, json_each(profile, '$.tags')
`).all();
```

## 5. 全文搜索（FTS5）

FTS5 是 SQLite 的全文搜索扩展，支持高效的关键词搜索。

### 5.1 创建 FTS5 表

```sql
-- 创建 FTS5 虚拟表（不是普通表）
CREATE VIRTUAL TABLE articles USING fts5(
    title,      -- 搜索列
    content,    -- 搜索列
    author,     -- 搜索列
    tokenize = 'porter'  -- 使用 Porter Stemmer 分词（英文）
);

-- 中文分词（需要自定义 tokenizer，或使用 icu 扩展）
-- CREATE VIRTUAL TABLE articles_cn USING fts5(
--     title, content, tokenize = 'icu zh_CN'
-- );

-- 插入数据
INSERT INTO articles (title, content, author) VALUES
    ('SQLite 入门', 'SQLite 是一个嵌入式数据库引擎...', '张三'),
    ('Node.js 教程', 'Node.js 是基于 Chrome V8 引擎的 JavaScript 运行环境...', '李四'),
    ('Web 开发指南', 'Web 开发涉及前端和后端技术...', '王五');

-- 基础搜索
SELECT * FROM articles WHERE articles MATCH 'SQLite';  -- 单列搜索
SELECT * FROM articles WHERE title MATCH 'SQLite';     -- 指定列搜索

-- 多关键词（AND 关系）
SELECT * FROM articles WHERE articles MATCH 'SQLite 数据库';

-- 排除关键词
SELECT * FROM articles WHERE articles MATCH '数据库 -SQLite';  -- 包含"数据库"但不包含"SQLite"

-- 短语搜索
SELECT * FROM articles WHERE articles MATCH '"嵌入式数据库"';

-- 前缀搜索
SELECT * FROM articles WHERE articles MATCH 'Sql*';  -- 匹配 SQLite、SqlServer 等

-- NEAR 搜索（词距）
SELECT * FROM articles WHERE articles MATCH 'SQLite NEAR/5 数据库';
-- "SQLite" 和 "数据库" 之间不超过 5 个词

-- 列过滤
SELECT * FROM articles WHERE articles MATCH 'title:SQLite';  -- 只在 title 中搜索
```

### 5.2 结果排序与高亮

```sql
-- 相关性排序（默认按 rank 排序）
SELECT * FROM articles
WHERE articles MATCH '数据库'
ORDER BY rank;

-- 高亮显示（3.52+ 支持更多高亮选项）
SELECT
    highlight(articles, 0, '<b>', '</b>') AS title_highlighted,
    highlight(articles, 1, '<b>', '</b>') AS content_highlighted,
    snippet(articles, 1, '<b>', '</b>', '...', 10) AS snippet
FROM articles
WHERE articles MATCH 'SQLite';

-- highlight(表名, 列索引, 前缀, 后缀)
-- snippet(表名, 列索引, 前缀, 后缀, 省略号, 上下文词数)
```

### 5.3 辅助函数

```sql
-- 搜索建议（自动补全）
CREATE VIRTUAL TABLE suggestions USING fts5(
    phrase,
    tokenize = 'porter'
);

-- 从 articles 提取词组插入 suggestions（实现略）

-- bm25 相关性评分
SELECT title, bm25(articles) AS score
FROM articles
WHERE articles MATCH '数据库'
ORDER BY score;

-- 查看 FTS 表结构
SELECT * FROM articles_data;    -- 存储的实际数据
SELECT * FROM articles_idx;     -- 倒排索引
SELECT * FROM articles_content; -- 原始内容（如果使用了 contentless）
```

### 5.4 外部内容表（节省空间）

```sql
-- 普通表存储数据，FTS 只存索引
CREATE TABLE docs (
    id INTEGER PRIMARY KEY,
    title TEXT,
    body TEXT
);

-- content='' 表示 FTS 不存储内容，只存索引
CREATE VIRTUAL TABLE docs_fts USING fts5(
    title, body,
    content = docs,      -- 外部内容表
    content_rowid = id   -- 关联的列
);

-- 保持同步（通过触发器）
CREATE TRIGGER docs_insert AFTER INSERT ON docs BEGIN
    INSERT INTO docs_fts (rowid, title, body)
    VALUES (new.id, new.title, new.body);
END;

CREATE TRIGGER docs_delete AFTER DELETE ON docs BEGIN
    INSERT INTO docs_fts (docs_fts, rowid, title, body)
    VALUES ('delete', old.id, old.title, old.body);
END;

CREATE TRIGGER docs_update AFTER UPDATE ON docs BEGIN
    INSERT INTO docs_fts (docs_fts, rowid, title, body)
    VALUES ('delete', old.id, old.title, old.body);
    INSERT INTO docs_fts (rowid, title, body)
    VALUES (new.id, new.title, new.body);
END;

-- 查询时 JOIN
SELECT d.* FROM docs d
JOIN docs_fts f ON d.id = f.rowid
WHERE docs_fts MATCH '关键词';
```

## 6. 窗口函数

SQLite 3.25+ 支持窗口函数，可以在不减少行数的情况下进行聚合计算。

### 6.1 基础窗口函数

```sql
CREATE TABLE sales (
    id INTEGER PRIMARY KEY,
    product TEXT,
    region TEXT,
    amount REAL,
    sale_date TEXT
);

INSERT INTO sales VALUES
    (1, 'A', 'East', 100, '2024-01-01'),
    (2, 'B', 'East', 200, '2024-01-02'),
    (3, 'A', 'West', 150, '2024-01-03'),
    (4, 'B', 'West', 300, '2024-01-04'),
    (5, 'A', 'East', 120, '2024-01-05');

-- ROW_NUMBER()：行号
SELECT
    product,
    amount,
    ROW_NUMBER() OVER (ORDER BY amount DESC) AS rank_num
FROM sales;

-- RANK()：排名（有并列时跳号）
SELECT
    product,
    amount,
    RANK() OVER (ORDER BY amount DESC) AS rank_num
FROM sales;

-- DENSE_RANK()：密集排名（有并列不跳号）
SELECT
    product,
    amount,
    DENSE_RANK() OVER (ORDER BY amount DESC) AS rank_num
FROM sales;

-- NTILE(n)：分成 n 组
SELECT
    product,
    amount,
    NTILE(2) OVER (ORDER BY amount) AS bucket
FROM sales;  -- 分成两组

-- 分区窗口函数
SELECT
    product,
    region,
    amount,
    ROW_NUMBER() OVER (PARTITION BY region ORDER BY amount DESC) AS region_rank,
    SUM(amount) OVER (PARTITION BY region) AS region_total,
    AVG(amount) OVER (PARTITION BY region) AS region_avg
FROM sales;

-- 结果：
-- product | region | amount | region_rank | region_total | region_avg
-- B       | East   | 200    | 1           | 420          | 140
-- A       | East   | 120    | 2           | 420          | 140
-- A       | East   | 100    | 3           | 420          | 140
-- B       | West   | 300    | 1           | 450          | 225
-- A       | West   | 150    | 2           | 450          | 225
```

### 6.2 滑动窗口

```sql
-- 累计求和
SELECT
    sale_date,
    amount,
    SUM(amount) OVER (ORDER BY sale_date) AS cumulative_sum
FROM sales;

-- 前 N 行到当前行
SELECT
    sale_date,
    amount,
    AVG(amount) OVER (
        ORDER BY sale_date
        ROWS BETWEEN 2 PRECEDING AND CURRENT ROW
    ) AS moving_avg
FROM sales;

-- 当前分区所有行
SELECT
    product,
    amount,
    amount - AVG(amount) OVER (PARTITION BY product) AS diff_from_avg
FROM sales;

-- 窗口命名（简化复杂查询）
SELECT
    product,
    amount,
    SUM(amount) OVER w AS total,
    AVG(amount) OVER w AS avg_amount
FROM sales
WINDOW w AS (PARTITION BY region ORDER BY sale_date);
```

## 7. 触发器

### 7.1 基础触发器

```sql
-- 审计日志触发器
CREATE TABLE users (
    id INTEGER PRIMARY KEY,
    name TEXT,
    email TEXT
);

CREATE TABLE audit_log (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    table_name TEXT,
    action TEXT,
    old_data TEXT,
    new_data TEXT,
    changed_at TEXT DEFAULT (datetime('now'))
);

-- INSERT 触发器
CREATE TRIGGER trg_users_insert
AFTER INSERT ON users
BEGIN
    INSERT INTO audit_log (table_name, action, new_data)
    VALUES ('users', 'INSERT', json_object('id', NEW.id, 'name', NEW.name, 'email', NEW.email));
END;

-- UPDATE 触发器
CREATE TRIGGER trg_users_update
AFTER UPDATE ON users
BEGIN
    INSERT INTO audit_log (table_name, action, old_data, new_data)
    VALUES (
        'users',
        'UPDATE',
        json_object('id', OLD.id, 'name', OLD.name, 'email', OLD.email),
        json_object('id', NEW.id, 'name', NEW.name, 'email', NEW.email)
    );
END;

-- DELETE 触发器
CREATE TRIGGER trg_users_delete
AFTER DELETE ON users
BEGIN
    INSERT INTO audit_log (table_name, action, old_data)
    VALUES ('users', 'DELETE', json_object('id', OLD.id, 'name', OLD.name, 'email', OLD.email));
END;

-- 测试
INSERT INTO users (name, email) VALUES ('张三', 'zs@example.com');
UPDATE users SET email = 'new@example.com' WHERE id = 1;
DELETE FROM users WHERE id = 1;

SELECT * FROM audit_log;
```

### 7.2 高级触发器用法

```sql
-- 防止删除（INSTEAD OF）
CREATE TRIGGER trg_prevent_delete
INSTEAD OF DELETE ON users
BEGIN
    -- 实际不删除，只标记为已删除
    -- 需要在表中有 is_deleted 列
    UPDATE users SET is_deleted = 1 WHERE id = OLD.id;
END;

-- 级联删除
CREATE TRIGGER trg_cascade_delete
AFTER DELETE ON departments
BEGIN
    DELETE FROM employees WHERE dept_id = OLD.id;
END;

-- 自动更新时间戳
CREATE TRIGGER trg_updated_at
AFTER UPDATE ON users
BEGIN
    UPDATE users SET updated_at = datetime('now') WHERE id = NEW.id;
END;

-- 限制每用户最多 10 条记录
CREATE TRIGGER trg_limit_per_user
BEFORE INSERT ON posts
BEGIN
    SELECT CASE
        WHEN (SELECT COUNT(*) FROM posts WHERE user_id = NEW.user_id) >= 10
        THEN RAISE(ABORT, '每个用户最多 10 条记录')
    END;
END;

-- 查看所有触发器
SELECT name, sql FROM sqlite_schema WHERE type = 'trigger';

-- 删除触发器
DROP TRIGGER IF EXISTS trg_users_insert;
```

## 10. 性能调优

### 10.1 PRAGMA 调优

```sql
-- 启用外键约束（默认关闭！）
PRAGMA foreign_keys = ON;

-- 同步模式（平衡安全与性能）
PRAGMA synchronous = FULL;    -- 最安全，每次 fsync（默认）
PRAGMA synchronous = NORMAL;  -- 推荐，WAL 模式下安全
PRAGMA synchronous = OFF;     -- 最快，但崩溃可能损坏数据

-- 缓存大小
PRAGMA cache_size = 10000;    -- 页面数（默认 2000）
PRAGMA cache_size = -64000;   -- 64MB（负数表示 KB）

-- 内存映射 I/O（3.7.17+）
PRAGMA mmap_size = 30000000000;  -- 30GB（0 表示禁用）

-- 临时存储位置
PRAGMA temp_store = MEMORY;   -- 临时表存在内存中
PRAGMA temp_store = FILE;     -- 临时表存在文件中（默认）

-- 页大小（只能在建库前设置）
PRAGMA page_size = 4096;      -- 默认 4096 字节

-- 自动 vacuum
PRAGMA auto_vacuum = NONE;     -- 默认，不自动回收空间
PRAGMA auto_vacuum = FULL;     -- 自动回收，但会产生碎片
PRAGMA auto_vacuum = INCREMENTAL; -- 增量回收

-- 查询优化器限制
PRAGMA optimize;               -- 运行优化器
PRAGMA analysis_limit = 1000;  -- 限制 ANALYZE 采样行数
```

### 10.2 批量插入优化

```javascript
const db = require('better-sqlite3')('app.db');

// 方法 1：事务包裹（最重要）
db.transaction(() => {
    const insert = db.prepare('INSERT INTO users (name) VALUES (?)');
    for (let i = 0; i < 100000; i++) {
        insert.run(`user_${i}`);
    }
})();

// 方法 2：关闭同步（导入大数据时临时使用）
db.pragma('synchronous = OFF');
db.pragma('journal_mode = MEMORY');
// 导入完成后恢复
db.pragma('synchronous = NORMAL');
db.pragma('journal_mode = WAL');

// 方法 3：预编译 + 批量绑定
const insert = db.prepare('INSERT INTO users (name) VALUES (?)');
const insertMany = db.transaction((users) => {
    for (const user of users) insert.run(user);
});

// 分批提交
const batchSize = 1000;
for (let i = 0; i < allUsers.length; i += batchSize) {
    const batch = allUsers.slice(i, i + batchSize);
    insertMany(batch);
}
```
