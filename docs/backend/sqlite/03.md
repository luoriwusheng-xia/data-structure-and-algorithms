```javascript
// database.js - 数据库连接管理
const Database = require('better-sqlite3');
const fs = require('fs');
const path = require('path');

class DatabaseManager {
    constructor(dbPath = 'tasks.db') {
        this.db = new Database(dbPath);
        this.init();
    }

    init() {
        // 启用外键约束（默认关闭）
        this.db.pragma('foreign_keys = ON');
        // WAL 模式
        this.db.pragma('journal_mode = WAL');
        // 设置 busy timeout
        this.db.pragma('busy_timeout = 5000');
        // 同步模式
        this.db.pragma('synchronous = NORMAL');

        // 执行 schema
        const schemaPath = path.join(__dirname, 'schema.sql');
        if (fs.existsSync(schemaPath)) {
            this.db.exec(fs.readFileSync(schemaPath, 'utf8'));
        }
    }

    getDb() {
        return this.db;
    }

    close() {
        this.db.close();
    }
}

module.exports = DatabaseManager;
```

```javascript
class UserModel {
    constructor(db) {
        this.db = db;
    }

    create({ username, email, passwordHash, avatar }) {
        const stmt = this.db.prepare(`
            INSERT INTO users (username, email, password_hash, avatar)
            VALUES (@username, @email, @passwordHash, @avatar)
        `);
        const result = stmt.run({ username, email, passwordHash, avatar });
        return this.findById(result.lastInsertRowid);
    }

    findById(id) {
        return this.db.prepare('SELECT * FROM users WHERE id = ?').get(id);
    }

    findByUsername(username) {
        return this.db.prepare('SELECT * FROM users WHERE username = ?').get(username);
    }

    update(id, updates) {
        const keys = Object.keys(updates);
        if (keys.length === 0) return this.findById(id);

        const setClause = keys.map(k => `${k} = @${k}`).join(', ');
        const stmt = this.db.prepare(`
            UPDATE users SET ${setClause} WHERE id = @id
        `);
        stmt.run({ ...updates, id });
        return this.findById(id);
    }

    delete(id) {
        const result = this.db.prepare('DELETE FROM users WHERE id = ?').run(id);
        return result.changes > 0;
    }
}

module.exports = UserModel;
```

```javascript
// models/Task.js
class TaskModel {
    constructor(db) {
        this.db = db;
    }

    create({ title, description, projectId, assigneeId, priority, dueDate }) {
        const stmt = this.db.prepare(`
            INSERT INTO tasks (title, description, project_id, assignee_id, priority, due_date)
            VALUES (@title, @description, @projectId, @assigneeId, @priority, @dueDate)
        `);
        const result = stmt.run({ title, description, projectId, assigneeId, priority, dueDate });
        return this.findById(result.lastInsertRowid);
    }

    findById(id) {
        return this.db.prepare(`
            SELECT
                t.*,
                p.name AS project_name,
                u.username AS assignee_name,
                json_group_array(
                    json_object('id', tg.id, 'name', tg.name, 'color', tg.color)
                ) AS tags_json
            FROM tasks t
            LEFT JOIN projects p ON t.project_id = p.id
            LEFT JOIN users u ON t.assignee_id = u.id
            LEFT JOIN task_tags tt ON t.id = tt.task_id
            LEFT JOIN tags tg ON tt.tag_id = tg.id
            WHERE t.id = ? AND t.is_deleted = 0
            GROUP BY t.id
        `).get(id);
    }

    // 高级查询：带过滤和分页
    findAll({ status, priority, projectId, assigneeId, search, page = 1, limit = 20 } = {}) {
        const conditions = ['t.is_deleted = 0'];
        const params = {};

        if (status) {
            conditions.push('t.status = @status');
            params.status = status;
        }
        if (priority) {
            conditions.push('t.priority = @priority');
            params.priority = priority;
        }
        if (projectId) {
            conditions.push('t.project_id = @projectId');
            params.projectId = projectId;
        }
        if (assigneeId) {
            conditions.push('t.assignee_id = @assigneeId');
            params.assigneeId = assigneeId;
        }
        if (search) {
            conditions.push('(t.title LIKE @search OR t.description LIKE @search)');
            params.search = `%${search}%`;
        }

        const whereClause = conditions.join(' AND ');
        const offset = (page - 1) * limit;

        // 查询总数
        const countStmt = this.db.prepare(`
            SELECT COUNT(*) AS total FROM tasks t WHERE ${whereClause}
        `);
        const { total } = countStmt.get(params);

        // 查询数据
        const stmt = this.db.prepare(`
            SELECT
                t.*,
                p.name AS project_name,
                u.username AS assignee_name
            FROM tasks t
            LEFT JOIN projects p ON t.project_id = p.id
            LEFT JOIN users u ON t.assignee_id = u.id
            WHERE ${whereClause}
            ORDER BY
                CASE t.priority
                    WHEN 'urgent' THEN 1
                    WHEN 'high' THEN 2
                    WHEN 'medium' THEN 3
                    WHEN 'low' THEN 4
                END,
                t.due_date ASC NULLS LAST,
                t.sort_order ASC,
                t.created_at DESC
            LIMIT @limit OFFSET @offset
        `);

        const rows = stmt.all({ ...params, limit, offset });

        return {
            data: rows,
            pagination: {
                page,
                limit,
                total,
                totalPages: Math.ceil(total / limit)
            }
        };
    }

    // 完成任务
    complete(id) {
        const stmt = this.db.prepare(`
            UPDATE tasks
            SET status = 'done', completed_at = datetime('now')
            WHERE id = ?
        `);
        const result = stmt.run(id);
        return result.changes > 0 ? this.findById(id) : null;
    }

    // 软删除
    softDelete(id) {
        const stmt = this.db.prepare(`
            UPDATE tasks SET is_deleted = 1 WHERE id = ?
        `);
        return stmt.run(id).changes > 0;
    }

    // 统计仪表盘
    getDashboardStats(userId) {
        return this.db.prepare(`
            SELECT
                COUNT(*) FILTER (WHERE status = 'todo') AS todo_count,
                COUNT(*) FILTER (WHERE status = 'in_progress') AS in_progress_count,
                COUNT(*) FILTER (WHERE status = 'done') AS done_count,
                COUNT(*) FILTER (WHERE due_date < date('now') AND status != 'done') AS overdue_count,
                COUNT(*) FILTER (WHERE due_date = date('now') AND status != 'done') AS due_today_count
            FROM tasks
            WHERE is_deleted = 0
            AND (assignee_id = @userId OR assignee_id IS NULL)
        `).get({ userId });
    }

    // 添加标签
    addTag(taskId, tagId) {
        const stmt = this.db.prepare(`
            INSERT OR IGNORE INTO task_tags (task_id, tag_id) VALUES (?, ?)
        `);
        return stmt.run(taskId, tagId).changes > 0;
    }

    // 移除标签
    removeTag(taskId, tagId) {
        const stmt = this.db.prepare(`
            DELETE FROM task_tags WHERE task_id = ? AND tag_id = ?
        `);
        return stmt.run(taskId, tagId).changes > 0;
    }
}

module.exports = TaskModel;
```

### 2.1 通用 DAO 封装

```javascript
// utils/BaseDao.js
class BaseDao {
    constructor(db, tableName, primaryKey = 'id') {
        this.db = db;
        this.tableName = tableName;
        this.primaryKey = primaryKey;
    }

    // 动态构建 WHERE 子句
    _buildWhere(conditions) {
        if (!conditions || Object.keys(conditions).length === 0) {
            return { clause: '', params: {} };
        }
        const clauses = [];
        const params = {};
        for (const [key, value] of Object.entries(conditions)) {
            if (value === null) {
                clauses.push(`${key} IS NULL`);
            } else if (Array.isArray(value)) {
                const placeholders = value.map((_, i) => `@${key}_${i}`).join(', ');
                clauses.push(`${key} IN (${placeholders})`);
                value.forEach((v, i) => { params[`${key}_${i}`] = v; });
            } else {
                clauses.push(`${key} = @${key}`);
                params[key] = value;
            }
        }
        return { clause: 'WHERE ' + clauses.join(' AND '), params };
    }

    findById(id) {
        return this.db.prepare(
            `SELECT * FROM ${this.tableName} WHERE ${this.primaryKey} = ?`
        ).get(id);
    }

    findOne(conditions) {
        const { clause, params } = this._buildWhere(conditions);
        return this.db.prepare(
            `SELECT * FROM ${this.tableName} ${clause} LIMIT 1`
        ).get(params);
    }

    findAll(conditions = {}, { orderBy, limit, offset } = {}) {
        const { clause, params } = this._buildWhere(conditions);
        let sql = `SELECT * FROM ${this.tableName} ${clause}`;
        if (orderBy) sql += ` ORDER BY ${orderBy}`;
        if (limit) sql += ` LIMIT @limit`;
        if (offset) sql += ` OFFSET @offset`;
        return this.db.prepare(sql).all({ ...params, limit, offset });
    }

    create(data) {
        const keys = Object.keys(data);
        const columns = keys.join(', ');
        const placeholders = keys.map(k => `@${k}`).join(', ');
        const stmt = this.db.prepare(
            `INSERT INTO ${this.tableName} (${columns}) VALUES (${placeholders})`
        );
        const result = stmt.run(data);
        return this.findById(result.lastInsertRowid);
    }

    update(id, data) {
        const keys = Object.keys(data);
        if (keys.length === 0) return this.findById(id);
        const setClause = keys.map(k => `${k} = @${k}`).join(', ');
        const stmt = this.db.prepare(
            `UPDATE ${this.tableName} SET ${setClause} WHERE ${this.primaryKey} = @id`
        );
        stmt.run({ ...data, id });
        return this.findById(id);
    }

    delete(id) {
        const result = this.db.prepare(
            `DELETE FROM ${this.tableName} WHERE ${this.primaryKey} = ?`
        ).run(id);
        return result.changes > 0;
    }

    count(conditions = {}) {
        const { clause, params } = this._buildWhere(conditions);
        const result = this.db.prepare(
            `SELECT COUNT(*) AS count FROM ${this.tableName} ${clause}`
        ).get(params);
        return result.count;
    }

    exists(conditions) {
        return this.findOne(conditions) !== undefined;
    }
}

module.exports = BaseDao;
```

## 3. 数据库迁移方案

```javascript
// migrations/MigrationManager.js
const fs = require('fs');
const path = require('path');

class MigrationManager {
    constructor(db, migrationsDir = './migrations') {
        this.db = db;
        this.migrationsDir = migrationsDir;
        this.init();
    }

    init() {
        // 创建迁移记录表
        this.db.exec(`
            CREATE TABLE IF NOT EXISTS __migrations (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                version INTEGER NOT NULL UNIQUE,
                name TEXT NOT NULL,
                applied_at TEXT DEFAULT (datetime('now'))
            )
        `);
    }

    getCurrentVersion() {
        const row = this.db.prepare(
            'SELECT MAX(version) AS version FROM __migrations'
        ).get();
        return row.version || 0;
    }

    getPendingMigrations() {
        const currentVersion = this.getCurrentVersion();
        const files = fs.readdirSync(this.migrationsDir)
            .filter(f => f.endsWith('.sql'))
            .sort();

        return files
            .map(f => {
                const match = f.match(/^(\d+)_(.+?)\.sql$/);
                if (!match) return null;
                return {
                    version: parseInt(match[1]),
                    name: match[2],
                    file: f
                };
            })
            .filter(Boolean)
            .filter(m => m.version > currentVersion);
    }

    migrate() {
        const pending = this.getPendingMigrations();
        if (pending.length === 0) {
            console.log('No pending migrations');
            return;
        }

        for (const migration of pending) {
            console.log(`Applying migration ${migration.version}: ${migration.name}`);

            const sql = fs.readFileSync(
                path.join(this.migrationsDir, migration.file),
                'utf8'
            );

            this.db.transaction(() => {
                this.db.exec(sql);
                this.db.prepare(
                    'INSERT INTO __migrations (version, name) VALUES (?, ?)'
                ).run(migration.version, migration.name);
            })();

            console.log(`Migration ${migration.version} applied successfully`);
        }
    }

    rollback(steps = 1) {
        const migrations = this.db.prepare(
            'SELECT * FROM __migrations ORDER BY version DESC LIMIT ?'
        ).all(steps);

        for (const migration of migrations) {
            const downFile = path.join(
                this.migrationsDir,
                `${migration.version}_${migration.name}.down.sql`
            );

            if (fs.existsSync(downFile)) {
                const sql = fs.readFileSync(downFile, 'utf8');
                this.db.exec(sql);
            }

            this.db.prepare('DELETE FROM __migrations WHERE version = ?').run(migration.version);
            console.log(`Rolled back migration ${migration.version}`);
        }
    }
}

module.exports = MigrationManager;
```

```javascript
// utils/Profiler.js
class QueryProfiler {
    constructor(db, { enabled = true, slowThreshold = 100 } = {}) {
        this.db = db;
        this.enabled = enabled;
        this.slowThreshold = slowThreshold;
        this.stats = [];
    }

    profile(name, fn) {
        if (!this.enabled) return fn();

        const start = process.hrtime.bigint();
        try {
            return fn();
        } finally {
            const duration = Number(process.hrtime.bigint() - start) / 1000000; // ms
            this.stats.push({ name, duration, timestamp: new Date() });

            if (duration > this.slowThreshold) {
                console.warn(`[SLOW QUERY] ${name}: ${duration.toFixed(2)}ms`);
            }
        }
    }

    // 准备带性能监控的语句
    prepare(sql) {
        const stmt = this.db.prepare(sql);
        return {
            run: (...args) => this.profile(sql, () => stmt.run(...args)),
            get: (...args) => this.profile(sql, () => stmt.get(...args)),
            all: (...args) => this.profile(sql, () => stmt.all(...args))
        };
    }

    getStats() {
        const total = this.stats.length;
        const slow = this.stats.filter(s => s.duration > this.slowThreshold);
        const avg = total > 0
            ? this.stats.reduce((a, b) => a + b.duration, 0) / total
            : 0;

        return { total, slow: slow.length, avg: avg.toFixed(2), top: slow.slice(0, 10) };
    }
}

// 数据库健康检查
function healthCheck(db) {
    const checks = {};

    // 检查 WAL 文件大小
    const walSize = db.prepare("SELECT page_count * page_size AS size FROM pragma_wal_checkpoint(), pragma_page_size()").get();
    checks.walSize = walSize;

    // 检查空闲页比例
    const pages = db.prepare("SELECT page_count, freelist_count FROM pragma_page_count(), pragma_freelist_count()").get();
    checks.fragmentation = (pages.freelist_count / pages.page_count * 100).toFixed(2) + '%';

    // 检查表大小
    checks.tableSizes = db.prepare(`
        SELECT name, pageno, ncell
        FROM dbstat
        WHERE aggregate = TRUE
        ORDER BY pageno DESC
    `).all();

    // 检查慢查询（使用 sqlite3 的内部统计）
    checks.statementStats = db.prepare(`
        SELECT sql, run, vm_step, run/vm_step AS efficiency
        FROM sqlite_stmt
        WHERE run > 0
        ORDER BY run DESC
    `).all();

    // 检查未使用的索引
    checks.unusedIndexes = db.prepare(`
        SELECT name, tbl_name
        FROM sqlite_schema
        WHERE type = 'index'
        AND name NOT IN (
            SELECT name FROM sqlite_stat1
        )
    `).all();

    return checks;
}
```

```javascript
// utils/ImportExport.js
const fs = require('fs');
const { Parser } = require('json2csv');

class DataImporterExporter {
    constructor(db) {
        this.db = db;
    }

    // 导出为 CSV
    exportToCSV(tableName, outputPath) {
        const rows = this.db.prepare(`SELECT * FROM ${tableName}`).all();
        if (rows.length === 0) return;

        const parser = new Parser();
        const csv = parser.parse(rows);
        fs.writeFileSync(outputPath, csv);
    }

    // 从 CSV 导入
    importFromCSV(tableName, csvPath, { headers = true, delimiter = ',' } = {}) {
        const content = fs.readFileSync(csvPath, 'utf8');
        const lines = content.trim().split('\n');
        const startRow = headers ? 1 : 0;
        const columnNames = headers
            ? lines[0].split(delimiter).map(h => h.trim())
            : null;

        const columns = columnNames || this.getColumnNames(tableName);
        const placeholders = columns.map(() => '?').join(', ');

        const stmt = this.db.prepare(
            `INSERT INTO ${tableName} (${columns.join(', ')}) VALUES (${placeholders})`
        );

        const insertMany = this.db.transaction((rows) => {
            for (const row of rows) {
                stmt.run(...row);
            }
        });

        const rows = lines.slice(startRow).map(line =>
            line.split(delimiter).map(v => v.trim())
        );

        insertMany(rows);
        return rows.length;
    }

    // 导出为 JSON
    exportToJSON(tableName, outputPath) {
        const rows = this.db.prepare(`SELECT * FROM ${tableName}`).all();
        fs.writeFileSync(outputPath, JSON.stringify(rows, null, 2));
    }

    // 从 JSON 导入
    importFromJSON(tableName, jsonPath) {
        const items = JSON.parse(fs.readFileSync(jsonPath, 'utf8'));
        if (items.length === 0) return 0;

        const columns = Object.keys(items[0]);
        const placeholders = columns.map(() => `@${}`).join(', ');
        const colStr = columns.join(', ');

        const stmt = this.db.prepare(
            `INSERT INTO ${tableName} (${colStr}) VALUES (${placeholders})`
        );

        const insertMany = this.db.transaction((rows) => {
            for (const row of rows) stmt.run(row);
        });

        insertMany(items);
        return items.length;
    }

    // 导出为 SQL
    exportToSQL(outputPath, { tables = [], dataOnly = false } = {}) {
        let sql = '';

        if (!dataOnly) {
            // 导出 schema
            const schema = this.db.prepare(`
                SELECT sql FROM sqlite_schema
                WHERE type IN ('table', 'index', 'trigger', 'view')
                AND name NOT LIKE 'sqlite_%'
                AND name NOT LIKE '__%'
            `).all();
            sql += schema.map(s => s.sql + ';').join('\n\n') + '\n\n';
        }

        // 导出数据
        const tableList = tables.length > 0
            ? tables
            : this.db.prepare("SELECT name FROM sqlite_schema WHERE type = 'table' AND name NOT LIKE 'sqlite_%' AND name NOT LIKE '__%'").all().map(t => t.name);

        for (const table of tableList) {
            const rows = this.db.prepare(`SELECT * FROM ${table}`).all();
            if (rows.length === 0) continue;

            const columns = Object.keys(rows[0]);
            const colStr = columns.join(', ');

            for (const row of rows) {
                const values = columns.map(c => {
                    const v = row[c];
                    if (v === null) return 'NULL';
                    if (typeof v === 'string') return `'${v.replace(/'/g, "''")}'`;
                    if (typeof v === 'object') return `'${JSON.stringify(v).replace(/'/g, "''")}'`;
                    return v;
                });
                sql += `INSERT INTO ${table} (${colStr}) VALUES (${values.join(', ')});\n`;
            }
            sql += '\n';
        }

        fs.writeFileSync(outputPath, sql);
    }

    getColumnNames(tableName) {
        const info = this.db.prepare(`PRAGMA table_info(${tableName})`).all();
        return info.map(col => col.name);
    }
}

module.exports = DataImporterExporter;
```

## 6. 备份与恢复

```javascript
// utils/Backup.js
const fs = require('fs');
const path = require('path');

class BackupManager {
    constructor(db, backupDir = './backups') {
        this.db = db;
        this.backupDir = backupDir;
        if (!fs.existsSync(backupDir)) {
            fs.mkdirSync(backupDir, { recursive: true });
        }
    }

    // 热备份（在线备份，不中断服务）
    backup(filename = null) {
        const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
        const backupFile = filename || `backup_${timestamp}.db`;
        const backupPath = path.join(this.backupDir, backupFile);

        // better-sqlite3 支持热备份
        this.db.backup(backupPath);

        console.log(`Backup created: ${backupPath}`);
        return backupPath;
    }

    // 增量备份（基于 WAL 的增量）
    incrementalBackup() {
        const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
        const backupPath = path.join(this.backupDir, `backup_${timestamp}.db`);

        // 先 checkpoint WAL
        this.db.pragma('wal_checkpoint(TRUNCATE)');

        // 复制数据库文件
        fs.copyFileSync(this.db.name, backupPath);

        return backupPath;
    }

    // 自动清理旧备份
    cleanup(maxAgeDays = 7, maxCount = 30) {
        const files = fs.readdirSync(this.backupDir)
            .filter(f => f.endsWith('.db'))
            .map(f => {
                const stat = fs.statSync(path.join(this.backupDir, f));
                return { name: f, mtime: stat.mtime, size: stat.size };
            })
            .sort((a, b) => b.mtime - a.mtime);

        const cutoff = new Date();
        cutoff.setDate(cutoff.getDate() - maxAgeDays);

        for (let i = maxCount; i < files.length; i++) {
            const file = files[i];
            if (file.mtime < cutoff) {
                fs.unlinkSync(path.join(this.backupDir, file.name));
                console.log(`Deleted old backup: ${file.name}`);
            }
        }
    }

    // 恢复到指定备份
    restore(backupFile) {
        const backupPath = path.join(this.backupDir, backupFile);
        if (!fs.existsSync(backupPath)) {
            throw new Error(`Backup not found: ${backupPath}`);
        }

        // 先关闭当前连接
        this.db.close();

        // 替换数据库文件
        const dbPath = 'app.db'; // 你的数据库路径
        fs.copyFileSync(backupPath, dbPath);

        // 重新连接
        const Database = require('better-sqlite3');
        this.db = new Database(dbPath);

        console.log(`Restored from: ${backupPath}`);
        return this.db;
    }

    // 列出备份
    listBackups() {
        return fs.readdirSync(this.backupDir)
            .filter(f => f.endsWith('.db'))
            .map(f => {
                const stat = fs.statSync(path.join(this.backupDir, f));
                return {
                    name: f,
                    size: (stat.size / 1024 / 1024).toFixed(2) + ' MB',
                    created: stat.mtime
                };
            })
            .sort((a, b) => b.created - a.created);
    }
}

// 定时备份
function scheduleBackups(db, intervalHours = 24) {
    const manager = new BackupManager(db);

    // 立即备份一次
    manager.backup();

    // 定时备份
    setInterval(() => {
        manager.backup();
        manager.cleanup();
    }, intervalHours * 60 * 60 * 1000);

    return manager;
}

module.exports = { BackupManager, scheduleBackups };
```

## 8. 常见问题与解决方案

### 8.1 数据库被锁定（database is locked）

```javascript
// 问题：并发写入导致
// 解决 1：启用 WAL 模式
db.pragma('journal_mode = WAL');

// 解决 2：设置 busy timeout
db.pragma('busy_timeout = 5000');  // 等待 5 秒

// 解决 3：使用连接池（单文件数据库不需要，但多进程需要）
// 每个进程使用独立的数据库连接

// 解决 4：队列化写入操作
class WriteQueue {
    constructor(db) {
        this.db = db;
        this.queue = [];
        this.running = false;
    }

    async enqueue(fn) {
        return new Promise((resolve, reject) => {
            this.queue.push({ fn, resolve, reject });
            this.process();
        });
    }

    async process() {
        if (this.running || this.queue.length === 0) return;
        this.running = true;

        const { fn, resolve, reject } = this.queue.shift();
        try {
            const result = await fn(this.db);
            resolve(result);
        } catch (e) {
            reject(e);
        } finally {
            this.running = false;
            this.process();
        }
    }
}
```
