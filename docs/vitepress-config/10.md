`/.vitepress/theme/MyLayout.vue`

- 使用默认的主题， 然后引入viewerjs包。
- layout 是全局的，在 不同的页面切换的时候，我们需要重新获取 图片，不然就出现只在第一个页面可以预览图片， 其他的无效。
- 下面核心的就是 `loadViewerPlugin` 方法
-


```vue
<script setup lang="ts">
import { onMounted, watch, nextTick, provide } from "vue";
import DefaultTheme from "vitepress/theme";
import Viewer from "viewerjs";
import "viewerjs/dist/viewer.min.css";
import {useData} from 'vitepress'

import { useRoute } from "vitepress";

const { isDark } = useData()


const { Layout } = DefaultTheme;

// 判断浏览器是否支持 View Transitions API
const enableTransitions = () => ('startViewTransition' in document &&
  window.matchMedia('(prefers-reduced-motion: no-preference)').matches)

// 监听切换主题事件
provide('toggle-appearance', async ({ clientX: x, clientY: y }: MouseEvent) => {
  if (!enableTransitions()) {
    isDark.value = !isDark.value
    return
  }

  const clipPath = [
    `circle(0px at ${x}px ${y}px)`,
    `circle(${Math.hypot(
      Math.max(x, innerWidth - x),
      Math.max(y, innerHeight - y)
    )}px at ${x}px ${y}px)`
  ]

  await document.startViewTransition(async () => {
    isDark.value = !isDark.value
    await nextTick()
  }).ready

  document.documentElement.animate(
    { clipPath: isDark.value ? clipPath.reverse() : clipPath },
    {
      duration: 300,
      easing: 'ease-in',
      pseudoElement: `::view-transition-${isDark.value ? 'old' : 'new'}(root)`
    }
  )
})

const loadViewerPlugin = () => {
    const container = document.querySelectorAll(".vp-doc img");

    // 将所有图片放到一个共同的容器里，然后只初始化一次 Viewer
    if (container.length > 0) {
        // 找到最近的公共父节点，通常是 .vp-doc
        const parent = document.querySelector(".vp-doc");

        if (parent.__viewer) {
            parent.__viewer.destroy();
        }

        parent.__viewer = new Viewer(parent, {
            toolbar: true, // 显示底部工具栏
            navbar: false, // 不显示左侧缩略图导航栏
            title: false, // 不显示图片标题
            scalable: true, // 允许图片缩放
            movable: false, // 禁止拖动图片
            zoomable: true, // 允许缩放图片
            transition: true, // 开启动画过渡效果
            // 只预览 img 元素
            filter(image) {
                return image.tagName.toLowerCase() === "img";
            },
            viewed() {
                // 可根据需要添加回调
            }
        });
    }
};

onMounted(() => {
    loadViewerPlugin();
});

const route = useRoute();

watch(
    () => route.path,
    () => {
        nextTick(() => {
            loadViewerPlugin();
        });
    }
);
</script>

<template>
    <Layout> </Layout>
</template>

<style>
::view-transition-old(root),
::view-transition-new(root) {
  animation: none;
  mix-blend-mode: normal;
}

::view-transition-old(root),
.dark::view-transition-new(root) {
  z-index: 1;
}

::view-transition-new(root),
.dark::view-transition-old(root) {
  z-index: 9999;
}

.VPSwitchAppearance {
  width: 22px !important;
}

.VPSwitchAppearance .check {
  transform: none !important;
}
</style>
```
