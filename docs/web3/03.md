# Web3 技术落地：从编写合约到构建 DApp

本章把前面学到的概念和工具串起来，带你完成一个完整的 Web3 应用开发流程。你会学到：**编写智能合约、本地测试、部署上链、构建前端、连接钱包**。

## 一、开发环境准备

### 1.1 安装依赖

```bash

node -v
```

## 二、编写第一个智能合约

### 2.1 Counter 合约

```solidity
// contracts/Counter.sol
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.19;

contract Counter {
    uint256 public count;

    event CountChanged(uint256 newCount);

    function increment() public {
        count += 1;
        emit CountChanged(count);
    }

    function decrement() public {
        require(count > 0, "Count cannot be negative");
        count -= 1;
        emit CountChanged(count);
    }

    function getCount() public view returns (uint256) {
        return count;
    }
}
```

**关键语法解释**：

- `uint256`：无符号 256 位整数
- `public`：变量自动获得 getter 函数，函数可被外部调用
- `view`：只读函数，不消耗 Gas（外部调用时）
- `event`：事件，用于前端监听链上状态变化
- `emit`：触发事件
- `require`：条件检查，失败时回滚交易

### 2.2 编译合约

```bash
npx hardhat compile
```

编译成功后会生成 `artifacts/` 和 `cache/` 目录，其中 ABI 文件在 `artifacts/contracts/Counter.sol/Counter.json`。

## 三、测试智能合约

### 3.1 编写测试

```javascript
// test/Counter.js
const { expect } = require('chai');
const { ethers } = require('hardhat');

describe('Counter', function () {
  let counter;
  let owner;

  beforeEach(async function () {
    [owner] = await ethers.getSigners();

    const Counter = await ethers.getContractFactory('Counter');
    counter = await Counter.deploy();
    await counter.waitForDeployment();
  });

  it('初始值应为 0', async function () {
    expect(await counter.getCount()).to.equal(0);
  });

  it('increment 应正确累加', async function () {
    await counter.increment();
    await counter.increment();
    expect(await counter.getCount()).to.equal(2);
  });

  it('decrement 在 0 时应失败', async function () {
    await expect(counter.decrement()).to.be.revertedWith('Count cannot be negative');
  });

  it('应触发 CountChanged 事件', async function () {
    await expect(counter.increment())
      .to.emit(counter, 'CountChanged')
      .withArgs(1);
  });
});
```


## 四、部署智能合约

### 4.1 部署脚本

```javascript
// scripts/deploy.js
const { ethers } = require('hardhat');

async function main() {
  const Counter = await ethers.getContractFactory('Counter');
  const counter = await Counter.deploy();

  await counter.waitForDeployment();

  console.log('Counter deployed to:', await counter.getAddress());
}

main()
  .then(() => process.exit(0))
  .catch((error) => {
    console.error(error);
    process.exit(1);
  });
```

### 4.2 部署到本地网络

```bash
# 终端 1：启动本地节点
npx hardhat node

# 终端 2：部署
npx hardhat run scripts/deploy.js --network localhost
```

### 4.3 部署到测试网

```bash
npx hardhat run scripts/deploy.js --network sepolia
```

部署成功后，复制合约地址，稍后的前端会用到。

### 4.4 验证合约源码

```bash
npx hardhat verify --network sepolia CONTRACT_ADDRESS
```

验证后可以在 Etherscan 上直接查看合约源码，便于他人审计和交互。

### 5.3 前端 JavaScript

```javascript
// main.js
import { ethers } from 'ethers';
import CounterABI from '../artifacts/contracts/Counter.sol/Counter.json';

const CONTRACT_ADDRESS = '0x...your_contract_address...';

let provider;
let signer;
let contract;

const connectBtn = document.getElementById('connect');
const incrementBtn = document.getElementById('increment');
const decrementBtn = document.getElementById('decrement');
const accountSpan = document.getElementById('account');
const countSpan = document.getElementById('count');
const statusP = document.getElementById('status');

async function connectWallet() {
  if (!window.ethereum) {
    alert('请安装 MetaMask');
    return;
  }

  provider = new ethers.BrowserProvider(window.ethereum);
  await provider.send('eth_requestAccounts', []);
  signer = await provider.getSigner();

  const address = await signer.getAddress();
  accountSpan.textContent = address;

  contract = new ethers.Contract(CONTRACT_ADDRESS, CounterABI.abi, signer);

  await loadCount();

  // 监听事件
  contract.on('CountChanged', (newCount) => {
    countSpan.textContent = newCount.toString();
  });
}

async function loadCount() {
  const count = await contract.getCount();
  countSpan.textContent = count.toString();
}

async function increment() {
  try {
    statusP.textContent = '交易发送中...';
    const tx = await contract.increment();
    statusP.textContent = '等待确认...';
    await tx.wait();
    statusP.textContent = '交易成功!';
  } catch (err) {
    statusP.textContent = `错误: ${err.reason || err.message}`;
  }
}

async function decrement() {
  try {
    statusP.textContent = '交易发送中...';
    const tx = await contract.decrement();
    await tx.wait();
    statusP.textContent = '交易成功!';
  } catch (err) {
    statusP.textContent = `错误: ${err.reason || err.message}`;
  }
}

connectBtn.addEventListener('click', connectWallet);
incrementBtn.addEventListener('click', increment);
decrementBtn.addEventListener('click', decrement);
```

## 六、核心开发模式

### 6.1 读取 vs 写入

```javascript
// 读取：免费，不需要签名
const count = await contract.getCount();

// 写入：需要支付 Gas，会弹出签名窗口
const tx = await contract.increment();
await tx.wait();
```

### 6.2 事件监听

```javascript
// 监听所有 Transfer 事件
contract.on('Transfer', (from, to, amount, event) => {
  console.log(`${from} -> ${to}: ${amount}`);
});

// 只监听一次
contract.once('Transfer', handler);

// 移除监听
contract.removeAllListeners('Transfer');
```

### 6.3 错误处理

```javascript
try {
  const tx = await contract.decrement();
  await tx.wait();
} catch (err) {
  if (err.code === 'ACTION_REJECTED') {
    console.log('用户取消了签名');
  } else if (err.reason) {
    console.log('合约回滚:', err.reason);
  } else {
    console.log('未知错误:', err);
  }
}
```

## 八、常见开发陷阱

| 问题 | 原因 | 解决方案 |
|------|------|----------|
| MetaMask 不弹出 | 网络不匹配 | 确认链 ID 一致 |
| Gas estimation failed | 合约条件不满足 | 检查 `require` 条件 |
| nonce too low | 交易顺序错乱 | 在 MetaMask 重置账号 |
| provider 为空 | 未连接钱包 | 先调用 `eth_requestAccounts` |
| BigNumber 比较失败 | JS Number 精度不够 | 使用 `ethers.parseUnits` / `formatUnits` |
