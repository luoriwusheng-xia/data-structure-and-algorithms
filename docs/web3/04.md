# Web3 实战：构建一个去中心化任务悬赏平台

本章通过一个完整的项目，把前面学到的知识融会贯通。我们要做一个**去中心化任务悬赏平台（Bounty Board）**：

- 发布者可以发布任务并质押赏金
- 完成者提交任务结果
- 发布者确认后，赏金自动转账给完成者
- 无人接单可取消任务并取回赏金

## 一、项目需求分析

### 1.1 功能列表

| 角色 | 功能 |
|------|------|
| 发布者（Poster） | 发布任务 + 质押赏金、确认完成、取消任务 |
| 完成者（Hunter） | 接受任务、提交结果 |
| 平台 | 托管赏金、超时自动释放 |

### 1.2 合约设计

```solidity
struct Bounty {
    address poster;       // 发布者
    address hunter;       // 完成者
    string title;         // 任务标题
    string description;   // 任务描述
    uint256 reward;       // 赏金
    BountyStatus status;  // 任务状态
    uint256 createdAt;    // 创建时间
    uint256 deadline;     // 截止时间
}

enum BountyStatus {
    Open,      // 开放中
    InProgress,// 进行中
    Completed, // 已完成
    Cancelled  // 已取消
}
```

## 二、智能合约实现

```solidity
// contracts/BountyBoard.sol
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.19;

contract BountyBoard {
    enum BountyStatus { Open, InProgress, Completed, Cancelled }

    struct Bounty {
        address poster;
        address hunter;
        string title;
        string description;
        uint256 reward;
        BountyStatus status;
        uint256 createdAt;
        uint256 deadline;
    }

    uint256 public bountyCount;
    mapping(uint256 => Bounty) public bounties;

    uint256 public constant PLATFORM_FEE = 500; // 5% = 500 / 10000
    uint256 public constant FEE_DENOMINATOR = 10000;
    uint256 public constant MIN_DEADLINE_DAYS = 1;
    uint256 public constant MAX_DEADLINE_DAYS = 90;

    address public owner;
    uint256 public accumulatedFees;

    event BountyCreated(
        uint256 indexed bountyId,
        address indexed poster,
        uint256 reward,
        uint256 deadline
    );
    event BountyAccepted(uint256 indexed bountyId, address indexed hunter);
    event SubmissionDelivered(uint256 indexed bountyId, string submissionLink);
    event BountyCompleted(uint256 indexed bountyId, address indexed hunter, uint256 reward);
    event BountyCancelled(uint256 indexed bountyId);
    event BountyExpired(uint256 indexed bountyId);

    modifier onlyPoster(uint256 _bountyId) {
        require(bounties[_bountyId].poster == msg.sender, "Not the poster");
        _;
    }

    modifier onlyHunter(uint256 _bountyId) {
        require(bounties[_bountyId].hunter == msg.sender, "Not the hunter");
        _;
    }

    modifier bountyExists(uint256 _bountyId) {
        require(_bountyId < bountyCount, "Bounty does not exist");
        _;
    }

    constructor() {
        owner = msg.sender;
    }

    function createBounty(
        string memory _title,
        string memory _description,
        uint256 _deadlineDays
    ) public payable returns (uint256) {
        require(msg.value > 0, "Reward must be greater than 0");
        require(bytes(_title).length > 0, "Title cannot be empty");
        require(
            _deadlineDays >= MIN_DEADLINE_DAYS && _deadlineDays <= MAX_DEADLINE_DAYS,
            "Invalid deadline"
        );

        uint256 bountyId = bountyCount++;
        bounties[bountyId] = Bounty({
            poster: msg.sender,
            hunter: address(0),
            title: _title,
            description: _description,
            reward: msg.value,
            status: BountyStatus.Open,
            createdAt: block.timestamp,
            deadline: block.timestamp + (_deadlineDays * 1 days)
        });

        emit BountyCreated(bountyId, msg.sender, msg.value, bounties[bountyId].deadline);
        return bountyId;
    }

    function acceptBounty(uint256 _bountyId) public bountyExists(_bountyId) {
        Bounty storage bounty = bounties[_bountyId];

        require(bounty.status == BountyStatus.Open, "Bounty not open");
        require(bounty.poster != msg.sender, "Poster cannot accept");
        require(block.timestamp < bounty.deadline, "Bounty expired");

        bounty.hunter = msg.sender;
        bounty.status = BountyStatus.InProgress;

        emit BountyAccepted(_bountyId, msg.sender);
    }

    function submitWork(
        uint256 _bountyId,
        string memory _submissionLink
    ) public bountyExists(_bountyId) onlyHunter(_bountyId) {
        Bounty storage bounty = bounties[_bountyId];
        require(bounty.status == BountyStatus.InProgress, "Bounty not in progress");
        require(bytes(_submissionLink).length > 0, "Submission link cannot be empty");

        emit SubmissionDelivered(_bountyId, _submissionLink);
    }

    function completeBounty(
        uint256 _bountyId
    ) public bountyExists(_bountyId) onlyPoster(_bountyId) {
        Bounty storage bounty = bounties[_bountyId];
        require(bounty.status == BountyStatus.InProgress, "Bounty not in progress");

        uint256 fee = (bounty.reward * PLATFORM_FEE) / FEE_DENOMINATOR;
        uint256 hunterReward = bounty.reward - fee;

        accumulatedFees += fee;
        bounty.status = BountyStatus.Completed;

        (bool success, ) = payable(bounty.hunter).call{value: hunterReward}("");
        require(success, "Transfer failed");

        emit BountyCompleted(_bountyId, bounty.hunter, hunterReward);
    }

    function cancelBounty(
        uint256 _bountyId
    ) public bountyExists(_bountyId) onlyPoster(_bountyId) {
        Bounty storage bounty = bounties[_bountyId];
        require(bounty.status == BountyStatus.Open, "Bounty not open");
        require(block.timestamp < bounty.deadline, "Bounty expired");

        uint256 reward = bounty.reward;
        bounty.status = BountyStatus.Cancelled;
        bounty.reward = 0;

        (bool success, ) = payable(bounty.poster).call{value: reward}("");
        require(success, "Refund failed");

        emit BountyCancelled(_bountyId);
    }

    function claimExpiredBounty(
        uint256 _bountyId
    ) public bountyExists(_bountyId) onlyHunter(_bountyId) {
        Bounty storage bounty = bounties[_bountyId];
        require(bounty.status == BountyStatus.InProgress, "Bounty not in progress");
        require(block.timestamp >= bounty.deadline, "Bounty not expired");

        uint256 fee = (bounty.reward * PLATFORM_FEE) / FEE_DENOMINATOR;
        uint256 hunterReward = bounty.reward - fee;

        accumulatedFees += fee;
        bounty.status = BountyStatus.Completed;

        (bool success, ) = payable(bounty.hunter).call{value: hunterReward}("");
        require(success, "Transfer failed");

        emit BountyExpired(_bountyId);
    }

    function withdrawFees() public {
        require(msg.sender == owner, "Only owner");
        uint256 amount = accumulatedFees;
        accumulatedFees = 0;

        (bool success, ) = payable(owner).call{value: amount}("");
        require(success, "Withdraw failed");
    }

    function getBounty(uint256 _bountyId) public view returns (Bounty memory) {
        return bounties[_bountyId];
    }
}
```

## 三、合约安全要点

### 3.1 重入攻击防护

本合约中，我们在转账**之前**就修改了状态变量：

```solidity
bounty.status = BountyStatus.Completed; // 先改状态
(bool success, ) = payable(bounty.hunter).call{value: hunterReward}(""); // 再转账
```

这遵循了 **Checks-Effects-Interactions** 模式，有效防止重入攻击。

### 3.2 整数溢出

Solidity 0.8.x 默认开启溢出检查，不需要再手动使用 SafeMath。

### 3.3 平台费率计算

使用 `FEE_DENOMINATOR` 避免浮点数，保证精度。

```solidity
uint256 fee = (bounty.reward * PLATFORM_FEE) / FEE_DENOMINATOR;
```

## 四、测试用例

```javascript
// test/BountyBoard.js
const { expect } = require('chai');
const { ethers } = require('hardhat');

describe('BountyBoard', function () {
  let bountyBoard;
  let owner, poster, hunter, other;

  beforeEach(async function () {
    [owner, poster, hunter, other] = await ethers.getSigners();

    const BountyBoard = await ethers.getContractFactory('BountyBoard');
    bountyBoard = await BountyBoard.deploy();
    await bountyBoard.waitForDeployment();
  });

  it('应该允许发布任务并质押赏金', async function () {
    const reward = ethers.parseEther('1');

    await expect(
      bountyBoard.connect(poster).createBounty('写一篇文章', '关于 Solidity', 7, { value: reward })
    )
      .to.emit(bountyBoard, 'BountyCreated')
      .withArgs(0, poster.address, reward, await time.latest() + 7 * 86400);

    const bounty = await bountyBoard.bounties(0);
    expect(bounty.reward).to.equal(reward);
    expect(bounty.status).to.equal(0); // Open
  });

  it('应该允许 Hunter 接受任务', async function () {
    await bountyBoard.connect(poster).createBounty('任务', '描述', 7, {
      value: ethers.parseEther('1'),
    });

    await expect(bountyBoard.connect(hunter).acceptBounty(0))
      .to.emit(bountyBoard, 'BountyAccepted')
      .withArgs(0, hunter.address);

    const bounty = await bountyBoard.bounties(0);
    expect(bounty.hunter).to.equal(hunter.address);
    expect(bounty.status).to.equal(1); // InProgress
  });

  it('完成任务后应转账给 Hunter', async function () {
    const reward = ethers.parseEther('1');

    await bountyBoard.connect(poster).createBounty('任务', '描述', 7, { value: reward });
    await bountyBoard.connect(hunter).acceptBounty(0);

    const hunterBalanceBefore = await ethers.provider.getBalance(hunter.address);

    await bountyBoard.connect(poster).completeBounty(0);

    const hunterBalanceAfter = await ethers.provider.getBalance(hunter.address);
    const expectedReward = (reward * 9500n) / 10000n; // 扣除 5% 手续费

    expect(hunterBalanceAfter - hunterBalanceBefore).to.equal(expectedReward);
  });

  it('发布者应能取消开放中的任务', async function () {
    const reward = ethers.parseEther('1');
    await bountyBoard.connect(poster).createBounty('任务', '描述', 7, { value: reward });

    await expect(bountyBoard.connect(poster).cancelBounty(0))
      .to.emit(bountyBoard, 'BountyCancelled')
      .withArgs(0);

    const bounty = await bountyBoard.bounties(0);
    expect(bounty.status).to.equal(3); // Cancelled
  });
});
```

### 5.2 核心交互逻辑

```javascript
// main.js
import { ethers } from 'ethers';
import BountyBoardABI from '../artifacts/contracts/BountyBoard.sol/BountyBoard.json';

const CONTRACT_ADDRESS = '0x...';

let provider, signer, contract, account;

async function connect() {
  provider = new ethers.BrowserProvider(window.ethereum);
  await provider.send('eth_requestAccounts', []);
  signer = await provider.getSigner();
  account = await signer.getAddress();

  document.getElementById('account').textContent = account;
  contract = new ethers.Contract(CONTRACT_ADDRESS, BountyBoardABI.abi, signer);

  await loadBounties();
}

async function createBounty() {
  const title = document.getElementById('title').value;
  const description = document.getElementById('description').value;
  const deadline = document.getElementById('deadline').value;
  const reward = document.getElementById('reward').value;

  try {
    const tx = await contract.createBounty(title, description, deadline, {
      value: ethers.parseEther(reward),
    });
    await tx.wait();
    alert('任务发布成功');
    await loadBounties();
  } catch (err) {
    alert(`失败: ${err.reason || err.message}`);
  }
}

async function acceptBounty(id) {
  const tx = await contract.acceptBounty(id);
  await tx.wait();
  await loadBounties();
}

async function completeBounty(id) {
  const tx = await contract.completeBounty(id);
  await tx.wait();
  await loadBounties();
}

async function loadBounties() {
  const count = await contract.bountyCount();
  const list = document.getElementById('list');
  list.innerHTML = '';

  for (let i = count - 1n; i >= 0n; i--) {
    const bounty = await contract.getBounty(i);
    const status = ['开放中', '进行中', '已完成', '已取消'][bounty.status];

    const div = document.createElement('div');
    div.className = 'bounty-card';
    div.innerHTML = `
      <h3>${bounty.title}</h3>
      <p>${bounty.description}</p>
      <p>赏金: ${ethers.formatEther(bounty.reward)} ETH</p>
      <p>状态: ${status}</p>
      <p>发布者: ${bounty.poster.slice(0, 6)}...${bounty.poster.slice(-4)}</p>
      ${renderActions(bounty, i)}
    `;
    list.appendChild(div);
  }
}

function renderActions(bounty, id) {
  if (bounty.status === 0 && bounty.poster.toLowerCase() !== account.toLowerCase()) {
    return `<button onclick="acceptBounty(${id})">接受任务</button>`;
  }
  if (bounty.status === 1 && bounty.poster.toLowerCase() === account.toLowerCase()) {
    return `<button onclick="completeBounty(${id})">确认完成</button>`;
  }
  return '';
}

document.getElementById('connect').addEventListener('click', connect);
document.getElementById('createBounty').addEventListener('click', createBounty);

// 暴露到 window 供内联 onclick 使用
window.acceptBounty = acceptBounty;
window.completeBounty = completeBounty;
```
