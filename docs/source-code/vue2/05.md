```js
// 初始化全局API的入口
export function initGlobalAPI(Vue) {
    // config
    const configDef = {}
    configDef.get = () => config
    if (process.env.NODE_ENV !== 'production') {
        // 开发环境，不允许使用 Vue.config = {} 这种方式，这是直接替换了Vue的整个默认配置
        // 可以使用 Vue.config.xxx = xxx 的方式， 添加属性
        configDef.set = () => {
            warn(
                'Do not replace the Vue.config object, set individual fields instead.'
            )
        }
    }

    // 给Vue全局属性初始化config对象
    Object.defineProperty(Vue, 'config', configDef)

    // exposed util methods.
    // NOTE: these are not considered part of the public API - avoid relying on
    // them unless you are aware of the risk.
    // Vue内部暴露的一些公共方法， 官方建议不要轻易使用，除非很清楚这些工具方法以及风险
    Vue.util = {
        // 警告日志
        warn,
        // 类似选项合并
        extend,
        // 合并选项
        mergeOptions,
        // 设置响应式
        defineReactive
    }

    Vue.set = set
    Vue.delete = del
    Vue.nextTick = nextTick

    // 2.6 explicit observable API
    // 响应式方法
    Vue.observable = (obj) => {
        observe(obj)
        return obj
    }

    // 初始化全局的 Vue.options.components 、 Vue.options.directives 、Vue.options.filters
    // 目前仅仅是初始化为一个没有原型的空对象
    Vue.options = Object.create(null)
    ASSET_TYPES.forEach(type => {
        Vue.options[type + 's'] = Object.create(null)
    })

    // this is used to identify the "base" constructor to extend all plain-object
    // components with in Weex's multi-instance scenarios.
    //  将 Vue 构造函数挂载到 Vue.options._base 上
    Vue.options._base = Vue

    extend(Vue.options.components, builtInComponents)

    // 初始化 Vue.use
    initUse(Vue)
    // 初始化 Vue.mixin
    initMixin(Vue)
    // 初始化 Vue.extend
    initExtend(Vue)
    // 初始化 Vue.component/directive/filter
    initAssetRegisters(Vue)
}
```

### set

```js
export function set (target, key, val) {
  // 兼容处理，set的目标对象不能是基础类型（number, string, symble undefined, null）
  if (process.env.NODE_ENV !== 'production' &&
    (isUndef(target) || isPrimitive(target))
  ) {
    warn(`Cannot set reactive property on undefined, null, or primitive value: ${(target)}`)
  }

  // 处理set 目标对象为数组的场景   this.$set(arr, index, 1)
  if (Array.isArray(target) && isValidArrayIndex(key)) {
    target.length = Math.max(target.length, key)
    // 利用数组的splice 实现 元素的插入/替换
    target.splice(key, 1, val)
    return val
  }

  // 处理对象的场景 this.$set({}, 'age', 12)
  if (key in target && !(key in Object.prototype)) {
    target[key] = val
    return val
  }
  const ob = (target).__ob__
  if (target._isVue || (ob && ob.vmCount)) {
    process.env.NODE_ENV !== 'production' && warn(
      'Avoid adding reactive properties to a Vue instance or its root $data ' +
      'at runtime - declare it upfront in the data option.'
    )
    return val
  }
  if (!ob) {
    target[key] = val
    return val
  }

  // 对新属性做 getter  setter 触发依赖收集等
  defineReactive(ob.value, key, val)
  ob.dep.notify()
  return val
}
```

### del

```js
export function del (target, key) {
  if (process.env.NODE_ENV !== 'production' &&
    (isUndef(target) || isPrimitive(target))
  ) {
    warn(`Cannot delete reactive property on undefined, null, or primitive value: ${(target: any)}`)
  }

  // 数组还是利用 splice删除 元素
  if (Array.isArray(target) && isValidArrayIndex(key)) {
    target.splice(key, 1)
    return
  }
  const ob = (target).__ob__
  if (target._isVue || (ob && ob.vmCount)) {
    process.env.NODE_ENV !== 'production' && warn(
      'Avoid deleting properties on a Vue instance or its root $data ' +
      '- just set it to null.'
    )
    return
  }
  // 如果删除一个在对象上不存在的key(属性)， 什么都不做
  if (!hasOwn(target, key)) {
    return
  }

  // 使用原生 delete 符号，删除对象属性
  delete target[key]

  // 如果这个对象不是一个响应式，则什么都不做， 响应式对象都有一个属性，__ob__
  if (!ob) {
    return
  }

  // 触发依赖通知更新
  ob.dep.notify()
}
```

```js
/**
 * 这里的nextTick 就是 this.$nextTick()
 * @param {*} cb
 * @param {*} ctx
 * @returns
 */
export function nextTick (cb?: Function, ctx?: Object) {
  let _resolve
  callbacks.push(() => {
    if (cb) {

      // 因为nextTick是提供给外部开发者可以直接调用，所以这里使用 try...catch 包装一层，保证程序的健壮性

      try {
        cb.call(ctx)
      } catch (e) {
        // vue内部检测到包装过的函数报错，就直接抛出异常
        handleError(e, ctx, 'nextTick')
      }
    } else if (_resolve) {
      _resolve(ctx)
    }
  })
  if (!pending) {
    pending = true
    timerFunc()
  }
  // $flow-disable-line
  if (!cb && typeof Promise !== 'undefined') {
    return new Promise(resolve => {
      _resolve = resolve
    })
  }
}
```

```js
export function initUse (Vue) {
  // 注册插件

  // Vue.use(element, {})
  Vue.use = function (plugin) {

    // 如果插件已经注册了，就返回 this, this指向的就是 Vue实例， 所以可以链式调用， 避免插件重复注册
    // Vue.use(APlugin).use(BPlugin)
    const installedPlugins = (this._installedPlugins || (this._installedPlugins = []))
    if (installedPlugins.indexOf(plugin) > -1) {
      return this
    }

    // additional parameters
    const args = toArray(arguments, 1)

    // 这里就是为啥  install 方法第一个参数是 Vue实例
    args.unshift(this)

    // 这就是插件需要提供的 install 方法
    // {install: (Vue) => {})}  一种是对象写法
    // (Vue) => {}  一种是直接传递一个函数。

    if (typeof plugin.install === 'function') {
      // plugin是对象形式
      plugin.install.apply(plugin, args)
    } else if (typeof plugin === 'function') {
      // plugin 是函数形式
      plugin.apply(null, args)
    }

    // 将插件添加到插件列表中
    installedPlugins.push(plugin)
    return this
  }
}
```

```js
import { mergeOptions } from '../util/index'

/**
 *  定义 Vue.mixin，负责全局混入选项，影响之后所有创建的 Vue 实例，这些实例会合并全局混入的选项
 * @param {*} Vue
 * @returns 返回 Vue 实例
 */
export function initMixin (Vue: GlobalAPI) {
  //  在 Vue 的默认配置项上合并 mixin 对象
  Vue.mixin = function (mixin: Object) {
    this.options = mergeOptions(this.options, mixin)
    return this
  }
}
```

#### mergeOptions

```js
export function mergeOptions (
  parent: Object,
  child: Object,
  vm?: Component
): Object {
  if (process.env.NODE_ENV !== 'production') {
    checkComponents(child)
  }

  if (typeof child === 'function') {
    child = child.options
  }

  // 选项的标准化处理
  normalizeProps(child, vm)
  normalizeInject(child, vm)
  normalizeDirectives(child)

  // Apply extends and mixins on the child options,
  // but only if it is a raw options object that isn't
  // the result of another mergeOptions call.
  // Only merged options has the _base property.

  // 递归合并选项
  if (!child._base) {
    if (child.extends) {
      parent = mergeOptions(parent, child.extends, vm)
    }

    // 存在mixin 将mixin中的选项和当前进行合并
    if (child.mixins) {
      for (let i = 0, l = child.mixins.length; i < l; i++) {
        parent = mergeOptions(parent, child.mixins[i], vm)
      }
    }
  }

  const options = {}
  let key
  for (key in parent) {
    mergeField(key)
  }

  // 如果存在子类的选项和父类冲突，重名，则子类覆盖父类
  for (key in child) {
    if (!hasOwn(parent, key)) {
      mergeField(key)
    }
  }
  function mergeField (key) {
    const strat = strats[key] || defaultStrat
    options[key] = strat(parent[key], child[key], vm, key)
  }
  return options
}
```
