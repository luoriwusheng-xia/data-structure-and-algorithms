#### 响应式处理

在初始化完 data 以后，会对数据进行响应式处理 `initData` 方法中 `observe(data, true)`

src/core/observer/index.js

```javascript
/***
 * @param {any} value 需要进行响应式处理的对象
 * @param {boolean} asRootData 是否是 Vue根实例中的 data
 */
export function observe(value, asRootData) {
    // 如果值不是对象或者只不是VNode实例， 则不需要转换，说明可能就传入的是一个 基础类型
    if (!isObject(value) || value instanceof VNode) {
        return
    }

    // ob表示一个响应式对象
    let ob

    // 如果value这个对象上有 __ob__ 私有属性， 则表示value已经是一个响应式对象,直接返回即可
    if (hasOwn(value, '__ob__') && value.__ob__ instanceof Observer) {
        ob = value.__ob__
    } else if (
        shouldObserve &&
        !isServerRendering() &&
        (Array.isArray(value) || isPlainObject(value)) &&
        Object.isExtensible(value) &&
        !value._isVue
    ) {

        // 实例化一个Observer 进行响应式处理
        ob = new Observer(value)
    }
    if (asRootData && ob) {
        ob.vmCount++
    }

    // 直接返回响应式对象
    return ob
}
```

响应式处理，本质是实例化了一个 `Observer` 实例 `(src/core/observer/index.js) 路径`

1. 使用 `Object.defineProperty` 给对象身上自定义了一个内部属性 `__ob__`, 如果存在这个属性，则表明该对应已经经过响应式处理， 设定为不可枚举

```javascript
// src/core/observer/index.js
def(value, '__ob__', this)

// src/core/util/lang.js
export function def(obj, key, val, enumerable) {
    Object.defineProperty(obj, key, {
        value: val,
        enumerable: !!enumerable,
        writable: true,
        configurable: true
    })
}
```

2. 判断传入的值是  `数组`还是`对象`

如果是对象，则递归调用该对象，执行 `defineReactive` 方法； 如果是数组， 则调用 `observeArray` 遍历执行 `observe`

3. 最终都是到 `defineReactive`

`defineReactive` 里面就是  初始化了一个 `dep` ， 一个 `属性key` 对应一个 `dep` 实例； 递归处理传入 `defineReactive` 中为对象的形式， 使用 ` Object.defineProperty` 对对象进行 劫持操作

#### 数组响应式处理、对象响应式处理

vue2不能对数组做响应式处理，通过重新方法实现劫持

```javascript
// src/core/util/env.js
export const hasProto = '__proto__' in {}

// src/core/observer/index.js
export class Observer {
    constructor(value) {
        // 数组
        if (Array.isArray(value)) {
            // 判断是否有 __proto__属性；__proto__  就是可以通过 obj.__proto__ 访问对象的原型链
            if (hasProto) {
                // 浏览器支持 __proto__ 这个私有属性  则直接拿到原型链上的方法， 对其进行劫持
                protoAugment(value, arrayMethods)
            } else {
                copyAugment(value, arrayMethods, arrayKeys)
            }

            this.observeArray(value)
        } else {
            // 处理对象响应式
            this.walk(value)
        }
    }
}


function protoAugment (target, src) {
  // 使用重写的数组原型方法 -覆盖默认的原型方法
  target.__proto__ = src
}


function copyAugment (target, src, keys) {
  for (let i = 0, l = keys.length; i < l; i++) {
    const key = keys[i]
    // target就是我们的目标数组， key就是  push. splice  , pop 等方法
    // def 就是使用 Object.definedProperty('list', 'push', {})  重新定义我们的方法， 去覆盖原型链上的方法
    def(target, key, src[key])
  }
}
```

`__proto__` 是一个非标准的属性，可以访问`原型链`, 对 数组的 几个方法进行劫持（重写）

```javascript
  'push',
  'pop',
  'shift',
  'unshift',
  'splice',
  'sort',
  'reverse'
```

### 问题

#### watch、computed、methods

* 本质都是 `Watcher`
* computed 默认是懒执行， 切不可通过配置改变这种行为， watch 可以配置
* 使用场景不一样， computed不能是`异步`方法， 也就是不能通过 `async` 或者 `setTimeout` 等返回结果， `watch` 可以

**场景**

* methods 一般用于封装一些较为复杂的逻辑（同步， 异步）
* computed 一般用于封装一些简单的同步逻辑， 将经过处理的数据返回， 然后显示在模版中，以减轻模版的重量（复杂度）
* watch 一般用于需要在数据变化时执行异步或者开销较大的操作

**区别**

- methods VS computed
> 如果在一次渲染中，有多个地方使用了同一个 methods 或者 computed属性， methods会执行多次， 而computed只执行一次

> 通过阅读源码， 在一次渲染中，多次访问 computed的属性， 值会在第一次执行computed 属性的回调函数，后续的其他访问，则直接使用第一次的执行结果（`watcher.value`） ， 而这一系的实现原理则是通过对 `watcher.dirty` 属性的控制实现的。 而methods，每一次的访问则是简单的方法调用（`this.xxxMethods()`）

**computed VS  watch**

> 通过阅读源码我们知道， computed 和watch 的本质是一样的，内部都是通过 Watcher 来实现的，其实没有什么区别，
> 1、使用场景上， computed是懒执行， 且不可更改


**methods vs watch**
> methods 和 watch 之间没有什么可比性， 完全是2个东西， 不过可以把watch中的逻辑抽离到 methods中，提高代码的可读性


#### computed，watch, methods 优先级

* `props`中的`key` 优先级 >  `methods` 中的 `key` > `data`中`key`的优先级

#### 1、bind

vue@2.6.14 版本中工具类实现了 bind的兼容写法

```javascript
/*
 * 原生支持
 */
function nativeBind(fn, ctx) {
    return fn.bind(ctx)
}

function polyfillBind(fn, ctx) {
    return boundFn(a) {
        const args = arguments.length

        // 有参数
        if (args) {
            return args > 1 ? fn.apply(ctx, arguments) : fn.call(ctx, a)
        } else {
            // 无参数
            return fn.call(ctx)
        }
    }

    boundFn._length = fn.length

    return boundFn
}

export const bind = Function.prototype.bind ? nativeBind : polyfillBind
```

应用： `src/core/instance/state.js` 中 `initMethods`
